111     void SettingsDialog::initTranslationsList()
112     {
113         const QString current = mTranslator->getCurrentLanguage();
114         QList<TranslationInfo> translations = mTranslator->getTranslations();
115         foreach (TranslationInfo translation, translations) {
116             QListWidgetItem *item = new QListWidgetItem;
117             item->setText(translation.mName);
118             item->setData(mLangCodeRole, QVariant(translation.mCode));
119             mUI.mListLanguages->addItem(item);
120             if (translation.mCode == current || translation.mCode == current.mid(0, 2))
121                 mUI.mListLanguages->setCurrentItem(item);
122         }
123     }

156     void SettingsDialog::saveSettingValues() const
157     {
158         int jobs = mUI.mJobs->text().toInt();
159         if (jobs <= 0) {
160             jobs = 1;
161         }
162     
163         QSettings settings;
164         settings.setValue(SETTINGS_CHECK_THREADS, jobs);
165         saveCheckboxValue(&settings, mUI.mForce, SETTINGS_CHECK_FORCE);
166         saveCheckboxValue(&settings, mUI.mSaveAllErrors, SETTINGS_SAVE_ALL_ERRORS);
167         saveCheckboxValue(&settings, mUI.mSaveFullPath, SETTINGS_SAVE_FULL_PATH);
168         saveCheckboxValue(&settings, mUI.mShowFullPath, SETTINGS_SHOW_FULL_PATH);
169         saveCheckboxValue(&settings, mUI.mShowNoErrorsMessage, SETTINGS_SHOW_NO_ERRORS);
170         saveCheckboxValue(&settings, mUI.mShowDebugWarnings, SETTINGS_SHOW_DEBUG_WARNINGS);
171         saveCheckboxValue(&settings, mUI.mInlineSuppressions, SETTINGS_INLINE_SUPPRESSIONS);
172         saveCheckboxValue(&settings, mUI.mEnableInconclusive, SETTINGS_INCONCLUSIVE_ERRORS);
173         saveCheckboxValue(&settings, mUI.mShowStatistics, SETTINGS_SHOW_STATISTICS);
174         saveCheckboxValue(&settings, mUI.mShowErrorId, SETTINGS_SHOW_ERROR_ID);
175         settings.setValue(SETTINGS_PYTHON_PATH, mUI.mEditPythonPath->text());
176         settings.setValue(SETTINGS_MISRA_FILE, mUI.mEditMisraFile->text());
177     
178     #ifdef Q_OS_WIN
179         settings.setValue(SETTINGS_CLANG_PATH, mUI.mEditClangPath->text());
180         settings.setValue(SETTINGS_VS_INCLUDE_PATHS, mUI.mEditVsIncludePaths->text());
181     #endif
182     
183         const QListWidgetItem *currentLang = mUI.mListLanguages->currentItem();
184         if (currentLang) {
185             const QString langcode = currentLang->data(mLangCodeRole).toString();
186             settings.setValue(SETTINGS_LANGUAGE, langcode);
187         }
188         CodeEditorStyle::saveSettings(&settings, *mCurrentStyle);
189     }

197     void SettingsDialog::addApplication()
198     {
199         Application app;
200         ApplicationDialog dialog(tr("Add a new application"), app, this);
201     
202         if (dialog.exec() == QDialog::Accepted) {
203             mTempApplications->addApplication(app);
204             mUI.mListWidget->addItem(app.getName());
205         }
206     }

208     void SettingsDialog::removeApplication()
209     {
210         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
211         foreach (QListWidgetItem *item, selected) {
212             const int removeIndex = mUI.mListWidget->row(item);
213             const int currentDefault = mTempApplications->getDefaultApplication();
214             mTempApplications->removeApplication(removeIndex);
215             if (removeIndex == currentDefault)
216                 // If default app is removed set default to unknown
217                 mTempApplications->setDefault(-1);
218             else if (removeIndex < currentDefault)
219                 // Move default app one up if earlier app was removed
220                 mTempApplications->setDefault(currentDefault - 1);
221         }
222         mUI.mListWidget->clear();
223         populateApplicationList();
224     }

226     void SettingsDialog::editApplication()
227     {
228         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
229         QListWidgetItem *item = nullptr;
230         foreach (item, selected) {
231             int row = mUI.mListWidget->row(item);
232             Application& app = mTempApplications->getApplication(row);
233             ApplicationDialog dialog(tr("Modify an application"), app, this);
234     
235             if (dialog.exec() == QDialog::Accepted) {
236                 QString name = app.getName();
237                 if (mTempApplications->getDefaultApplication() == row)
238                     name += tr(" [Default]");
239                 item->setText(name);
240             }
241         }
242     }
243     
244     void SettingsDialog::defaultApplication()
245     {
246         QList<QListWidgetItem *> selected = mUI.mListWidget->selectedItems();
247         if (!selected.isEmpty()) {
248             int index = mUI.mListWidget->row(selected[0]);
249             mTempApplications->setDefault(index);
250             mUI.mListWidget->clear();
251             populateApplicationList();
252         }
253     }
254     
255     void SettingsDialog::populateApplicationList()
256     {
257         const int defapp = mTempApplications->getDefaultApplication();
258         for (int i = 0; i < mTempApplications->getApplicationCount(); i++) {
259             const Application& app = mTempApplications->getApplication(i);
260             QString name = app.getName();
261             if (i == defapp) {
262                 name += " ";
263                 name += tr("[Default]");
264             }
265             mUI.mListWidget->addItem(name);
266         }
267     
268         // Select default application, or if there is no default app then the
269         // first item.
270         if (defapp == -1)
271             mUI.mListWidget->setCurrentRow(0);
272         else {
273             if (mTempApplications->getApplicationCount() > defapp)
274                 mUI.mListWidget->setCurrentRow(defapp);
275             else
276                 mUI.mListWidget->setCurrentRow(0);
277         }
278     }

330     // Slot to set default light style
331     void SettingsDialog::setCodeEditorStyleDefault()
332     {
333         if (mUI.mThemeSystem->isChecked())
334             *mCurrentStyle = CodeEditorStyle::getSystemTheme();
335         if (mUI.mThemeLight->isChecked())
336             *mCurrentStyle = defaultStyleLight;
337         if (mUI.mThemeDark->isChecked())
338             *mCurrentStyle = defaultStyleDark;
339         manageStyleControls();
340     }

342     // Slot to edit custom style
343     void SettingsDialog::editCodeEditorStyle()
344     {
345         StyleEditDialog dlg(*mCurrentStyle, this);
346         int nResult = dlg.exec();
347         if (nResult == QDialog::Accepted) {
348             *mCurrentStyle = dlg.getStyle();
349             manageStyleControls();
350         }
351     }

354     {
355         QString selectedDir = QFileDialog::getExistingDirectory(this,
356                               tr("Select clang path"),

361         }

364     void SettingsDialog::manageStyleControls()
365     {
366         bool isSystemTheme = mCurrentStyle->isSystemTheme();
367         bool isDefaultLight = !isSystemTheme && *mCurrentStyle == defaultStyleLight;
368         bool isDefaultDark =  !isSystemTheme && *mCurrentStyle == defaultStyleDark;
369         mUI.mThemeSystem->setChecked(isSystemTheme);
370         mUI.mThemeLight->setChecked(isDefaultLight && !isDefaultDark);
371         mUI.mThemeDark->setChecked(!isDefaultLight && isDefaultDark);
372         mUI.mThemeCustom->setChecked(!isSystemTheme && !isDefaultLight && !isDefaultDark);
373         mUI.mBtnEditTheme->setEnabled(!isSystemTheme && !isDefaultLight && !isDefaultDark);
374     }
