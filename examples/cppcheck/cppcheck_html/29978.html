19     #include <QObject>
20     #include "testxmlreportv2.h"
21     #include "xmlreportv2.h"
22     #include "erroritem.h"
23     #include "errorlogger.h"
24     
25     void TestXmlReportV2::readXml()
26     {
27         const QString filepath(QString(SRCDIR) + "/../data/xmlfiles/xmlreport_v2.xml");
28         XmlReportV2 report(filepath);
29         QVERIFY(report.open());
30         QList<ErrorItem> errors = report.read();
31         QCOMPARE(errors.size(), 6);
32     
33         const ErrorItem &item = errors[0];
34         QCOMPARE(item.errorPath.size(), 1);
35         QCOMPARE(item.errorPath[0].file, QString("test.cxx"));
36         QCOMPARE(item.errorPath[0].line, (unsigned int)11);
37         QCOMPARE(item.errorId, QString("unreadVariable"));
38         QCOMPARE(GuiSeverity::toString(item.severity), QString("style"));
39         QCOMPARE(item.summary, QString("Variable 'a' is assigned a value that is never used"));
40         QCOMPARE(item.message, QString("Variable 'a' is assigned a value that is never used"));
41     
42         const ErrorItem &item2 = errors[3];
43         QCOMPARE(item2.errorPath.size(), 2);
44         QCOMPARE(item2.errorPath[0].file, QString("test.cxx"));
45         QCOMPARE(item2.errorPath[0].line, (unsigned int)16);
46         QCOMPARE(item2.errorPath[1].file, QString("test.cxx"));
47         QCOMPARE(item2.errorPath[1].line, (unsigned int)32);
48         QCOMPARE(item2.errorId, QString("mismatchAllocDealloc"));
49         QCOMPARE(GuiSeverity::toString(item2.severity), QString("error"));
50         QCOMPARE(item2.summary, QString("Mismatching allocation and deallocation: k"));
51         QCOMPARE(item2.message, QString("Mismatching allocation and deallocation: k"));
52     }
53     
54     QTEST_MAIN(TestXmlReportV2)
