124     /**
125      * @brief Creates a string suitable for passing as the filter argument to
126      * methods like QFileDialog::getOpenFileName.
127      * @param filters A map of filter descriptions to the associated file name
128      * patterns.
129      * @param addAllSupported If set to true (the default), the function will
130      * include a filter entry containing all the file name patterns found in
131      * \p filters. This entry will be the first in the resulting filter string.
132      * @param addAll If set to true (the default), the function will
133      * include a filter entry displaying all files. This entry will be placed
134      * after the entry for \p addAllSupported files.
135      *
136      * Example usage:
137      *
138      * @code
139      * QMap<QString,QString> filters;
140      * filters[tr("Supported images")] = "*.bmp *.jpg *.png";
141      * filters[tr("Plain text")] = "*.txt";
142      *
143      * const QString filterString = toFilterString(filters);
144      *
145      * // filterString contains "All supported files (*.txt *.bmp *.jpg *.png);;All files (*.*);;Plain text (*.txt);;Supported images (*.bmp *.jpg *.png)"
146      * @endcode
147      */
148     QString toFilterString(const QMap<QString,QString>& filters, bool addAllSupported=true, bool addAll=true);
