35     void visitAstNodes(const Token *ast, std::function<ChildrenToVisit(const Token *)> visitor)
36     {
37         std::stack<const Token *> tokens;
38         tokens.push(ast);
39         while (!tokens.empty()) {
40             const Token *tok = tokens.top();
41             tokens.pop();
42             if (!tok)
43                 continue;
44     
45             ChildrenToVisit c = visitor(tok);
46     
47             if (c == ChildrenToVisit::done)
48                 break;
49             if (c == ChildrenToVisit::op1 || c == ChildrenToVisit::op1_and_op2)
50                 tokens.push(tok->astOperand1());
51             if (c == ChildrenToVisit::op2 || c == ChildrenToVisit::op1_and_op2)
52                 tokens.push(tok->astOperand2());
53         }
54     }

56     static void astFlattenRecursive(const Token *tok, std::vector<const Token *> *result, const char* op, nonneg int depth = 0)
57     {
58         ++depth;
59         if (!tok || depth >= 100)
60             return;
61         if (tok->str() == op) {
62             astFlattenRecursive(tok->astOperand1(), result, op, depth);
63             astFlattenRecursive(tok->astOperand2(), result, op, depth);
64         } else {
65             result->push_back(tok);
66         }
67     }

69     std::vector<const Token*> astFlatten(const Token* tok, const char* op)
70     {
71         std::vector<const Token*> result;
72         astFlattenRecursive(tok, &result, op);
73         return result;
74     }

183     const Token * astIsVariableComparison(const Token *tok, const std::string &comp, const std::string &rhs, const Token **vartok)
184     {
185         if (!tok)
186             return nullptr;
187     
188         const Token *ret = nullptr;
189         if (tok->isComparisonOp()) {
190             if (tok->astOperand1() && match(tok->astOperand1(), rhs)) {
191                 // Invert comparator
192                 std::string s = tok->str();
193                 if (s[0] == '>')
194                     s[0] = '<';
195                 else if (s[0] == '<')
196                     s[0] = '>';
197                 if (s == comp) {
198                     ret = tok->astOperand2();
199                 }
200             } else if (tok->str() == comp && tok->astOperand2() && match(tok->astOperand2(), rhs)) {
201                 ret = tok->astOperand1();
202             }
203         } else if (comp == "!=" && rhs == std::string("0")) {
204             ret = tok;
205         } else if (comp == "==" && rhs == std::string("0")) {
206             if (tok->str() == "!") {
207                 ret = tok->astOperand1();
208                 // handle (!(x!=0)) as (x==0)
209                 astIsVariableComparison(ret, "!=", "0", &ret);
210             }
211         }
212         while (ret && ret->str() == ".")
213             ret = ret->astOperand2();
214         if (ret && ret->str() == "=" && ret->astOperand1() && ret->astOperand1()->varId())
215             ret = ret->astOperand1();
216         else if (ret && ret->varId() == 0U)
217             ret = nullptr;
218         if (vartok)
219             *vartok = ret;
220         return ret;
221     }

243     const Token * nextAfterAstRightmostLeaf(const Token * tok)
244     {
245         const Token * rightmostLeaf = tok;
246         if (!rightmostLeaf || !rightmostLeaf->astOperand1())
247             return nullptr;
248         do {
249             if (rightmostLeaf->astOperand2())
250                 rightmostLeaf = rightmostLeaf->astOperand2();
251             else
252                 rightmostLeaf = rightmostLeaf->astOperand1();
253         } while (rightmostLeaf->astOperand1());
254         while (Token::Match(rightmostLeaf->next(), "]|)") && !hasToken(rightmostLeaf->next()->link(), rightmostLeaf->next(), tok))
255             rightmostLeaf = rightmostLeaf->next();
256         if (rightmostLeaf->str() == "{" && rightmostLeaf->link())
257             rightmostLeaf = rightmostLeaf->link();
258         return rightmostLeaf->next();
259     }

301     bool isAliasOf(const Token *tok, nonneg int varid)
302     {
303         if (tok->varId() == varid)
304             return false;
305         if (tok->varId() == 0)
306             return false;
307         for (const ValueFlow::Value &val : tok->values()) {
308             if (!val.isLocalLifetimeValue())
309                 continue;
310             if (val.isInconclusive())
311                 continue;
312             if (val.tokvalue->varId() == varid)
313                 return true;
314         }
315         return false;
316     }

318     static bool isAliased(const Token *startTok, const Token *endTok, nonneg int varid)
319     {
320         if (!precedes(startTok, endTok))
321             return false;
322         for (const Token *tok = startTok; tok != endTok; tok = tok->next()) {
323             if (Token::Match(tok, "= & %varid% ;", varid))
324                 return true;
325             if (isAliasOf(tok, varid))
326                 return true;
327         }
328         return false;
329     }

343     bool exprDependsOnThis(const Token* expr, nonneg int depth)
344     {
345         if (!expr)
346             return false;
347         if (depth >= 1000)
348             // Abort recursion to avoid stack overflow
349             return true;
350         ++depth;
351         // calling nonstatic method?
352         if (Token::Match(expr->previous(), "!!:: %name% (") && expr->function() && expr->function()->nestedIn && expr->function()->nestedIn->isClassOrStruct()) {
353             // is it a method of this?
354             const Scope *nestedIn = expr->scope()->functionOf;
355             if (nestedIn && nestedIn->function)
356                 nestedIn = nestedIn->function->token->scope();
357             while (nestedIn && nestedIn != expr->function()->nestedIn) {
358                 nestedIn = nestedIn->nestedIn;
359             }
360             return nestedIn == expr->function()->nestedIn;
361         } else if (Token::Match(expr, "%var%") && expr->variable()) {
362             const Variable* var = expr->variable();
363             return (var->isPrivate() || var->isPublic() || var->isProtected());
364         }
365         if (Token::simpleMatch(expr, "."))
366             return exprDependsOnThis(expr->astOperand1(), depth);
367         return exprDependsOnThis(expr->astOperand1(), depth) || exprDependsOnThis(expr->astOperand2(), depth);
368     }

370     /// This takes a token that refers to a variable and it will return the token
371     /// to the expression that the variable is assigned to. If its not valid to
372     /// make such substitution then it will return the original token.
373     static const Token * followVariableExpression(const Token * tok, bool cpp, const Token * end = nullptr)
374     {
375         if (!tok)
376             return tok;
377         // Skip following variables that is across multiple files
378         if (end && end->fileIndex() != tok->fileIndex())
379             return tok;
380         // Skip array access
381         if (Token::Match(tok, "%var% ["))
382             return tok;
383         // Skip pointer indirection
384         if (tok->astParent() && tok->isUnaryOp("*"))
385             return tok;
386         // Skip following variables if it is used in an assignment
387         if (Token::Match(tok->next(), "%assign%"))
388             return tok;
389         const Variable * var = tok->variable();
390         const Token * varTok = getVariableInitExpression(var);
391         if (!varTok)
392             return tok;
393         // Bailout. If variable value depends on value of "this".
394         if (exprDependsOnThis(varTok))
395             return tok;
396         // Skip array access
397         if (Token::simpleMatch(varTok, "["))
398             return tok;
399         if (var->isVolatile())
400             return tok;
401         if (!var->isLocal() && !var->isConst())
402             return tok;
403         if (var->isStatic() && !var->isConst())
404             return tok;
405         if (var->isArgument())
406             return tok;
407         const Token * lastTok = precedes(tok, end) ? end : tok;
408         // If this is in a loop then check if variables are modified in the entire scope
409         const Token * endToken = (isInLoopCondition(tok) || isInLoopCondition(varTok) || var->scope() != tok->scope()) ? var->scope()->bodyEnd : lastTok;
410         if (!var->isConst() && (!precedes(varTok, endToken) || isVariableChanged(varTok, endToken, tok->varId(), false, nullptr, cpp)))
411             return tok;
412         if (precedes(varTok, endToken) && isAliased(varTok, endToken, tok->varId()))
413             return tok;
414         // Start at beginning of initialization
415         const Token * startToken = varTok;
416         while (Token::Match(startToken, "%op%|.|(|{") && startToken->astOperand1())
417             startToken = startToken->astOperand1();
418         // Skip if the variable its referring to is modified
419         for (const Token * tok2 = startToken; tok2 != endToken; tok2 = tok2->next()) {
420             if (Token::simpleMatch(tok2, ";"))
421                 break;
422             if (tok2->astParent() && tok2->isUnaryOp("*"))
423                 return tok;
424             if (tok2->tokType() == Token::eIncDecOp ||
425                 tok2->isAssignmentOp() ||
426                 Token::Match(tok2, "%name% .|[|++|--|%assign%")) {
427                 return tok;
428             }
429     
430             if (const Variable * var2 = tok2->variable()) {
431                 if (!var2->scope())
432                     return tok;
433                 const Token * endToken2 = var2->scope() != tok->scope() ? var2->scope()->bodyEnd : endToken;
434                 if (!var2->isLocal() && !var2->isConst() && !var2->isArgument())
435                     return tok;
436                 if (var2->isStatic() && !var2->isConst())
437                     return tok;
438                 if (!var2->isConst() && (!precedes(tok2, endToken2) || isVariableChanged(tok2, endToken2, tok2->varId(), false, nullptr, cpp)))
439                     return tok;
440                 if (precedes(tok2, endToken2) && isAliased(tok2, endToken2, tok2->varId()))
441                     return tok;
442                 // Recognized as a variable but the declaration is unknown
443             } else if (tok2->varId() > 0) {
444                 return tok;
445             } else if (tok2->tokType() == Token::eName && !Token::Match(tok2, "sizeof|decltype|typeof") && !tok2->function()) {
446                 return tok;
447             }
448         }
449         return varTok;
450     }

452     static void followVariableExpressionError(const Token *tok1, const Token *tok2, ErrorPath* errors)
453     {
454         if (!errors)
455             return;
456         if (!tok1)
457             return;
458         if (!tok2)
459             return;
460         ErrorPathItem item = std::make_pair(tok2, "'" + tok1->str() + "' is assigned value '" + tok2->expressionString() + "' here.");
461         if (std::find(errors->begin(), errors->end(), item) != errors->end())
462             return;
463         errors->push_back(item);
464     }

496     bool isSameExpression(bool cpp, bool macro, const Token *tok1, const Token *tok2, const Library& library, bool pure, bool followVar, ErrorPath* errors)
497     {
498         if (tok1 == nullptr && tok2 == nullptr)
499             return true;
500         if (tok1 == nullptr || tok2 == nullptr)
501             return false;
502         if (cpp) {
503             if (tok1->str() == "." && tok1->astOperand1() && tok1->astOperand1()->str() == "this")
504                 tok1 = tok1->astOperand2();
505             if (tok2->str() == "." && tok2->astOperand1() && tok2->astOperand1()->str() == "this")
506                 tok2 = tok2->astOperand2();
507         }
508         // Skip double not
509         if (Token::simpleMatch(tok1, "!") && Token::simpleMatch(tok1->astOperand1(), "!") && !Token::simpleMatch(tok1->astParent(), "=")) {
510             return isSameExpression(cpp, macro, tok1->astOperand1()->astOperand1(), tok2, library, pure, followVar, errors);
511         }
512         if (Token::simpleMatch(tok2, "!") && Token::simpleMatch(tok2->astOperand1(), "!") && !Token::simpleMatch(tok2->astParent(), "=")) {
513             return isSameExpression(cpp, macro, tok1, tok2->astOperand1()->astOperand1(), library, pure, followVar, errors);
514         }
515         if (tok1->str() != tok2->str() && isDifferentKnownValues(tok1, tok2))
516             return false;
517         // Follow variable
518         if (followVar && tok1->str() != tok2->str() && (Token::Match(tok1, "%var%") || Token::Match(tok2, "%var%"))) {
519             const Token * varTok1 = followVariableExpression(tok1, cpp, tok2);
520             if (varTok1->str() == tok2->str()) {
521                 followVariableExpressionError(tok1, varTok1, errors);
522                 return isSameExpression(cpp, macro, varTok1, tok2, library, true, followVar, errors);
523             }
524             const Token * varTok2 = followVariableExpression(tok2, cpp, tok1);
525             if (tok1->str() == varTok2->str()) {
526                 followVariableExpressionError(tok2, varTok2, errors);
527                 return isSameExpression(cpp, macro, tok1, varTok2, library, true, followVar, errors);
528             }
529             if (varTok1->str() == varTok2->str()) {
530                 followVariableExpressionError(tok1, varTok1, errors);
531                 followVariableExpressionError(tok2, varTok2, errors);
532                 return isSameExpression(cpp, macro, varTok1, varTok2, library, true, followVar, errors);
533             }
534         }
535         if (tok1->varId() != tok2->varId() || tok1->str() != tok2->str() || tok1->originalName() != tok2->originalName()) {
536             if ((Token::Match(tok1,"<|>")   && Token::Match(tok2,"<|>")) ||
537                 (Token::Match(tok1,"<=|>=") && Token::Match(tok2,"<=|>="))) {
538                 return isSameExpression(cpp, macro, tok1->astOperand1(), tok2->astOperand2(), library, pure, followVar, errors) &&
539                        isSameExpression(cpp, macro, tok1->astOperand2(), tok2->astOperand1(), library, pure, followVar, errors);
540             }
541             return false;
542         }
543         if (macro && (tok1->isExpandedMacro() || tok2->isExpandedMacro() || tok1->isTemplateArg() || tok2->isTemplateArg()))
544             return false;
545         if (tok1->isComplex() != tok2->isComplex())
546             return false;
547         if (tok1->isLong() != tok2->isLong())
548             return false;
549         if (tok1->isUnsigned() != tok2->isUnsigned())
550             return false;
551         if (tok1->isSigned() != tok2->isSigned())
552             return false;
553         if (pure && tok1->isName() && tok1->next()->str() == "(" && tok1->str() != "sizeof") {
554             if (!tok1->function()) {
555                 if (Token::simpleMatch(tok1->previous(), ".")) {
556                     const Token *lhs = tok1->previous();
557                     while (Token::Match(lhs, "(|.|["))
558                         lhs = lhs->astOperand1();
559                     const bool lhsIsConst = (lhs->variable() && lhs->variable()->isConst()) ||
560                                             (lhs->valueType() && lhs->valueType()->constness > 0) ||
561                                             (Token::Match(lhs, "%var% . %name% (") && library.isFunctionConst(lhs->tokAt(2)));
562                     if (!lhsIsConst)
563                         return false;
564                 } else {
565                     const Token * ftok = tok1;
566                     if (Token::simpleMatch(tok1->previous(), "::"))
567                         ftok = tok1->previous();
568                     if (!library.isFunctionConst(ftok) && !ftok->isAttributeConst() && !ftok->isAttributePure())
569                         return false;
570                 }
571             } else {
572                 if (tok1->function() && !tok1->function()->isConst() && !tok1->function()->isAttributeConst() && !tok1->function()->isAttributePure())
573                     return false;
574             }
575         }
576         // templates/casts
577         if ((Token::Match(tok1, "%name% <") && tok1->next()->link()) ||
578             (Token::Match(tok2, "%name% <") && tok2->next()->link())) {
579     
580             // non-const template function that is not a dynamic_cast => return false
581             if (pure && Token::simpleMatch(tok1->next()->link(), "> (") &&
582                 !(tok1->function() && tok1->function()->isConst()) &&
583                 tok1->str() != "dynamic_cast")
584                 return false;
585     
586             // some template/cast stuff.. check that the template arguments are same
587             const Token *t1 = tok1->next();
588             const Token *t2 = tok2->next();
589             const Token *end1 = t1->link();
590             const Token *end2 = t2->link();
591             while (t1 && t2 && t1 != end1 && t2 != end2) {
592                 if (t1->str() != t2->str())
593                     return false;
594                 t1 = t1->next();
595                 t2 = t2->next();
596             }
597             if (t1 != end1 || t2 != end2)
598                 return false;
599         }
600         if (tok1->tokType() == Token::eIncDecOp || tok1->isAssignmentOp())
601             return false;
602         // bailout when we see ({..})
603         if (tok1->str() == "{")
604             return false;
605         // cast => assert that the casts are equal
606         if (tok1->str() == "(" && tok1->previous() &&
607             !tok1->previous()->isName() &&
608             !(tok1->previous()->str() == ">" && tok1->previous()->link())) {
609             const Token *t1 = tok1->next();
610             const Token *t2 = tok2->next();
611             while (t1 && t2 &&
612                    t1->str() == t2->str() &&
613                    t1->isLong() == t2->isLong() &&
614                    t1->isUnsigned() == t2->isUnsigned() &&
615                    t1->isSigned() == t2->isSigned() &&
616                    (t1->isName() || t1->str() == "*")) {
617                 t1 = t1->next();
618                 t2 = t2->next();
619             }
620             if (!t1 || !t2 || t1->str() != ")" || t2->str() != ")")
621                 return false;
622         }
623         bool noncommutativeEquals =
624             isSameExpression(cpp, macro, tok1->astOperand1(), tok2->astOperand1(), library, pure, followVar, errors);
625         noncommutativeEquals = noncommutativeEquals &&
626                                isSameExpression(cpp, macro, tok1->astOperand2(), tok2->astOperand2(), library, pure, followVar, errors);
627     
628         if (noncommutativeEquals)
629             return true;
630     
631         // in c++, a+b might be different to b+a, depending on the type of a and b
632         if (cpp && tok1->str() == "+" && tok1->isBinaryOp()) {
633             const ValueType* vt1 = tok1->astOperand1()->valueType();
634             const ValueType* vt2 = tok1->astOperand2()->valueType();
635             if (!(vt1 && (vt1->type >= ValueType::VOID || vt1->pointer) && vt2 && (vt2->type >= ValueType::VOID || vt2->pointer)))
636                 return false;
637         }
638     
639         const bool commutative = tok1->isBinaryOp() && Token::Match(tok1, "%or%|%oror%|+|*|&|&&|^|==|!=");
640         bool commutativeEquals = commutative &&
641                                  isSameExpression(cpp, macro, tok1->astOperand2(), tok2->astOperand1(), library, pure, followVar, errors);
642         commutativeEquals = commutativeEquals &&
643                             isSameExpression(cpp, macro, tok1->astOperand1(), tok2->astOperand2(), library, pure, followVar, errors);
644     
645     
646         return commutativeEquals;
647     }

667     bool isOppositeCond(bool isNot, bool cpp, const Token * const cond1, const Token * const cond2, const Library& library, bool pure, bool followVar, ErrorPath* errors)
668     {
669         if (!cond1 || !cond2)
670             return false;
671     
672         if (cond1->str() == "!") {
673             if (cond2->str() == "!=") {
674                 if (cond2->astOperand1() && cond2->astOperand1()->str() == "0")
675                     return isSameExpression(cpp, true, cond1->astOperand1(), cond2->astOperand2(), library, pure, followVar, errors);
676                 if (cond2->astOperand2() && cond2->astOperand2()->str() == "0")
677                     return isSameExpression(cpp, true, cond1->astOperand1(), cond2->astOperand1(), library, pure, followVar, errors);
678             }
679             return isSameExpression(cpp, true, cond1->astOperand1(), cond2, library, pure, followVar, errors);
680         }
681     
682         if (cond2->str() == "!")
683             return isOppositeCond(isNot, cpp, cond2, cond1, library, pure, followVar, errors);
684     
685         if (!isNot) {
686             if (cond1->str() == "==" && cond2->str() == "==") {
687                 if (isSameExpression(cpp, true, cond1->astOperand1(), cond2->astOperand1(), library, pure, followVar, errors))
688                     return isDifferentKnownValues(cond1->astOperand2(), cond2->astOperand2());
689                 if (isSameExpression(cpp, true, cond1->astOperand2(), cond2->astOperand2(), library, pure, followVar, errors))
690                     return isDifferentKnownValues(cond1->astOperand1(), cond2->astOperand1());
691             }
692             // TODO: Handle reverse conditions
693             if (Library::isContainerYield(cond1, Library::Container::Yield::EMPTY, "empty") &&
694                 Library::isContainerYield(cond2->astOperand1(), Library::Container::Yield::SIZE, "size") &&
695                 cond1->astOperand1()->astOperand1()->varId() == cond2->astOperand1()->astOperand1()->astOperand1()->varId()) {
696                 return !isZeroBoundCond(cond2);
697             }
698     
699             if (Library::isContainerYield(cond2, Library::Container::Yield::EMPTY, "empty") &&
700                 Library::isContainerYield(cond1->astOperand1(), Library::Container::Yield::SIZE, "size") &&
701                 cond2->astOperand1()->astOperand1()->varId() == cond1->astOperand1()->astOperand1()->astOperand1()->varId()) {
702                 return !isZeroBoundCond(cond1);
703             }
704         }
705     
706     
707         if (!cond1->isComparisonOp() || !cond2->isComparisonOp())
708             return false;
709     
710         const std::string &comp1 = cond1->str();
711     
712         // condition found .. get comparator
713         std::string comp2;
714         if (isSameExpression(cpp, true, cond1->astOperand1(), cond2->astOperand1(), library, pure, followVar, errors) &&
715             isSameExpression(cpp, true, cond1->astOperand2(), cond2->astOperand2(), library, pure, followVar, errors)) {
716             comp2 = cond2->str();
717         } else if (isSameExpression(cpp, true, cond1->astOperand1(), cond2->astOperand2(), library, pure, followVar, errors) &&
718                    isSameExpression(cpp, true, cond1->astOperand2(), cond2->astOperand1(), library, pure, followVar, errors)) {
719             comp2 = cond2->str();
720             if (comp2[0] == '>')
721                 comp2[0] = '<';
722             else if (comp2[0] == '<')
723                 comp2[0] = '>';
724         }
725     
726         if (!isNot && comp2.empty()) {
727             const Token *expr1 = nullptr, *value1 = nullptr, *expr2 = nullptr, *value2 = nullptr;
728             std::string op1 = cond1->str(), op2 = cond2->str();
729             if (cond1->astOperand2()->hasKnownIntValue()) {
730                 expr1 = cond1->astOperand1();
731                 value1 = cond1->astOperand2();
732             } else if (cond1->astOperand1()->hasKnownIntValue()) {
733                 expr1 = cond1->astOperand2();
734                 value1 = cond1->astOperand1();
735                 if (op1[0] == '>')
736                     op1[0] = '<';
737                 else if (op1[0] == '<')
738                     op1[0] = '>';
739             }
740             if (cond2->astOperand2()->hasKnownIntValue()) {
741                 expr2 = cond2->astOperand1();
742                 value2 = cond2->astOperand2();
743             } else if (cond2->astOperand1()->hasKnownIntValue()) {
744                 expr2 = cond2->astOperand2();
745                 value2 = cond2->astOperand1();
746                 if (op2[0] == '>')
747                     op2[0] = '<';
748                 else if (op2[0] == '<')
749                     op2[0] = '>';
750             }
751             if (!expr1 || !value1 || !expr2 || !value2) {
752                 return false;
753             }
754             if (!isSameExpression(cpp, true, expr1, expr2, library, pure, followVar, errors))
755                 return false;
756     
757             const ValueFlow::Value &rhsValue1 = value1->values().front();
758             const ValueFlow::Value &rhsValue2 = value2->values().front();
759     
760             if (op1 == "<" || op1 == "<=")
761                 return (op2 == "==" || op2 == ">" || op2 == ">=") && (rhsValue1.intvalue < rhsValue2.intvalue);
762             else if (op1 == ">=" || op1 == ">")
763                 return (op2 == "==" || op2 == "<" || op2 == "<=") && (rhsValue1.intvalue > rhsValue2.intvalue);
764     
765             return false;
766         }
767     
768         // is condition opposite?
769         return ((comp1 == "==" && comp2 == "!=") ||
770                 (comp1 == "!=" && comp2 == "==") ||
771                 (comp1 == "<"  && comp2 == ">=") ||
772                 (comp1 == "<=" && comp2 == ">") ||
773                 (comp1 == ">"  && comp2 == "<=") ||
774                 (comp1 == ">=" && comp2 == "<") ||
775                 (!isNot && ((comp1 == "<" && comp2 == ">") ||
776                             (comp1 == ">" && comp2 == "<") ||
777                             (comp1 == "==" && (comp2 == "!=" || comp2 == ">" || comp2 == "<")) ||
778                             ((comp1 == "!=" || comp1 == ">" || comp1 == "<") && comp2 == "==")
779                            )));
780     }

797         if (!tok)

831     bool isUniqueExpression(const Token* tok)
832     {
833         if (!tok)
834             return true;
835         if (tok->function()) {
836             const Function * fun = tok->function();
837             const Scope * scope = fun->nestedIn;
838             if (!scope)
839                 return true;
840             const std::string returnType = fun->retType ? fun->retType->name() : fun->retDef->stringifyList(fun->tokenDef);
841             for (const Function& f:scope->functionList) {
842                 if (f.type != Function::eFunction)
843                     continue;
844     
845                 const std::string freturnType = f.retType ? f.retType->name() : f.retDef->stringifyList(f.tokenDef);
846                 if (f.argumentList.size() == fun->argumentList.size() &&
847                     returnType == freturnType &&
848                     f.name() != fun->name()) {
849                     return false;
850                 }
851             }
852         } else if (tok->variable()) {
853             const Variable * var = tok->variable();
854             const Scope * scope = var->scope();
855             if (!scope)
856                 return true;
857             const Type * varType = var->type();
858             // Iterate over the variables in scope and the parameters of the function if possible
859             const Function * fun = scope->function;
860             const std::list<Variable>* setOfVars[] = {&scope->varlist, fun ? &fun->argumentList : nullptr};
861     
862             for (const std::list<Variable>* vars:setOfVars) {
863                 if (!vars)
864                     continue;
865                 bool other = std::any_of(vars->cbegin(), vars->cend(), [=](const Variable &v) {
866                     if (varType)
867                         return v.type() && v.type()->name() == varType->name() && v.name() != var->name();
868                     return v.isFloatingType() == var->isFloatingType() &&
869                            v.isEnumType() == var->isEnumType() &&
870                            v.isClass() == var->isClass() &&
871                            v.isArray() == var->isArray() &&
872                            v.isPointer() == var->isPointer() &&
873                            v.name() != var->name();
874                 });
875                 if (other)
876                     return false;
877             }
878         } else if (!isUniqueExpression(tok->astOperand1())) {
879             return false;
880         }
881     
882         return isUniqueExpression(tok->astOperand2());
883     }

901     bool isReturnScope(const Token * const endToken, const Settings * settings, bool functionScope)
902     {
903         if (!endToken || endToken->str() != "}")
904             return false;
905     
906         const Token *prev = endToken->previous();
907         while (prev && Token::simpleMatch(prev->previous(), "; ;"))
908             prev = prev->previous();
909         if (prev && Token::simpleMatch(prev->previous(), "} ;"))
910             prev = prev->previous();
911     
912         if (Token::simpleMatch(prev, "}")) {
913             if (Token::simpleMatch(prev->link()->tokAt(-2), "} else {"))
914                 return isReturnScope(prev, settings, functionScope) && isReturnScope(prev->link()->tokAt(-2), settings, functionScope);
915             if (Token::simpleMatch(prev->link()->previous(), ") {") &&
916                 Token::simpleMatch(prev->link()->linkAt(-1)->previous(), "switch (") &&
917                 !Token::findsimplematch(prev->link(), "break", prev)) {
918                 return true;
919             }
920             if (isEscaped(prev->link()->astTop(), functionScope))
921                 return true;
922             if (Token::Match(prev->link()->previous(), "[;{}] {"))
923                 return isReturnScope(prev, settings, functionScope);
924         } else if (Token::simpleMatch(prev, ";")) {
925             if (Token::simpleMatch(prev->previous(), ") ;") && Token::Match(prev->linkAt(-1)->tokAt(-2), "[;{}] %name% (")) {
926                 const Token * ftok = prev->linkAt(-1)->previous();
927                 const Function * function = ftok->function();
928                 if (function) {
929                     if (function->isEscapeFunction())
930                         return true;
931                     if (function->isAttributeNoreturn())
932                         return true;
933                 } else if (settings) {
934                     if (settings->library.isnoreturn(ftok))
935                         return true;
936                 }
937                 return false;
938             }
939             if (Token::simpleMatch(prev->previous(), ") ;") && prev->previous()->link() &&
940                 isEscaped(prev->previous()->link()->astTop(), functionScope))
941                 return true;
942             if (isEscaped(prev->previous()->astTop(), functionScope))
943                 return true;
944             // return/goto statement
945             prev = prev->previous();
946             while (prev && !Token::Match(prev, ";|{|}") && !isEscapedOrJump(prev, functionScope))
947                 prev = prev->previous();
948             return prev && prev->isName();
949         }
950         return false;
951     }

976     const Token * getTokenArgumentFunction(const Token * tok, int& argn)
977     {
978         argn = -1;
979         {
980             const Token *parent = tok->astParent();
981             if (parent && parent->isUnaryOp("&"))
982                 parent = parent->astParent();
983             while (parent && parent->isCast())
984                 parent = parent->astParent();
985     
986             // passing variable to subfunction?
987             if (Token::Match(parent, "[(,{]"))
988                 ;
989             else if (Token::simpleMatch(parent, ":")) {
990                 while (Token::Match(parent, "[?:]"))
991                     parent = parent->astParent();
992                 while (Token::simpleMatch(parent, ","))
993                     parent = parent->astParent();
994                 if (!parent || parent->str() != "(")
995                     return nullptr;
996             } else
997                 return nullptr;
998         }
999     
1000         // goto start of function call and get argn
1001         argn = 0;
1002         while (tok && !Token::simpleMatch(tok, ";") && !isScopeBracket(tok)) {
1003             if (tok->str() == ",")
1004                 ++argn;
1005             else if (Token::Match(tok, ")|}"))
1006                 tok = tok->link();
1007             else if (Token::Match(tok->previous(), "%name% (|{"))
1008                 break;
1009             else if (Token::Match(tok->previous(), "> (|{") && tok->previous()->link())
1010                 break;
1011             tok = tok->previous();
1012         }
1013         if (!Token::Match(tok, "{|("))
1014             return nullptr;
1015         tok = tok->previous();
1016         if (tok && tok->link() && tok->str() == ">")
1017             tok = tok->link()->previous();
1018         if (!Token::Match(tok, "%name% [({<]"))
1019             return nullptr;
1020         return tok;
1021     }

1023     bool isVariableChangedByFunctionCall(const Token *tok, int indirect, const Settings *settings, bool *inconclusive)
1024     {
1025         if (!tok)
1026             return false;
1027     
1028         const Token * const tok1 = tok;
1029     
1030         // address of variable
1031         const bool addressOf = tok->astParent() && tok->astParent()->isUnaryOp("&");
1032     
1033         int argnr;
1034         tok = getTokenArgumentFunction(tok, argnr);
1035         if (!tok)
1036             return false; // not a function => variable not changed
1037         const Token * parenTok = tok->next();
1038         if (Token::simpleMatch(parenTok, "<") && parenTok->link())
1039             parenTok = parenTok->link()->next();
1040         const bool possiblyPassedByReference = (parenTok->next() == tok1 || Token::Match(tok1->previous(), ", %name% [,)}]"));
1041     
1042         // Constructor call
1043         if (tok->variable() && tok->variable()->nameToken() == tok) {
1044             // Find constructor..
1045             const int argCount = numberOfArguments(tok);
1046             const Scope *typeScope = tok->variable()->typeScope();
1047             if (typeScope) {
1048                 for (const Function &function : typeScope->functionList) {
1049                     if (!function.isConstructor() || function.argCount() < argCount)
1050                         continue;
1051                     const Variable *arg = function.getArgumentVar(argnr);
1052                     if (arg && arg->isReference() && !arg->isConst())
1053                         return true;
1054                 }
1055                 return false;
1056             }
1057             if (inconclusive)
1058                 *inconclusive = true;
1059             return false;
1060         }
1061     
1062         if (!tok->function()) {
1063             // Check if direction (in, out, inout) is specified in the library configuration and use that
1064             if (!addressOf && settings) {
1065                 const Library::ArgumentChecks::Direction argDirection = settings->library.getArgDirection(tok, 1 + argnr);
1066                 if (argDirection == Library::ArgumentChecks::Direction::DIR_IN)
1067                     return false;
1068                 else if (argDirection == Library::ArgumentChecks::Direction::DIR_OUT ||
1069                          argDirection == Library::ArgumentChecks::Direction::DIR_INOUT) {
1070                     // With out or inout the direction of the content is specified, not a pointer itself, so ignore pointers for now
1071                     const ValueType * const valueType = tok1->valueType();
1072                     if (valueType && valueType->pointer == indirect) {
1073                         return true;
1074                     }
1075                 }
1076             }
1077     
1078             // if the library says 0 is invalid
1079             // => it is assumed that parameter is an in parameter (TODO: this is a bad heuristic)
1080             if (!addressOf && settings && settings->library.isnullargbad(tok, 1+argnr))
1081                 return false;
1082             // possible pass-by-reference => inconclusive
1083             if (possiblyPassedByReference) {
1084                 if (inconclusive != nullptr)
1085                     *inconclusive = true;
1086                 return false;
1087             }
1088             // Safe guess: Assume that parameter is changed by function call
1089             return true;
1090         }
1091     
1092         const Variable *arg = tok->function()->getArgumentVar(argnr);
1093     
1094         if (addressOf || (indirect > 0 && arg && arg->isPointer())) {
1095             if (!(arg && arg->isConst()))
1096                 return true;
1097             // If const is applied to the pointer, then the value can still be modified
1098             if (arg && Token::simpleMatch(arg->typeEndToken(), "* const"))
1099                 return true;
1100         }
1101     
1102         return arg && !arg->isConst() && arg->isReference();
1103     }

1105     bool isVariableChanged(const Token *tok, int indirect, const Settings *settings, bool cpp, int depth)
1106     {
1107         if (!tok)
1108             return false;
1109         const Token *tok2 = tok;
1110         while (Token::simpleMatch(tok2->astParent(), "*") || (Token::simpleMatch(tok2->astParent(), ".") && !Token::simpleMatch(tok2->astParent()->astParent(), "(")) ||
1111                (Token::simpleMatch(tok2->astParent(), "[") && tok2 == tok2->astParent()->astOperand1()))
1112             tok2 = tok2->astParent();
1113     
1114         while (Token::simpleMatch(tok2->astParent(), "?") || (Token::simpleMatch(tok2->astParent(), ":") && Token::simpleMatch(tok2->astParent()->astParent(), "?")))
1115             tok2 = tok2->astParent();
1116     
1117         if (Token::Match(tok2->astParent(), "++|--"))
1118             return true;
1119     
1120         if (tok2->astParent() && tok2->astParent()->isAssignmentOp()) {
1121             if (tok2 == tok2->astParent()->astOperand1())
1122                 return true;
1123             // Check if assigning to a non-const lvalue
1124             const Variable * var = getLHSVariable(tok2->astParent());
1125             if (var && var->isReference() && !var->isConst() && var->nameToken() && var->nameToken()->next() == tok2->astParent()) {
1126                 if (!var->isLocal() || isVariableChanged(var, settings, cpp, depth - 1))
1127                     return true;
1128             }
1129         }
1130     
1131         if (isLikelyStreamRead(cpp, tok->previous()))
1132             return true;
1133     
1134         if (isLikelyStream(cpp, tok2))
1135             return true;
1136     
1137         // Member function call
1138         if (tok->variable() && Token::Match(tok2->astParent(), ". %name%") && isFunctionCall(tok2->astParent()->next()) && tok2->astParent()->astOperand1() == tok2) {
1139             const Variable * var = tok->variable();
1140             bool isConst = var && var->isConst();
1141             if (!isConst && var) {
1142                 const ValueType * valueType = var->valueType();
1143                 isConst = (valueType && valueType->pointer == 1 && valueType->constness == 1);
1144             }
1145     
1146             const Token *ftok = tok->tokAt(2);
1147             const Function * fun = ftok->function();
1148             if (!isConst && (!fun || !fun->isConst()))
1149                 return true;
1150             else
1151                 return false;
1152         }
1153     
1154         const Token *ftok = tok2;
1155         while (ftok && (!Token::Match(ftok, "[({]") || ftok->isCast()))
1156             ftok = ftok->astParent();
1157     
1158         if (ftok && Token::Match(ftok->link(), ")|} !!{")) {
1159             const Token * ptok = tok2;
1160             while (Token::Match(ptok->astParent(), ".|::|["))

1498     PathAnalysis::Progress PathAnalysis::forwardRange(const Token* startToken, const Token* endToken, Info info, const std::function<PathAnalysis::Progress(const Info&)>& f) const
1499     {
1500         for (const Token *tok = startToken; tok && tok != endToken; tok = tok->next()) {
1501             if (Token::Match(tok, "asm|goto|break|continue"))
1502                 return Progress::Break;
1503             if (Token::Match(tok, "return|throw")) {
1504                 forwardRecursive(tok, info, f);
1505                 return Progress::Break;
1506             }
1507             if (Token::simpleMatch(tok, "}") && Token::simpleMatch(tok->link()->previous(), ") {") && Token::Match(tok->link()->linkAt(-1)->previous(), "if|while|for (")) {
1508                 const Token * blockStart = tok->link()->linkAt(-1)->previous();
1509                 const Token * condTok = getCondTok(blockStart);
1510                 if (!condTok)
1511                     continue;
1512                 info.errorPath.emplace_back(condTok, "Assuming condition is true.");
1513                 // Traverse a loop a second time
1514                 if (Token::Match(blockStart, "for|while (")) {
1515                     const Token* endCond = blockStart->linkAt(1);
1516                     bool traverseLoop = true;
1517                     // Only traverse simple for loops
1518                     if (Token::simpleMatch(blockStart, "for") && !Token::Match(endCond->tokAt(-3), "; ++|--|%var% %var%|++|-- ) {"))
1519                         traverseLoop = false;
1520                     // Traverse loop a second time
1521                     if (traverseLoop) {
1522                         // Traverse condition
1523                         if (forwardRecursive(condTok, info, f) == Progress::Break)
1524                             return Progress::Break;
1525                         // TODO: Should we traverse the body: forwardRange(tok->link(), tok, info, f)?
1526                     }
1527                 }
1528             }
1529             if (Token::Match(tok, "if|while|for (") && Token::simpleMatch(tok->next()->link(), ") {")) {
1530                 const Token * endCond = tok->next()->link();
1531                 const Token * endBlock = endCond->next()->link();
1532                 const Token * condTok = getCondTok(tok);
1533                 if (!condTok)
1534                     continue;
1535                 // Traverse condition
1536                 if (forwardRange(tok->next(), tok->next()->link(), info, f) == Progress::Break)
1537                     return Progress::Break;
1538                 Info i = info;
1539                 i.known = false;
1540                 i.errorPath.emplace_back(condTok, "Assuming condition is true.");
1541     
1542                 // Check if condition is true or false
1543                 bool checkThen = false;
1544                 bool checkElse = false;
1545                 std::tie(checkThen, checkElse) = checkCond(condTok, i.known);
1546     
1547                 // Traverse then block
1548                 if (checkThen) {
1549                     if (forwardRange(endCond->next(), endBlock, i, f) == Progress::Break)
1550                         return Progress::Break;
1551                 }
1552                 // Traverse else block
1553                 if (Token::simpleMatch(endBlock, "} else {")) {
1554                     if (checkElse) {
1555                         i.errorPath.back().second = "Assuming condition is false.";
1556                         Progress result = forwardRange(endCond->next(), endBlock, i, f);
1557                         if (result == Progress::Break)
1558                             return Progress::Break;
1559                     }
1560                     tok = endBlock->linkAt(2);
1561                 } else {
1562                     tok = endBlock;
1563                 }
1564             } else if (Token::simpleMatch(tok, "} else {")) {
1565                 tok = tok->linkAt(2);
1566             } else {
1567                 info.tok = tok;
1568                 if (f(info) == Progress::Break)
1569                     return Progress::Break;
1570             }
1571             // Prevent infinite recursion
1572             if (tok->next() == start)
1573                 break;
1574         }
1575         return Progress::Continue;
1576     }

1625     struct FwdAnalysis::Result FwdAnalysis::checkRecursive(const Token *expr, const Token *startToken, const Token *endToken, const std::set<int> &exprVarIds, bool local, bool inInnerClass)
1626     {
1627         // Parse the given tokens
1628         for (const Token *tok = startToken; tok != endToken; tok = tok->next()) {
1629             if (Token::simpleMatch(tok, "try {")) {
1630                 // TODO: handle try
1631                 return Result(Result::Type::BAILOUT);
1632             }
1633     
1634             if (Token::simpleMatch(tok, "break ;")) {
1635                 return Result(Result::Type::BREAK, tok);
1636             }
1637     
1638             if (Token::simpleMatch(tok, "goto"))
1639                 return Result(Result::Type::BAILOUT);
1640     
1641             if (!inInnerClass && tok->str() == "{" && tok->scope()->isClassOrStruct()) {
1642                 // skip returns from local class definition
1643                 FwdAnalysis::Result result = checkRecursive(expr, tok, tok->link(), exprVarIds, local, true);
1644                 if (result.type != Result::Type::NONE)
1645                     return result;
1646                 tok=tok->link();
1647             }
1648     
1649             if (tok->str() == "continue")
1650                 // TODO
1651                 return Result(Result::Type::BAILOUT);
1652     
1653             if (const Token *lambdaEndToken = findLambdaEndToken(tok)) {
1654                 tok = lambdaEndToken;
1655                 const Result lambdaResult = checkRecursive(expr, lambdaEndToken->link()->next(), lambdaEndToken, exprVarIds, local, inInnerClass);
1656                 if (lambdaResult.type == Result::Type::READ || lambdaResult.type == Result::Type::BAILOUT)
1657                     return lambdaResult;
1658             }
1659     
1660             if (Token::Match(tok, "return|throw")) {
1661                 // TODO: Handle these better
1662                 // Is expr variable used in expression?
1663                 const Token *end = tok->findExpressionStartEndTokens().second->next();
1664                 for (const Token *tok2 = tok; tok2 != end; tok2 = tok2->next()) {
1665                     if (!local && Token::Match(tok2, "%name% ("))
1666                         return Result(Result::Type::READ);
1667                     if (tok2->varId() && exprVarIds.find(tok2->varId()) != exprVarIds.end())
1668                         return Result(Result::Type::READ);
1669                 }
1670     
1671                 // #9167: if the return is inside an inner class, it does not tell us anything
1672                 if (!inInnerClass) {
1673                     if (!local && mWhat == What::Reassign)
1674                         return Result(Result::Type::BAILOUT);
1675     
1676                     return Result(Result::Type::RETURN);
1677                 }
1678             }
1679     
1680             if (tok->str() == "}") {
1681                 // Known value => possible value
1682                 if (tok->scope() == expr->scope())
1683                     mValueFlowKnown = false;
1684     
1685                 Scope::ScopeType scopeType = tok->scope()->type;
1686                 if (scopeType == Scope::eWhile || scopeType == Scope::eFor || scopeType == Scope::eDo) {
1687                     // check condition
1688                     const Token *conditionStart = nullptr;
1689                     const Token *conditionEnd = nullptr;
1690                     if (Token::simpleMatch(tok->link()->previous(), ") {")) {
1691                         conditionEnd = tok->link()->previous();
1692                         conditionStart = conditionEnd->link();
1693                     } else if (Token::simpleMatch(tok->link()->previous(), "do {") && Token::simpleMatch(tok, "} while (")) {
1694                         conditionStart = tok->tokAt(2);
1695                         conditionEnd = conditionStart->link();
1696                     }
1697                     if (conditionStart && conditionEnd) {
1698                         bool used = false;
1699                         for (const Token *condTok = conditionStart; condTok != conditionEnd; condTok = condTok->next()) {
1700                             if (exprVarIds.find(condTok->varId()) != exprVarIds.end())
1701                                 used = true;
1702                         }
1703                         if (used)
1704                             return Result(Result::Type::BAILOUT);
1705                     }
1706     
1707                     // check loop body again..
1708                     const struct FwdAnalysis::Result &result = checkRecursive(expr, tok->link(), tok, exprVarIds, local, inInnerClass);
1709                     if (result.type == Result::Type::BAILOUT || result.type == Result::Type::READ)
1710                         return result;
1711                 }
1712             }
1713     
1714             if (Token::simpleMatch(tok, "else {"))
1715                 tok = tok->linkAt(1);
1716     
1717             if (Token::simpleMatch(tok, "asm ("))
1718                 return Result(Result::Type::BAILOUT);
1719     
1720             if (mWhat == What::ValueFlow && (Token::Match(tok, "while|for (") || Token::simpleMatch(tok, "do {"))) {
1721                 const Token *bodyStart = nullptr;
1722                 const Token *conditionStart = nullptr;
1723                 if (Token::simpleMatch(tok, "do {")) {
1724                     bodyStart = tok->next();
1725                     if (Token::simpleMatch(bodyStart->link(), "} while ("))
1726                         conditionStart = bodyStart->link()->tokAt(2);
1727                 } else {
1728                     conditionStart = tok->next();
1729                     if (Token::simpleMatch(conditionStart->link(), ") {"))
1730                         bodyStart = conditionStart->link()->next();
1731                 }
1732     
1733                 if (!bodyStart || !conditionStart)
1734                     return Result(Result::Type::BAILOUT);
1735     
1736                 // Is expr changed in condition?
1737                 if (!isUnchanged(conditionStart, conditionStart->link(), exprVarIds, local))
1738                     return Result(Result::Type::BAILOUT);
1739     
1740                 // Is expr changed in loop body?
1741                 if (!isUnchanged(bodyStart, bodyStart->link(), exprVarIds, local))
1742                     return Result(Result::Type::BAILOUT);
1743             }
1744     
1745             if (!local && Token::Match(tok, "%name% (") && !Token::simpleMatch(tok->linkAt(1), ") {")) {
1746                 // TODO: this is a quick bailout
1747                 return Result(Result::Type::BAILOUT);
1748             }
1749     
1750             if (expr->isName() && Token::Match(tok, "%name% (") && tok->str().find("<") != std::string::npos && tok->str().find(expr->str()) != std::string::npos)
1751                 return Result(Result::Type::BAILOUT);
1752     
1753             if (exprVarIds.find(tok->varId()) != exprVarIds.end()) {
1754                 const Token *parent = tok;
1755                 bool other = false;
1756                 bool same = tok->astParent() && isSameExpression(mCpp, false, expr, tok, mLibrary, true, false, nullptr);
1757                 while (!same && Token::Match(parent->astParent(), "*|.|::|[|(|%cop%")) {
1758                     parent = parent->astParent();
1759                     if (parent && isSameExpression(mCpp, false, expr, parent, mLibrary, true, false, nullptr)) {
1760                         same = true;
1761                         if (mWhat == What::ValueFlow) {
1762                             KnownAndToken v;
1763                             v.known = mValueFlowKnown;
1764                             v.token = parent;
1765                             mValueFlow.push_back(v);
1766                         }
1767                     }
1768                     if (Token::Match(parent, ". %var%") && parent->next()->varId() && exprVarIds.find(parent->next()->varId()) == exprVarIds.end()) {
1769                         other = true;
1770                         break;
1771                     }
1772                 }
1773                 if (mWhat != What::ValueFlow && same && Token::simpleMatch(parent->astParent(), "[") && parent == parent->astParent()->astOperand2()) {
1774                     return Result(Result::Type::READ);
1775                 }
1776                 if (other)
1777                     continue;
1778                 if (Token::simpleMatch(parent->astParent(), "=") && parent == parent->astParent()->astOperand1()) {
1779                     if (!local && hasFunctionCall(parent->astParent()->astOperand2())) {
1780                         // TODO: this is a quick bailout
1781                         return Result(Result::Type::BAILOUT);
1782                     }
1783                     if (hasOperand(parent->astParent()->astOperand2(), expr)) {
1784                         if (mWhat == What::Reassign)
1785                             return Result(Result::Type::READ);
1786                         continue;
1787                     }
1788                     // ({ .. })
1789                     if (hasGccCompoundStatement(parent->astParent()->astOperand2()))
1790                         return Result(Result::Type::BAILOUT);
1791                     const bool reassign = isSameExpression(mCpp, false, expr, parent, mLibrary, false, false, nullptr);
1792                     if (reassign)
1793                         return Result(Result::Type::WRITE, parent->astParent());
1794                     return Result(Result::Type::READ);
1795                 } else if (mWhat == What::Reassign && parent->valueType() && parent->valueType()->pointer && Token::Match(parent->astParent(), "%assign%") && parent == parent->astParent()->astOperand1()) {
1796                     return Result(Result::Type::READ);
1797                 } else if (Token::Match(parent->astParent(), "%assign%") && !parent->astParent()->astParent() && parent == parent->astParent()->astOperand1()) {
1798                     continue;
1799                 } else {
1800                     // TODO: this is a quick bailout
1801                     return Result(Result::Type::BAILOUT, parent->astParent());
1802                 }
1803             }
1804     
1805             if (Token::simpleMatch(tok, ") {")) {
1806                 if (Token::simpleMatch(tok->link()->previous(), "switch ("))
1807                     // TODO: parse switch
1808                     return Result(Result::Type::BAILOUT);
1809                 const Result &result1 = checkRecursive(expr, tok->tokAt(2), tok->linkAt(1), exprVarIds, local, inInnerClass);
1810                 if (result1.type == Result::Type::READ || result1.type == Result::Type::BAILOUT)
1811                     return result1;
1812                 if (mWhat == What::ValueFlow && result1.type == Result::Type::WRITE)
1813                     mValueFlowKnown = false;
1814                 if (Token::simpleMatch(tok->linkAt(1), "} else {")) {
1815                     const Token *elseStart = tok->linkAt(1)->tokAt(2);
1816                     const Result &result2 = checkRecursive(expr, elseStart, elseStart->link(), exprVarIds, local, inInnerClass);
1817                     if (mWhat == What::ValueFlow && result2.type == Result::Type::WRITE)
1818                         mValueFlowKnown = false;
1819                     if (result2.type == Result::Type::READ || result2.type == Result::Type::BAILOUT)
1820                         return result2;
1821                     if (result1.type == Result::Type::WRITE && result2.type == Result::Type::WRITE)
1822                         return result1;
1823                     tok = elseStart->link();
1824                 } else {
1825                     tok = tok->linkAt(1);
1826                 }
1827             }
1828         }
1829     
1830         return Result(Result::Type::NONE);
1831     }

1833     bool FwdAnalysis::isGlobalData(const Token *expr) const
1834     {
1835         bool globalData = false;
1836         visitAstNodes(expr,
1837         [&](const Token *tok) {
1838             if (tok->varId() && !tok->variable()) {
1839                 // Bailout, this is probably global
1840                 globalData = true;
1841                 return ChildrenToVisit::none;
1842             }
1843             if (tok->originalName() == "->") {
1844                 // TODO check if pointer points at local data
1845                 globalData = true;
1846                 return ChildrenToVisit::none;
1847             } else if (Token::Match(tok, "[*[]") && tok->astOperand1() && tok->astOperand1()->variable()) {
1848                 // TODO check if pointer points at local data
1849                 const Variable *lhsvar = tok->astOperand1()->variable();
1850                 const ValueType *lhstype = tok->astOperand1()->valueType();
1851                 if (lhsvar->isPointer()) {
1852                     globalData = true;
1853                     return ChildrenToVisit::none;
1854                 } else if (lhsvar->isArgument() && lhsvar->isArray()) {
1855                     globalData = true;
1856                     return ChildrenToVisit::none;
1857                 } else if (lhsvar->isArgument() && (!lhstype || (lhstype->type <= ValueType::Type::VOID && !lhstype->container))) {
1858                     globalData = true;
1859                     return ChildrenToVisit::none;
1860                 }
1861             }
1862             if (tok->varId() == 0 && tok->isName() && tok->previous()->str() != ".") {
1863                 globalData = true;
1864                 return ChildrenToVisit::none;
1865             }
1866             if (tok->variable()) {
1867                 // TODO : Check references
1868                 if (tok->variable()->isReference() && tok != tok->variable()->nameToken()) {
1869                     globalData = true;
1870                     return ChildrenToVisit::none;
1871                 }
1872                 if (tok->variable()->isExtern()) {
1873                     globalData = true;
1874                     return ChildrenToVisit::none;
1875                 }
1876                 if (tok->previous()->str() != "." && !tok->variable()->isLocal() && !tok->variable()->isArgument()) {
1877                     globalData = true;
1878                     return ChildrenToVisit::none;
1879                 }
1880                 if (tok->variable()->isArgument() && tok->variable()->isPointer() && tok != expr) {
1881                     globalData = true;
1882                     return ChildrenToVisit::none;
1883                 }
1884                 if (tok->variable()->isPointerArray()) {
1885                     globalData = true;
1886                     return ChildrenToVisit::none;
1887                 }
1888             }
1889             // Unknown argument type => it might be some reference type..
1890             if (mCpp && tok->str() == "." && tok->astOperand1() && tok->astOperand1()->variable() && !tok->astOperand1()->valueType()) {
1891                 globalData = true;
1892                 return ChildrenToVisit::none;
1893             }
1894             if (Token::Match(tok, ".|["))
1895                 return ChildrenToVisit::op1;
1896             return ChildrenToVisit::op1_and_op2;
1897         });
1898         return globalData;
1899     }

1901     std::set<int> FwdAnalysis::getExprVarIds(const Token* expr, bool* localOut, bool* unknownVarIdOut) const
1902     {
1903         // all variable ids in expr.
1904         std::set<int> exprVarIds;
1905         bool local = true;
1906         bool unknownVarId = false;
1907         visitAstNodes(expr,
1908         [&](const Token *tok) {
1909             if (tok->varId() == 0 && tok->isName() && tok->previous()->str() != ".") {
1910                 // unknown variable
1911                 unknownVarId = true;
1912                 return ChildrenToVisit::none;
1913             }
1914             if (tok->varId() > 0) {
1915                 exprVarIds.insert(tok->varId());
1916                 if (!Token::simpleMatch(tok->previous(), ".")) {
1917                     const Variable *var = tok->variable();
1918                     if (var && var->isReference() && var->isLocal() && Token::Match(var->nameToken(), "%var% [=(]") && !isGlobalData(var->nameToken()->next()->astOperand2()))
1919                         return ChildrenToVisit::none;
1920                     const bool deref = tok->astParent() && (tok->astParent()->isUnaryOp("*") || (tok->astParent()->str() == "[" && tok == tok->astParent()->astOperand1()));
1921                     local &= !nonLocal(tok->variable(), deref);
1922                 }
1923             }
1924             return ChildrenToVisit::op1_and_op2;
1925         });
1926         if (localOut)
1927             *localOut = local;
1928         if (unknownVarIdOut)
1929             *unknownVarIdOut = unknownVarId;
1930         return exprVarIds;
1931     }

1933     FwdAnalysis::Result FwdAnalysis::check(const Token* expr, const Token* startToken, const Token* endToken)
1934     {
1935         // all variable ids in expr.
1936         bool local = true;
1937         bool unknownVarId = false;
1938         std::set<int> exprVarIds = getExprVarIds(expr, &local, &unknownVarId);
1939     
1940         if (unknownVarId)
1941             return Result(FwdAnalysis::Result::Type::BAILOUT);
1942     
1943         if (mWhat == What::Reassign && isGlobalData(expr))
1944             local = false;
1945     
1946         // In unused values checking we do not want to check assignments to
1947         // global data.
1948         if (mWhat == What::UnusedValue && isGlobalData(expr))
1949             return Result(FwdAnalysis::Result::Type::BAILOUT);
1950     
1951         Result result = checkRecursive(expr, startToken, endToken, exprVarIds, local, false);
1952     
1953         // Break => continue checking in outer scope
1954         while (mWhat!=What::ValueFlow && result.type == FwdAnalysis::Result::Type::BREAK) {
1955             const Scope *s = result.token->scope();
1956             while (s->type == Scope::eIf)
1957                 s = s->nestedIn;
1958             if (s->type != Scope::eSwitch && s->type != Scope::eWhile && s->type != Scope::eFor)
1959                 break;
1960             result = checkRecursive(expr, s->bodyEnd->next(), endToken, exprVarIds, local, false);
1961         }
1962     
1963         return result;
1964     }

1966     bool FwdAnalysis::hasOperand(const Token *tok, const Token *lhs) const
1967     {
1968         if (!tok)
1969             return false;
1970         if (isSameExpression(mCpp, false, tok, lhs, mLibrary, false, false, nullptr))
1971             return true;
1972         return hasOperand(tok->astOperand1(), lhs) || hasOperand(tok->astOperand2(), lhs);
1973     }

1975     const Token *FwdAnalysis::reassign(const Token *expr, const Token *startToken, const Token *endToken)
1976     {
1977         mWhat = What::Reassign;
1978         Result result = check(expr, startToken, endToken);
1979         return result.type == FwdAnalysis::Result::Type::WRITE ? result.token : nullptr;
1980     }

1982     bool FwdAnalysis::unusedValue(const Token *expr, const Token *startToken, const Token *endToken)
1983     {
1984         if (isEscapedAlias(expr))
1985             return false;
1986         mWhat = What::UnusedValue;
1987         Result result = check(expr, startToken, endToken);
1988         return (result.type == FwdAnalysis::Result::Type::NONE || result.type == FwdAnalysis::Result::Type::RETURN) && !possiblyAliased(expr, startToken);

1999     bool FwdAnalysis::possiblyAliased(const Token *expr, const Token *startToken) const
2000     {
2001         if (expr->isUnaryOp("*"))
2002             return true;
2003     
2004         const bool macro = false;
2005         const bool pure = false;
2006         const bool followVar = false;
2007         for (const Token *tok = startToken; tok; tok = tok->previous()) {
2008             if (tok->str() == "{" && tok->scope()->type == Scope::eFunction)
2009                 break;
2010     
2011             if (Token::Match(tok, "%name% (") && !Token::Match(tok, "if|while|for")) {
2012                 // Is argument passed by reference?
2013                 const std::vector<const Token*> args = getArguments(tok);
2014                 for (int argnr = 0; argnr < args.size(); ++argnr) {
2015                     if (!Token::Match(args[argnr], "%name%|.|::"))
2016                         continue;
2017                     if (tok->function() && tok->function()->getArgumentVar(argnr) && !tok->function()->getArgumentVar(argnr)->isReference() && !tok->function()->isConst())
2018                         continue;
2019                     for (const Token *subexpr = expr; subexpr; subexpr = subexpr->astOperand1()) {
2020                         if (isSameExpression(mCpp, macro, subexpr, args[argnr], mLibrary, pure, followVar))
2021                             return true;
2022                     }
2023                 }
2024                 continue;
2025             }
2026     
2027             const Token *addrOf = nullptr;
2028             if (Token::Match(tok, "& %name% ="))
2029                 addrOf = tok->tokAt(2)->astOperand2();
2030             else if (tok->isUnaryOp("&"))
2031                 addrOf = tok->astOperand1();
2032             else if (Token::simpleMatch(tok, "std :: ref ("))
2033                 addrOf = tok->tokAt(3)->astOperand2();
2034             else
2035                 continue;
2036     
2037             for (const Token *subexpr = expr; subexpr; subexpr = subexpr->astOperand1()) {
2038                 if (isSameExpression(mCpp, macro, subexpr, addrOf, mLibrary, pure, followVar))
2039                     return true;
2040             }
2041         }
2042         return false;
2043     }

2045     bool FwdAnalysis::isEscapedAlias(const Token* expr)
2046     {
2047         for (const Token *subexpr = expr; subexpr; subexpr = subexpr->astOperand1()) {
2048             for (const ValueFlow::Value &val : subexpr->values()) {
2049                 if (!val.isLocalLifetimeValue())
2050                     continue;
2051                 const Variable* var = val.tokvalue->variable();
2052                 if (!var)
2053                     continue;
2054                 if (!var->isLocal())
2055                     return true;
2056                 if (var->isArgument())
2057                     return true;
2058     
2059             }
2060         }
2061         return false;
2062     }
