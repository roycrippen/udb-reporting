42     bool ApplicationList::loadSettings()
43     {
44         QSettings settings;
45         QStringList names = settings.value(SETTINGS_APPLICATION_NAMES, QStringList()).toStringList();
46         QStringList paths = settings.value(SETTINGS_APPLICATION_PATHS, QStringList()).toStringList();
47         QStringList params = settings.value(SETTINGS_APPLICATION_PARAMS, QStringList()).toStringList();
48         int defapp = settings.value(SETTINGS_APPLICATION_DEFAULT, -1).toInt();
49     
50         // Params will be empty first time starting with the new setting.
51         // Return false and inform user about problem with application settings.
52         bool succeeded = true;
53         if (!names.empty() && !paths.empty() && params.empty()) {
54             for (int i = 0; i < paths.length(); i++)
55                 params << QString();
56             succeeded = false;
57         }
58     
59         if (names.empty() && paths.empty() && params.empty()) {
60     #ifndef _WIN32
61             // use as default for gnome environments
62             if (QFileInfo("/usr/bin/gedit").isExecutable()) {
63                 Application app;
64                 app.setName("gedit");
65                 app.setPath("/usr/bin/gedit");
66                 app.setParameters("+(line) (file)");
67                 addApplication(app);
68                 defapp = 0;
69             }
70             checkAndAddApplication("/usr/bin/geany","geany","+(line) (file)");
71             checkAndAddApplication("/usr/bin/qtcreator","Qt Creator","-client (file):(line)");
72             // use as default for kde environments
73             if (QFileInfo("/usr/bin/kate").isExecutable()) {
74                 Application app;
75                 app.setName("kate");
76                 app.setPath("/usr/bin/kate");
77                 app.setParameters("-l(line) (file)");
78                 addApplication(app);
79                 defapp = 0;
80             }
81     #else
82             if (findDefaultWindowsEditor()) {
83                 defapp = 0;
84             }
85     #endif
86         } else if (names.size() == paths.size()) {
87             for (int i = 0; i < names.size(); i++) {
88                 const Application app(names[i], paths[i], params[i]);
89                 addApplication(app);
90             }
91         }
92     
93         if (defapp == -1)
94             mDefaultApplicationIndex = 0;
95         else if (defapp < names.size())
96             mDefaultApplicationIndex = defapp;
97         else
98             mDefaultApplicationIndex = 0;
99     
100         return succeeded;
101     }

103     void ApplicationList::saveSettings() const
104     {
105         QSettings settings;
106         QStringList names;
107         QStringList paths;
108         QStringList params;
109     
110         for (int i = 0; i < getApplicationCount(); i++) {
111             const Application& app = getApplication(i);
112             names << app.getName();
113             paths << app.getPath();
114             params << app.getParameters();
115         }
116     
117         settings.setValue(SETTINGS_APPLICATION_NAMES, names);
118         settings.setValue(SETTINGS_APPLICATION_PATHS, paths);
119         settings.setValue(SETTINGS_APPLICATION_PARAMS, params);
120         settings.setValue(SETTINGS_APPLICATION_DEFAULT, mDefaultApplicationIndex);
121     }

148     void ApplicationList::addApplication(const Application &app)
149     {
150         if (app.getName().isEmpty() || app.getPath().isEmpty()) {
151             return;
152         }
153         mApplications << app;
154     }

156     void ApplicationList::removeApplication(const int index)
157     {
158         mApplications.removeAt(index);
159     }

161     void ApplicationList::setDefault(const int index)
162     {
163         if (index < mApplications.size() && index >= 0) {
164             mDefaultApplicationIndex = index;
165         }
166     }

168     void ApplicationList::copy(const ApplicationList *list)
169     {
170         if (!list) {
171             return;
172         }
173     
174         clear();
175         for (int i = 0; i < list->getApplicationCount(); i++) {
176             const Application& app = list->getApplication(i);
177             addApplication(app);
178         }
179         mDefaultApplicationIndex = list->getDefaultApplication();
180     }
181     
182     void ApplicationList::clear()
183     {
184         mApplications.clear();
185         mDefaultApplicationIndex = -1;
186     }

188     bool ApplicationList::checkAndAddApplication(QString appPath, QString name, QString parameters)
189     {
190         if (QFileInfo(appPath).exists() && QFileInfo(appPath).isExecutable()) {
191             Application app;
192             app.setName(name);
193             app.setPath("\"" + appPath + "\"");
194             app.setParameters(parameters);
195             addApplication(app);
196             return true;
197         }
198         return false;
199     }

201     bool ApplicationList::findDefaultWindowsEditor()
202     {
203         bool foundOne = false;
204     #ifdef WIN64 // As long as we do support 32-bit XP, we cannot be sure that the environment variable "ProgramFiles(x86)" exists
205         const QString appPathx86(getenv("ProgramFiles(x86)"));
206     #else
207         const QString appPathx86(getenv("ProgramFiles"));
208     #endif
209         const QString appPathx64(getenv("ProgramW6432"));
210         const QString windowsPath(getenv("windir"));
211     
212         if (checkAndAddApplication(appPathx86 + "\\Notepad++\\notepad++.exe", "Notepad++", "-n(line) (file)"))
213             foundOne = true;
214         else if (checkAndAddApplication(appPathx64 + "\\Notepad++\\notepad++.exe", "Notepad++", "-n(line) (file)"))
215             foundOne = true;
216     
217         if (checkAndAddApplication(appPathx86 + "\\Notepad2\\Notepad2.exe", "Notepad2", "/g (line) (file)"))
218             foundOne = true;
219         else if (checkAndAddApplication(appPathx64 + "\\Notepad2\\Notepad2.exe", "Notepad2", "/g (line) (file)"))
220             foundOne = true;
221     
222         if (checkAndAddApplication(windowsPath + "\\system32\\notepad.exe", "Notepad", "(file)"))
223             foundOne = true;
224     
225         QString regPath = "HKEY_CLASSES_ROOT\\Applications\\QtProject.QtCreator.pro\\shell\\Open\\command";
226         QSettings registry(regPath, QSettings::NativeFormat);
227         QString qtCreatorRegistry = registry.value("Default", QString()).toString();
228         QString qtCreatorPath = qtCreatorRegistry.left(qtCreatorRegistry.indexOf(".exe") + 4);
229         if (!qtCreatorRegistry.isEmpty() && checkAndAddApplication(qtCreatorPath, "Qt Creator", "-client (file):(line)")) {
230             foundOne = true;
231         }
232     
233         const QString regPathUEdit32 = "HKEY_CLASSES_ROOT\\Applications\\Uedit32.exe\\shell\\open\\Command";
234         const QSettings registryUEdit32(regPathUEdit32, QSettings::NativeFormat);
235         const QString uedit32Registry = registryUEdit32.value("Default", QString()).toString();
236         if (!uedit32Registry.isEmpty()) {
237             // Extract path to executable and make sure there is no single quotation mark at the beginning
238             const QString uedit32Path = uedit32Registry.left(uedit32Registry.indexOf(".exe") + 4).replace("\"", "");
239             if (checkAndAddApplication(uedit32Path, "UltraEdit 32", "(file)/(line)")) {
240                 foundOne = true;
241             }
242         }
243     
244         const QString regPathUEdit64 = "HKEY_CLASSES_ROOT\\Applications\\uedit64.exe\\shell\\open\\Command";
245         const QSettings registryUEdit64(regPathUEdit64, QSettings::NativeFormat);
246         const QString uedit64Registry = registryUEdit64.value("Default", QString()).toString();
247         if (!uedit64Registry.isEmpty()) {
248             // Extract path to executable and make sure there is no single quotation mark at the beginning
249             const QString uedit64Path = uedit64Registry.left(uedit64Registry.indexOf(".exe") + 4).replace("\"", "");
250             if (checkAndAddApplication(uedit64Path, "UltraEdit 64", "(file)/(line)")) {
251                 foundOne = true;
252             }
253         }
254     
255         const QString regPathMSVSCode = "HKEY_CLASSES_ROOT\\Applications\\Code.exe\\shell\\open\\command";
256         const QSettings registryMSVSCode(regPathMSVSCode, QSettings::NativeFormat);
257         const QString msvscodeRegistry = registryMSVSCode.value("Default", QString()).toString();
258         if (!msvscodeRegistry.isEmpty()) {
259             const QString msvscodePath = msvscodeRegistry.left(msvscodeRegistry.indexOf(".exe") + 4).replace("\"", "");
260             if (checkAndAddApplication(msvscodePath, "Microsoft VS Code", "-g (file):(line)")) {
261                 foundOne = true;
262             }
263         }
264     
265         return foundOne;
266     }
