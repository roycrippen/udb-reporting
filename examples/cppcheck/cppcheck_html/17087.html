29     std::string ExprEngine::str(int128_t value)
30     {
31         std::ostringstream ostr;
32     #ifdef __GNUC__
33         if (value == (int)value) {
34             ostr << (int) value;
35             return ostr.str();
36         }
37         if (value < 0) {
38             ostr << "-";
39             value = -value;
40         }
41     
42         uint64_t high = value >> 64;
43         uint64_t low = value;
44         if (high > 0)
45             ostr << "h" << std::hex << high << "l";
46         ostr << std::hex << low;
47     #else
48         ostr << value;
49     #endif
50         return ostr.str();
51     }

56         class TrackExecution {
57         public:
58             TrackExecution() : mDataIndex(0) {}
59             std::map<const Token *, std::vector<std::string>> map;
60             int getNewDataIndex() {
61                 return mDataIndex++;
62             }
63     
64             void symbolRange(const Token *tok, ExprEngine::ValuePtr value) {
65                 if (!tok || !value)
66                     return;
67                 const std::string &symbolicExpression = value->getSymbolicExpression();
68                 if (symbolicExpression[0] != '$')
69                     return;
70                 if (mSymbols.find(symbolicExpression) != mSymbols.end())
71                     return;
72                 mSymbols.insert(symbolicExpression);
73                 map[tok].push_back(symbolicExpression + "=" + value->getRange());
74     
75             }
76     
77             void state(const Token *tok, const std::string &s) {
78                 map[tok].push_back(s);
79             }
80     
81             void print(std::ostream &out) {
82                 std::set<std::pair<int,int>> locations;
83                 for (auto it : map) {
84                     locations.insert(std::pair<int,int>(it.first->linenr(), it.first->column()));
85                 }
86                 for (const std::pair<int,int> &loc : locations) {
87                     int lineNumber = loc.first;
88                     int column = loc.second;
89                     for (auto &it : map) {
90                         const Token *tok = it.first;
91                         if (lineNumber != tok->linenr())
92                             continue;
93                         if (column != tok->column())
94                             continue;
95                         const std::vector<std::string> &dumps = it.second;
96                         for (const std::string &dump : dumps)
97                             out << lineNumber << ":" << column << ": " << dump << "\n";
98                     }
99                 }
100             }
101         private:
102             int mDataIndex;
103             std::set<std::string> mSymbols;
104         };

107         public:
108             Data(int *symbolValueIndex, const Tokenizer *tokenizer, const Settings *settings, const std::vector<ExprEngine::Callback> &callbacks, TrackExecution *trackExecution)
109                 : DataBase(settings)
110                 , symbolValueIndex(symbolValueIndex)
111                 , tokenizer(tokenizer)
112                 , callbacks(callbacks)
113                 , mTrackExecution(trackExecution)
114                 , mDataIndex(trackExecution->getNewDataIndex()) {}
115             typedef std::map<nonneg int, std::shared_ptr<ExprEngine::Value>> Memory;
116             Memory memory;
117             int * const symbolValueIndex;
118             const Tokenizer * const tokenizer;
119             const std::vector<ExprEngine::Callback> &callbacks;
120     
121             void assignValue(const Token *tok, unsigned int varId, ExprEngine::ValuePtr value) {
122                 mTrackExecution->symbolRange(tok, value);
123                 if (value) {
124                     if (auto arr = std::dynamic_pointer_cast<ExprEngine::ArrayValue>(value)) {
125                         mTrackExecution->symbolRange(tok, arr->size);
126                         for (const auto &indexAndValue: arr->data)
127                             mTrackExecution->symbolRange(tok, indexAndValue.value);
128                     } else if (auto s = std::dynamic_pointer_cast<ExprEngine::StructValue>(value)) {
129                         for (const auto &m: s->member)
130                             mTrackExecution->symbolRange(tok, m.second);
131                     }
132                 }
133                 memory[varId] = value;
134             }

136             void assignStructMember(const Token *tok, ExprEngine::StructValue *structVal, const std::string &memberName, ExprEngine::ValuePtr value) {
137                 mTrackExecution->symbolRange(tok, value);
138                 structVal->member[memberName] = value;
139             }

141             std::vector<Data> getData(const Token *cond, bool trueData) {
142                 std::vector<Data> ret;
143                 ret.push_back(Data(symbolValueIndex, tokenizer, settings, callbacks, mTrackExecution));
144                 for (Memory::const_iterator mem = memory.cbegin(); mem != memory.cend(); ++mem) {
145                     for (Data &data : ret)
146                         data.memory[mem->first] = mem->second;
147     
148                     if (cond->isComparisonOp() && cond->astOperand1()->varId() == mem->first && cond->astOperand2()->isNumber()) {
149                         const int128_t rhsValue = MathLib::toLongNumber(cond->astOperand2()->str());
150                         if (auto intRange = std::dynamic_pointer_cast<ExprEngine::IntRange>(mem->second)) {
151                             if (cond->str() == ">") {
152                                 if (trueData) {
153                                     if (intRange->maxValue <= rhsValue)
154                                         return std::vector<Data>();
155                                     auto val = std::make_shared<ExprEngine::IntRange>(getNewSymbolName(), rhsValue + 1, intRange->maxValue);
156                                     ret[0].assignValue(cond, mem->first, val);
157                                 } else { /* if (!trueData) */
158                                     if (intRange->maxValue <= rhsValue)
159                                         return std::vector<Data>();
160                                     auto val = std::make_shared<ExprEngine::IntRange>(getNewSymbolName(), intRange->minValue, rhsValue);
161                                     ret[0].assignValue(cond, mem->first, val);
162                                 }
163                             }
164                         }
165                     }
166     
167                     else if (cond->varId() == mem->first) {
168                         if (auto intRange = std::dynamic_pointer_cast<ExprEngine::IntRange>(mem->second)) {
169                             if (trueData) {
170                                 if (intRange->minValue == 0 && intRange->maxValue == 0)
171                                     return std::vector<Data>();
172                                 if (intRange->minValue < 0) {
173                                     auto val = std::make_shared<ExprEngine::IntRange>(getNewSymbolName(), intRange->minValue, -1);
174                                     ret[0].assignValue(cond, mem->first, val);
175                                 }
176                                 if (intRange->maxValue > 0) {
177                                     auto val = std::make_shared<ExprEngine::IntRange>(getNewSymbolName(), 1, intRange->maxValue);
178                                     if (intRange->minValue < 0) {
179                                         // create additional intrange..
180                                         ret.push_back(Data(symbolValueIndex, tokenizer, settings, callbacks, mTrackExecution));
181                                         ret.back().memory = ret[0].memory;
182                                     }
183                                     ret[0].assignValue(cond, mem->first, val);
184                                 }
185                             } else { /* if (!trueData) */
186                                 if (intRange->maxValue < 0 || intRange->minValue > 0)
187                                     return std::vector<Data>();
188     
189                                 auto val = std::make_shared<ExprEngine::IntRange>(getNewSymbolName(), 0, 0);
190                                 ret[0].assignValue(cond, mem->first, val);
191                             }
192                         }
193                     }
194                 }
195                 return ret;
196             }

202             std::shared_ptr<ExprEngine::ArrayValue> getArrayValue(const Token *tok) {
203                 const Memory::iterator it = memory.find(tok->varId());

208                 auto val = std::make_shared<ExprEngine::ArrayValue>(this, tok->variable());

213             ExprEngine::ValuePtr getValue(unsigned int varId, const ValueType *valueType, const Token *tok) {
214                 const Memory::const_iterator it = memory.find(varId);
215                 if (it != memory.end())
216                     return it->second;
217                 if (!valueType)
218                     return ExprEngine::ValuePtr();
219                 ExprEngine::ValuePtr value = getValueRangeFromValueType(getNewSymbolName(), valueType, *settings);
220                 if (value) {
221                     assignValue(tok, varId, value);

226             void trackProgramState(const Token *tok) {
227                 if (memory.empty())
228                     return;
229                 const SymbolDatabase * const symbolDatabase = tokenizer->getSymbolDatabase();
230                 std::ostringstream s;
231                 s << "{"; // << mDataIndex << ":";
232                 for (auto mem : memory) {
233                     ExprEngine::ValuePtr value = mem.second;
234                     const Variable *var = symbolDatabase->getVariableFromVarId(mem.first);
235                     if (!var)
236                         continue;
237                     s << " " << var->name() << "=";
238                     if (!value)
239                         s << "(null)";
240                     else if (value->name[0] == '$' && value->getSymbolicExpression() != value->name)
241                         s << "(" << value->name << "," << value->getSymbolicExpression() << ")";
242                     else
243                         s << value->name;
244                 }
245                 s << "}";
246                 mTrackExecution->state(tok, s.str());
247             }

250             const int mDataIndex;
251         };

254     static ExprEngine::ValuePtr simplifyValue(ExprEngine::ValuePtr origValue)
255     {
256         auto b = std::dynamic_pointer_cast<ExprEngine::BinOpResult>(origValue);
257         if (!b)
258             return origValue;
259         if (!b->op1 || !b->op2)
260             return origValue;
261         auto intRange1 = std::dynamic_pointer_cast<ExprEngine::IntRange>(b->op1);
262         auto intRange2 = std::dynamic_pointer_cast<ExprEngine::IntRange>(b->op1);
263         if (intRange1 && intRange2 && intRange1->minValue == intRange1->maxValue && intRange2->minValue == intRange2->maxValue) {
264             const std::string &binop = b->binop;
265             int128_t v;
266             if (binop == "+")
267                 v = intRange1->minValue + intRange2->minValue;
268             else if (binop == "-")
269                 v = intRange1->minValue - intRange2->minValue;
270             else if (binop == "*")
271                 v = intRange1->minValue * intRange2->minValue;
272             else if (binop == "/" && intRange2->minValue != 0)
273                 v = intRange1->minValue / intRange2->minValue;
274             else if (binop == "%" && intRange2->minValue != 0)
275                 v = intRange1->minValue % intRange2->minValue;
276             else
277                 return origValue;
278             return std::make_shared<ExprEngine::IntRange>(ExprEngine::str(v), v, v);
279         }
280         return origValue;
281     }

283     static ExprEngine::ValuePtr translateUninitValueToRange(ExprEngine::ValuePtr value, const ::ValueType *valueType, Data &data)
284     {
285         if (!value)
286             return value;
287         if (value->type == ExprEngine::ValueType::UninitValue) {
288             auto rangeValue = getValueRangeFromValueType(data.getNewSymbolName(), valueType, *data.settings);
289             if (rangeValue)
290                 return rangeValue;
291         }
292         if (auto conditionalValue = std::dynamic_pointer_cast<ExprEngine::ConditionalValue>(value)) {
293             if (conditionalValue->values.size() == 1 && conditionalValue->values[0].second && conditionalValue->values[0].second->type == ExprEngine::ValueType::UninitValue) {
294                 auto rangeValue = getValueRangeFromValueType(data.getNewSymbolName(), valueType, *data.settings);
295                 if (rangeValue)
296                     return rangeValue;
297             }
298         }
299         return value;
300     }

302     static int128_t truncateInt(int128_t value, int bits, char sign)
303     {
304         value = value & (((int128_t)1 << bits) - 1);
305         // Sign extension
306         if (sign == 's' && value & (1ULL << (bits - 1)))
307             value |= ~(((int128_t)1 << bits) - 1);
308         return value;
309     }

318     ExprEngine::ArrayValue::ArrayValue(DataBase *data, const Variable *var)
319         : Value(data->getNewSymbolName(), ExprEngine::ValueType::ArrayValue)
320     {
321         if (var) {
322             int sz = 1;
323             for (const auto &dim : var->dimensions()) {
324                 if (!dim.known) {
325                     sz = -1;
326                     break;
327                 }
328                 sz *= dim.num;
329             }
330             if (sz >= 1)
331                 size = std::make_shared<ExprEngine::IntRange>(std::to_string(sz), sz, sz);
332         }
333         ValuePtr val;
334         if (var && !var->isGlobal() && !var->isStatic())
335             val = std::make_shared<ExprEngine::UninitValue>();
336         else if (var && var->valueType()) {
337             ::ValueType vt(*var->valueType());
338             vt.pointer = 0;
339             val = getValueRangeFromValueType(data->getNewSymbolName(), &vt, *data->settings);
340         }
341         assign(ExprEngine::ValuePtr(), val);
342     }

370     static bool isNonOverlapping(ExprEngine::ValuePtr v1, ExprEngine::ValuePtr v2)
371     {
372         if (!v1 || !v2)
373             return false; // Don't know!
374         auto intRange1 = std::dynamic_pointer_cast<ExprEngine::IntRange>(v1);
375         auto intRange2 = std::dynamic_pointer_cast<ExprEngine::IntRange>(v2);
376         if (intRange1 && intRange2 && (intRange1->minValue > intRange2->maxValue || intRange1->maxValue < intRange2->maxValue))
377             return true;
378         return false;
379     }

381     ExprEngine::ConditionalValue::Vector ExprEngine::ArrayValue::read(ExprEngine::ValuePtr index) const
382     {
383         ExprEngine::ConditionalValue::Vector ret;
384         for (const auto indexAndValue : data) {
385             if (isEqual(index, indexAndValue.index))
386                 ret.clear();
387             if (isNonOverlapping(index, indexAndValue.index))
388                 continue;
389             // Array contains string literal data...
390             if (!indexAndValue.index && indexAndValue.value->type == ExprEngine::ValueType::StringLiteralValue) {
391                 auto stringLiteral = std::dynamic_pointer_cast<ExprEngine::StringLiteralValue>(indexAndValue.value);
392                 if (!stringLiteral) {
393                     ret.push_back(std::pair<ValuePtr,ValuePtr>(indexAndValue.index, std::make_shared<ExprEngine::IntRange>("", -128, 128)));
394                     continue;
395                 }
396                 if (auto i = std::dynamic_pointer_cast<ExprEngine::IntRange>(index)) {
397                     if (stringLiteral && i->minValue >= 0 && i->minValue == i->maxValue) {
398                         int c = 0;
399                         if (i->minValue < stringLiteral->size())
400                             c = stringLiteral->string[i->minValue];
401                         ret.push_back(std::pair<ValuePtr,ValuePtr>(indexAndValue.index, std::make_shared<ExprEngine::IntRange>(std::to_string(c), c, c)));
402                         continue;
403                     }
404                 }
405                 int cmin = 0, cmax = 0;
406                 for (char c : stringLiteral->string) {
407                     if (c < cmin)
408                         cmin = c;
409                     else if (c > cmax)
410                         cmax = c;
411                 }
412                 ret.push_back(std::pair<ValuePtr,ValuePtr>(indexAndValue.index, std::make_shared<ExprEngine::IntRange>("", cmin, cmax)));
413                 continue;
414             }
415     
416             // Rename IntRange
417             if (auto i = std::dynamic_pointer_cast<ExprEngine::IntRange>(indexAndValue.value)) {
418                 ret.push_back(std::pair<ValuePtr,ValuePtr>(indexAndValue.index, std::make_shared<ExprEngine::IntRange>(indexAndValue.value->name + ":" + index->name, i->minValue, i->maxValue)));
419                 continue;
420             }
421     
422             ret.push_back(std::pair<ValuePtr,ValuePtr>(indexAndValue.index, indexAndValue.value));
423         }
424     
425         if (ret.size() == 1)
426             ret[0].first = ExprEngine::ValuePtr();
427         else if (ret.size() == 2 && !ret[0].first) {
428             ret[0].first = std::make_shared<ExprEngine::BinOpResult>("!=", index, ret[1].first);
429             ret[1].first = std::make_shared<ExprEngine::BinOpResult>("==", index, ret[1].first);
430         } else {
431             // FIXME!!
432             ret.clear();
433         }
434     
435         return ret;
436     }

439     {
440         std::ostringstream ostr;
441         ostr << "{";
442         bool first = true;
443         for (auto condvalue : values) {
444             ValuePtr cond = condvalue.first;
445             ValuePtr value = condvalue.second;
446     
447             if (!first)
448                 ostr << ",";
449             first = false;

502     std::string ExprEngine::BinOpResult::getRange() const
503     {
504         IntOrFloatValue minValue, maxValue;
505         getRange(&minValue, &maxValue);
506         const std::string s1 = minValue.isFloat()
507                                ? std::to_string(minValue.floatValue)
508                                : str(minValue.intValue);
509         const std::string s2 = maxValue.isFloat()
510                                ? std::to_string(maxValue.floatValue)
511                                : str(maxValue.intValue);
512     
513         if (s1 == s2)
514             return s1;
515         return s1 + ":" + s2;
516     }

518     void ExprEngine::BinOpResult::getRange(ExprEngine::BinOpResult::IntOrFloatValue *minValue, ExprEngine::BinOpResult::IntOrFloatValue *maxValue) const
519     {
520         std::map<ValuePtr, int> valueBit;
521         // Assign a bit number for each leaf
522         int bit = 0;
523         for (ValuePtr v : mLeafs) {
524             if (auto intRange = std::dynamic_pointer_cast<IntRange>(v)) {
525                 if (intRange->minValue == intRange->maxValue) {
526                     valueBit[v] = 30;
527                     continue;
528                 }
529             }
530     
531             valueBit[v] = bit++;
532         }
533     
534         if (bit > 24)
535             throw std::runtime_error("Internal error: bits");
536     
537         for (int test = 0; test < (1 << bit); ++test) {
538             auto result = evaluate(test, valueBit);
539             if (test == 0)
540                 *minValue = *maxValue = result;
541             else if (result.isFloat()) {
542                 if (result.floatValue < minValue->floatValue)
543                     *minValue = result;
544                 else if (result.floatValue > maxValue->floatValue)
545                     *maxValue = result;
546             } else {
547                 if (result.intValue < minValue->intValue)
548                     *minValue = result;
549                 else if (result.intValue > maxValue->intValue)
550                     *maxValue = result;
551             }
552         }
553     }

560     bool ExprEngine::BinOpResult::isIntValueInRange(int value) const
561     {
562         IntOrFloatValue minValue, maxValue;
563         getRange(&minValue, &maxValue);
564         return value >= minValue.intValue && value <= maxValue.intValue;

567     #define BINARY_OP(OP) \
568         if (binop == #OP) { \
569             struct ExprEngine::BinOpResult::IntOrFloatValue result(lhs); \
570             if (lhs.isFloat()) \
571             { result.type = lhs.type; result.floatValue = lhs.floatValue OP (rhs.isFloat() ? rhs.floatValue : rhs.intValue); } \
572             else if (rhs.isFloat()) \
573             { result.type = rhs.type; result.floatValue = lhs.intValue OP rhs.floatValue; } \
574             else { result.type = lhs.type; result.intValue = lhs.intValue OP rhs.intValue; } \
575             return result; \
576         }
577     
578     #define BINARY_RELATIONAL_COMPARISON(OP) \
579         if (binop == #OP) { \
580             struct ExprEngine::BinOpResult::IntOrFloatValue result(lhs); \
581             if (lhs.isFloat()) \
582             { result.setIntValue(lhs.floatValue OP (rhs.isFloat() ? rhs.floatValue : rhs.intValue)); } \
583             else if (rhs.isFloat()) \
584             { result.setIntValue(lhs.intValue OP rhs.floatValue); } \
585             else { result.setIntValue(lhs.intValue OP rhs.intValue); } \
586             return result; \
587         }
588     
589     #define BINARY_EQ_COMPARISON(OP) \
590         if (binop == #OP && !lhs.isFloat() && !rhs.isFloat()) { \
591             struct ExprEngine::BinOpResult::IntOrFloatValue result; \
592             result.setIntValue(lhs.intValue OP rhs.intValue); \
593             return result; \
594         }
595     
596     #define BINARY_INT_OP(OP) \
597         if (binop == #OP) { \
598             struct ExprEngine::BinOpResult::IntOrFloatValue result; \
599             result.setIntValue(lhs.intValue OP rhs.intValue); \
600             return result; \
601         }
602     
603     #define BINARY_OP_DIV(OP) \
604         if (binop == #OP) { \
605             struct ExprEngine::BinOpResult::IntOrFloatValue result(lhs); \
606             if (lhs.isFloat()) \
607             { result.type = lhs.type; result.floatValue = lhs.floatValue OP (rhs.isFloat() ? rhs.floatValue : rhs.intValue); } \
608             else if (rhs.isFloat()) \
609             { result.type = rhs.type; result.floatValue = lhs.intValue OP rhs.floatValue; } \
610             else if (rhs.intValue != 0) { result.type = lhs.type; result.intValue = lhs.intValue OP rhs.intValue; } \
611             return result; \
612         }
613     
614     ExprEngine::BinOpResult::IntOrFloatValue ExprEngine::BinOpResult::evaluate(int test, const std::map<ExprEngine::ValuePtr, int> &valueBit) const
615     {
616         const ExprEngine::BinOpResult::IntOrFloatValue lhs = evaluateOperand(test, valueBit, op1);
617         const ExprEngine::BinOpResult::IntOrFloatValue rhs = evaluateOperand(test, valueBit, op2);
618         BINARY_OP(+)
619         BINARY_OP(-)
620         BINARY_OP(*)
621         BINARY_OP_DIV(/)
622         BINARY_INT_OP(&)
623         BINARY_INT_OP(|)
624         BINARY_INT_OP(^)
625         BINARY_INT_OP(<<)
626         BINARY_INT_OP(>>)
627         BINARY_EQ_COMPARISON(==)
628         BINARY_EQ_COMPARISON(!=)
629         BINARY_RELATIONAL_COMPARISON(>=)
630         BINARY_RELATIONAL_COMPARISON(>)
631         BINARY_RELATIONAL_COMPARISON(<=)
632         BINARY_RELATIONAL_COMPARISON(<)
633     
634         if (binop == "%" && rhs.intValue != 0) {
635             struct ExprEngine::BinOpResult::IntOrFloatValue result;
636             result.setIntValue(lhs.intValue % rhs.intValue);
637             return result;
638         }
639     
640         throw std::runtime_error("Internal error: Unhandled operator;" + binop);
641     }

643     ExprEngine::BinOpResult::IntOrFloatValue ExprEngine::BinOpResult::evaluateOperand(int test, const std::map<ExprEngine::ValuePtr, int> &valueBit, ExprEngine::ValuePtr value) const
644     {
645         if (!value)
646             throw std::runtime_error("Internal error: null value");
647     
648         auto binOpResult = std::dynamic_pointer_cast<ExprEngine::BinOpResult>(value);
649         if (binOpResult)
650             return binOpResult->evaluate(test, valueBit);
651     
652         auto it = valueBit.find(value);
653         if (it == valueBit.end())
654             throw std::runtime_error("Internal error: valueBit not set properly");
655     
656         bool valueType = test & (1 << it->second);
657         if (auto intRange = std::dynamic_pointer_cast<IntRange>(value)) {
658             ExprEngine::BinOpResult::IntOrFloatValue result;
659             result.setIntValue(valueType ? intRange->minValue : intRange->maxValue);
660             return result;
661         }
662         if (auto floatRange = std::dynamic_pointer_cast<FloatRange>(value)) {
663             ExprEngine::BinOpResult::IntOrFloatValue result;
664             result.setFloatValue(valueType ? floatRange->minValue : floatRange->maxValue);
665             return result;
666         }
667         if (auto integerTruncation = std::dynamic_pointer_cast<IntegerTruncation>(value)) {
668             ExprEngine::BinOpResult::IntOrFloatValue result(evaluateOperand(test, valueBit, integerTruncation->inputValue));
669             if (result.isFloat())
670                 result.setIntValue(truncateInt(result.floatValue, integerTruncation->bits, integerTruncation->sign));
671             else
672                 result.setIntValue(truncateInt(result.intValue, integerTruncation->bits, integerTruncation->sign));
673             return result;
674         }
675         throw std::runtime_error("Internal error: Unhandled value:" + std::to_string((int)value->type));
676     }

679     static int getIntBitsFromValueType(const ValueType *vt, const cppcheck::Platform &platform)
680     {
681         if (!vt)
682             return 0;
683     
684         switch (vt->type) {
685         case ValueType::Type::BOOL:
686             return 1;
687         case ValueType::Type::CHAR:
688             return platform.char_bit;
689         case ValueType::Type::SHORT:
690             return platform.short_bit;
691         case ValueType::Type::INT:
692             return platform.int_bit;
693         case ValueType::Type::LONG:
694             return platform.long_bit;
695         case ValueType::Type::LONGLONG:
696             return platform.long_long_bit;
697         default:
698             return 0;
699         };
700     }

702     static ExprEngine::ValuePtr getValueRangeFromValueType(const std::string &name, const ValueType *vt, const cppcheck::Platform &platform)
703     {
704         if (!vt || !(vt->isIntegral() || vt->isFloat()) || vt->pointer)
705             return ExprEngine::ValuePtr();
706     
707         int bits = getIntBitsFromValueType(vt, platform);
708         if (bits == 1) {
709             return std::make_shared<ExprEngine::IntRange>(name, 0, 1);
710         } else if (bits > 1) {
711             if (vt->sign == ValueType::Sign::UNSIGNED) {
712                 return std::make_shared<ExprEngine::IntRange>(name, 0, ((int128_t)1 << bits) - 1);
713             } else {
714                 return std::make_shared<ExprEngine::IntRange>(name, -((int128_t)1 << (bits - 1)), ((int128_t)1 << (bits - 1)) - 1);
715             }
716         }
717     
718         switch (vt->type) {
719         case ValueType::Type::FLOAT:
720             return std::make_shared<ExprEngine::FloatRange>(name, std::numeric_limits<float>::min(), std::numeric_limits<float>::max());
721         case ValueType::Type::DOUBLE:
722             return std::make_shared<ExprEngine::FloatRange>(name, std::numeric_limits<double>::min(), std::numeric_limits<double>::max());
723         case ValueType::Type::LONGDOUBLE:
724             return std::make_shared<ExprEngine::FloatRange>(name, std::numeric_limits<long double>::min(), std::numeric_limits<long double>::max());
725         default:
726             return ExprEngine::ValuePtr();
727         };
728     }

748     static ExprEngine::ValuePtr truncateValue(ExprEngine::ValuePtr val, const ValueType *valueType, Data &data)
749     {
750         if (!valueType)
751             return val;
752         if (valueType->pointer != 0)
753             return val;
754         if (!valueType->isIntegral())
755             return val; // TODO
756     
757         int bits = getIntBitsFromValueType(valueType, *data.settings);
758         if (bits == 0)
759             // TODO
760             return val;
761     
762         if (auto range = std::dynamic_pointer_cast<ExprEngine::IntRange>(val)) {
763             if (range->minValue == range->maxValue) {
764                 int128_t newValue = truncateInt(range->minValue, bits, valueType->sign == ValueType::Sign::SIGNED ? 's' : 'u');
765                 if (newValue == range->minValue)
766                     return val;
767                 return std::make_shared<ExprEngine::IntRange>(ExprEngine::str(newValue), newValue, newValue);
768             }
769             if (auto typeRange = getValueRangeFromValueType("", valueType, *data.settings)) {
770                 auto typeIntRange = std::dynamic_pointer_cast<ExprEngine::IntRange>(typeRange);
771                 if (typeIntRange) {
772                     if (range->minValue >= typeIntRange->minValue && range->maxValue <= typeIntRange->maxValue)
773                         return val;
774                 }
775             }
776     
777             return std::make_shared<ExprEngine::IntegerTruncation>(data.getNewSymbolName(), val, bits, valueType->sign == ValueType::Sign::SIGNED ? 's' : 'u');
778         }
779         // TODO
780         return val;
781     }

783     static ExprEngine::ValuePtr executeAssign(const Token *tok, Data &data)
784     {
785         ExprEngine::ValuePtr rhsValue = executeExpression(tok->astOperand2(), data);
786         ExprEngine::ValuePtr assignValue;
787         if (tok->str() == "=")
788             assignValue = rhsValue;
789         else {
790             // "+=" => "+"
791             std::string binop(tok->str());
792             binop = binop.substr(0, binop.size() - 1);
793             ExprEngine::ValuePtr lhsValue = executeExpression(tok->astOperand1(), data);
794             assignValue = simplifyValue(std::make_shared<ExprEngine::BinOpResult>(binop, lhsValue, rhsValue));
795         }
796     
797         const Token *lhsToken = tok->astOperand1();
798         assignValue = truncateValue(assignValue, lhsToken->valueType(), data);
799         call(data.callbacks, tok, assignValue);
800     
801         if (lhsToken->varId() > 0) {
802             data.assignValue(lhsToken, lhsToken->varId(), assignValue);
803         } else if (lhsToken->str() == "[") {
804             auto arrayValue = data.getArrayValue(lhsToken->astOperand1());
805             if (arrayValue) {
806                 // Is it array initialization?
807                 const Token *arrayInit = lhsToken->astOperand1();
808                 if (arrayInit && arrayInit->variable() && arrayInit->variable()->nameToken() == arrayInit) {
809                     if (assignValue->type == ExprEngine::ValueType::StringLiteralValue)
810                         arrayValue->assign(ExprEngine::ValuePtr(), assignValue);
811                 } else {
812                     auto indexValue = executeExpression(lhsToken->astOperand2(), data);
813                     arrayValue->assign(indexValue, assignValue);
814                 }
815             }
816         } else if (lhsToken->isUnaryOp("*")) {
817             auto pval = executeExpression(lhsToken->astOperand1(), data);
818             if (pval && pval->type == ExprEngine::ValueType::AddressOfValue) {
819                 auto val = std::dynamic_pointer_cast<ExprEngine::AddressOfValue>(pval);
820                 if (val)
821                     data.assignValue(lhsToken, val->varId, assignValue);
822             } else if (pval && pval->type == ExprEngine::ValueType::BinOpResult) {
823                 auto b = std::dynamic_pointer_cast<ExprEngine::BinOpResult>(pval);
824                 if (b && b->binop == "+") {
825                     std::shared_ptr<ExprEngine::ArrayValue> arr;
826                     ExprEngine::ValuePtr offset;
827                     if (b->op1->type == ExprEngine::ValueType::ArrayValue) {
828                         arr = std::dynamic_pointer_cast<ExprEngine::ArrayValue>(b->op1);
829                         offset = b->op2;
830                     } else {
831                         arr = std::dynamic_pointer_cast<ExprEngine::ArrayValue>(b->op2);
832                         offset = b->op1;
833                     }
834                     if (arr && offset) {
835                         arr->assign(offset, assignValue);
836                     }
837                 }
838             }
839         } else if (Token::Match(lhsToken, ". %name%")) {
840             auto structVal = executeExpression(lhsToken->astOperand1(), data);
841             if (structVal && structVal->type == ExprEngine::ValueType::StructValue)
842                 data.assignStructMember(tok, &*std::static_pointer_cast<ExprEngine::StructValue>(structVal), lhsToken->strAt(1), assignValue);
843         }
844         return assignValue;
845     }

847     static ExprEngine::ValuePtr executeFunctionCall(const Token *tok, Data &data)
848     {
849         std::vector<ExprEngine::ValuePtr> argValues;
850         for (const Token *argtok : getArguments(tok)) {
851             auto val = executeExpression(argtok, data);
852             argValues.push_back(val);
853             if (!argtok->valueType() || (argtok->valueType()->constness & 1) == 1)
854                 continue;
855             if (auto arrayValue = std::dynamic_pointer_cast<ExprEngine::ArrayValue>(val)) {
856                 ValueType vt(*argtok->valueType());
857                 vt.pointer = 0;
858                 auto anyVal = getValueRangeFromValueType(data.getNewSymbolName(), &vt, *data.settings);
859                 arrayValue->assign(ExprEngine::ValuePtr(), anyVal);
860             } else if (auto addressOf = std::dynamic_pointer_cast<ExprEngine::AddressOfValue>(val)) {
861                 ValueType vt(*argtok->valueType());
862                 vt.pointer = 0;
863                 if (vt.isIntegral() && argtok->valueType()->pointer == 1)
864                     data.assignValue(argtok, addressOf->varId, getValueRangeFromValueType(data.getNewSymbolName(), &vt, *data.settings));
865             }
866         }
867     
868         if (!tok->valueType() && tok->astParent())
869             throw std::runtime_error("Expression '" + tok->expressionString() + "' has unknown type!");
870     
871         auto val = getValueRangeFromValueType(data.getNewSymbolName(), tok->valueType(), *data.settings);
872         call(data.callbacks, tok, val);
873         return val;
874     }

876     static ExprEngine::ValuePtr executeArrayIndex(const Token *tok, Data &data)
877     {
878         auto arrayValue = data.getArrayValue(tok->astOperand1());
879         if (arrayValue) {
880             auto indexValue = executeExpression(tok->astOperand2(), data);
881             auto conditionalValues = arrayValue->read(indexValue);
882             for (auto value: conditionalValues)
883                 call(data.callbacks, tok, value.second);
884             if (conditionalValues.size() == 1 && !conditionalValues[0].first)
885                 return conditionalValues[0].second;
886             return std::make_shared<ExprEngine::ConditionalValue>(data.getNewSymbolName(), conditionalValues);
887         }
888     
889         // TODO: Pointer value..
890         executeExpression(tok->astOperand1(), data);
891         executeExpression(tok->astOperand2(), data);
892     
893         return ExprEngine::ValuePtr();
894     }

896     static ExprEngine::ValuePtr executeCast(const Token *tok, Data &data)
897     {
898         const Token *expr = tok->astOperand2() ? tok->astOperand2() : tok->astOperand1();
899     
900         auto val = executeExpression(expr, data);
901     
902         if (expr->valueType() && expr->valueType()->type == ::ValueType::Type::VOID && expr->valueType()->pointer > 0) {
903             ::ValueType vt(*tok->valueType());
904             vt.pointer = 0;
905             auto range = getValueRangeFromValueType(data.getNewSymbolName(), &vt, *data.settings);
906     
907             if (tok->valueType()->pointer == 0)
908                 return range;
909     
910             bool uninit = false, null = false;
911             if (val && val->type == ExprEngine::ValueType::PointerValue) {
912                 null = std::static_pointer_cast<ExprEngine::PointerValue>(val)->null;
913                 uninit = std::static_pointer_cast<ExprEngine::PointerValue>(val)->uninitData;
914             }
915     
916             return std::make_shared<ExprEngine::PointerValue>(data.getNewSymbolName(), range, null, uninit);
917         }
918     
919         if (val)
920             // TODO: Cast this..
921             return val;
922     
923         return getValueRangeFromValueType(data.getNewSymbolName(), tok->valueType(), *data.settings);
924     }

926     static ExprEngine::ValuePtr executeDot(const Token *tok, Data &data)
927     {
928         if (!tok->astOperand1() || !tok->astOperand1()->varId())
929             return ExprEngine::ValuePtr();
930         std::shared_ptr<ExprEngine::StructValue> structValue = std::dynamic_pointer_cast<ExprEngine::StructValue>(data.getValue(tok->astOperand1()->varId(), nullptr, nullptr));
931         if (!structValue) {
932             if (tok->originalName() == "->") {
933                 std::shared_ptr<ExprEngine::PointerValue> pointerValue = std::dynamic_pointer_cast<ExprEngine::PointerValue>(data.getValue(tok->astOperand1()->varId(), nullptr, nullptr));
934                 if (pointerValue) {
935                     call(data.callbacks, tok->astOperand1(), pointerValue);
936                     structValue = std::dynamic_pointer_cast<ExprEngine::StructValue>(pointerValue->data);
937                 } else {
938                     call(data.callbacks, tok->astOperand1(), data.getValue(tok->astOperand1()->varId(), nullptr, nullptr));
939                 }
940             }
941             if (!structValue)
942                 return ExprEngine::ValuePtr();
943         }
944         call(data.callbacks, tok->astOperand1(), structValue);
945         return structValue->getValueOfMember(tok->astOperand2()->str());
946     }

1017     static ExprEngine::ValuePtr executeExpression1(const Token *tok, Data &data)
1018     {
1019         if (tok->str() == "return")
1020             return executeReturn(tok, data);
1021     
1022         if (tok->isAssignmentOp())
1023             // TODO: Handle more operators
1024             return executeAssign(tok, data);
1025     
1026         if (tok->astOperand1() && tok->astOperand2() && tok->str() == "[")
1027             return executeArrayIndex(tok, data);
1028     
1029         if (tok->str() == "(") {
1030             if (!tok->isCast())
1031                 return executeFunctionCall(tok, data);
1032             return executeCast(tok, data);
1033         }
1034     
1035         if (tok->str() == ".")
1036             return executeDot(tok, data);
1037     
1038         if (tok->astOperand1() && tok->astOperand2())
1039             return executeBinaryOp(tok, data);
1040     
1041         if (tok->isUnaryOp("&") && Token::Match(tok->astOperand1(), "%var%"))
1042             return executeAddressOf(tok, data);
1043     
1044         if (tok->isUnaryOp("*"))
1045             return executeDeref(tok, data);
1046     
1047         if (tok->varId())
1048             return executeVariable(tok, data);
1049     
1050         if (tok->isName() && tok->hasKnownIntValue())
1051             return executeKnownMacro(tok, data);
1052     
1053         if (tok->isNumber() || tok->tokType() == Token::Type::eChar)
1054             return executeNumber(tok);
1055     
1056         if (tok->tokType() == Token::Type::eString)
1057             return executeStringLiteral(tok, data);
1058     
1059         return ExprEngine::ValuePtr();
1060     }

1069     static void execute(const Token *start, const Token *end, Data &data)
1070     {
1071         for (const Token *tok = start; tok != end; tok = tok->next()) {
1072             if (Token::Match(tok, "[;{}]"))
1073                 data.trackProgramState(tok);
1074     
1075             if (Token::Match(tok, "for|while|switch ("))
1076                 // TODO this is a bailout
1077                 return;
1078     
1079             // Variable declaration..
1080             if (tok->variable() && tok->variable()->nameToken() == tok) {
1081                 if (Token::Match(tok, "%varid% ; %varid% =", tok->varId())) {
1082                     // if variable is not used in assignment rhs then we do not need to create a "confusing" variable value..
1083                     bool foundInRhs = false;
1084                     visitAstNodes(tok->tokAt(3)->astOperand2(), [&](const Token *rhs) {
1085                         if (rhs->varId()==tok->varId()) {
1086                             foundInRhs = true;
1087                             return ChildrenToVisit::done;
1088                         }
1089                         return ChildrenToVisit::op1_and_op2;
1090                     });
1091                     if (!foundInRhs) {
1092                         tok = tok->tokAt(2);
1093                         continue;
1094                     }
1095                 }
1096                 if (tok->variable()->isArray()) {
1097                     data.assignValue(tok, tok->varId(), std::make_shared<ExprEngine::ArrayValue>(&data, tok->variable()));
1098                     if (Token::Match(tok, "%name% ["))
1099                         tok = tok->linkAt(1);
1100                 } else if (Token::Match(tok, "%var% ;"))
1101                     data.assignValue(tok, tok->varId(), createVariableValue(*tok->variable(), data));
1102             } else if (!tok->astParent() && (tok->astOperand1() || tok->astOperand2()))
1103                 executeExpression(tok, data);
1104     
1105             else if (Token::simpleMatch(tok, "if (")) {
1106                 const Token *cond = tok->next()->astOperand2();
1107                 /*const ExprEngine::ValuePtr condValue =*/ executeExpression(cond,data);
1108                 std::vector<Data> trueData = data.getData(cond, true);
1109                 std::vector<Data> falseData = data.getData(cond, false);
1110                 const Token *thenStart = tok->linkAt(1)->next();
1111                 const Token *thenEnd = thenStart->link();
1112                 for (Data &d : trueData)
1113                     execute(thenStart->next(), end, d);
1114                 if (Token::simpleMatch(thenEnd, "} else {")) {
1115                     const Token *elseStart = thenEnd->tokAt(2);
1116                     for (Data &d : falseData)
1117                         execute(elseStart->next(), end, d);
1118                 } else {
1119                     for (Data &d : falseData)
1120                         execute(thenEnd->next(), end, d);
1121                 }
1122                 return;
1123             }
1124     
1125             if (Token::simpleMatch(tok, "} else {"))
1126                 tok = tok->linkAt(2);
1127         }
1128     }

1141         }

1144     static ExprEngine::ValuePtr createStructVal(const Scope *structScope, bool uninitData, Data &data)
1145     {
1146         if (!structScope)
1147             return ExprEngine::ValuePtr();
1148         std::shared_ptr<ExprEngine::StructValue> structValue = std::make_shared<ExprEngine::StructValue>(data.getNewSymbolName());
1149         auto uninitValue = std::make_shared<ExprEngine::UninitValue>();
1150         for (const Variable &member : structScope->varlist) {
1151             if (uninitData) {
1152                 if (member.isPointer()) {
1153                     structValue->member[member.name()] = uninitValue;
1154                     continue;
1155                 }
1156                 if (member.valueType() && member.valueType()->type >= ::ValueType::Type::CHAR) {
1157                     structValue->member[member.name()] = uninitValue;
1158                     continue;
1159                 }
1160             }
1161             ExprEngine::ValuePtr memberValue = createVariableValue(member, data);
1162             if (memberValue)
1163                 structValue->member[member.name()] = memberValue;
1164         }
1165         return structValue;
1166     }
1167     
1168     static ExprEngine::ValuePtr createVariableValue(const Variable &var, Data &data)
1169     {
1170         if (!var.nameToken())
1171             return ExprEngine::ValuePtr();
1172         const ValueType *valueType = var.valueType();
1173         if (!valueType || valueType->type == ValueType::Type::UNKNOWN_TYPE)
1174             valueType = var.nameToken()->valueType();
1175         if (!valueType || valueType->type == ValueType::Type::UNKNOWN_TYPE) {
1176             // variable with unknown type
1177             if (var.isLocal() && var.isPointer() && !var.isArray())
1178                 return std::make_shared<ExprEngine::UninitValue>();
1179             return ExprEngine::ValuePtr();
1180         }
1181     
1182         if (valueType->pointer > 0) {
1183             if (var.isLocal())
1184                 return std::make_shared<ExprEngine::UninitValue>();
1185             ValueType vt(*valueType);
1186             vt.pointer = 0;
1187             auto range = getValueRangeFromValueType(data.getNewSymbolName(), &vt, *data.settings);
1188             return std::make_shared<ExprEngine::PointerValue>(data.getNewSymbolName(), range, true, true);
1189         }
1190         if (var.isArray())
1191             return std::make_shared<ExprEngine::ArrayValue>(&data, &var);
1192         if (valueType->isIntegral())
1193             return getValueRangeFromValueType(data.getNewSymbolName(), valueType, *data.settings);
1194         if (valueType->type == ValueType::Type::RECORD)
1195             return createStructVal(valueType->typeScope, var.isLocal() && !var.isStatic(), data);
1196         if (valueType->smartPointerType) {
1197             auto structValue = createStructVal(valueType->smartPointerType->classScope, var.isLocal() && !var.isStatic(), data);
1198             return std::make_shared<ExprEngine::PointerValue>(data.getNewSymbolName(), structValue, true, false);
1199         }
1200         if (valueType->container && valueType->container->stdStringLike) {
1201             auto size = std::make_shared<ExprEngine::IntRange>(data.getNewSymbolName(), 0, ~0ULL);
1202             auto value = std::make_shared<ExprEngine::IntRange>(data.getNewSymbolName(), -128, 127);
1203             return std::make_shared<ExprEngine::ArrayValue>(data.getNewSymbolName(), size, value);
1204         }
1205         return ExprEngine::ValuePtr();
1206     }

1208     void ExprEngine::executeFunction(const Scope *functionScope, const Tokenizer *tokenizer, const Settings *settings, const std::vector<ExprEngine::Callback> &callbacks, std::ostream &trace)
1209     {
1210         if (!functionScope->bodyStart)
1211             return;
1212         const Function *function = functionScope->function;
1213         if (!function)
1214             return;
1215         if (functionScope->bodyStart->fileIndex() > 0)
1216             // TODO.. what about functions in headers?
1217             return;
1218     
1219         int symbolValueIndex = 0;
1220         TrackExecution trackExecution;
1221         Data data(&symbolValueIndex, tokenizer, settings, callbacks, &trackExecution);
1222     
1223         for (const Variable &arg : function->argumentList)
1224             data.assignValue(functionScope->bodyStart, arg.declarationId(), createVariableValue(arg, data));
1225     
1226         execute(functionScope->bodyStart, functionScope->bodyEnd, data);
1227     
1228         if (settings->debugVerification) {
1229             // TODO generate better output!!
1230             trackExecution.print(trace);
1231         }
1232     }

1234     void ExprEngine::runChecks(ErrorLogger *errorLogger, const Tokenizer *tokenizer, const Settings *settings)
1235     {
1236         std::function<void(const Token *, const ExprEngine::Value &)> divByZero = [=](const Token *tok, const ExprEngine::Value &value) {
1237             if (!Token::Match(tok->astParent(), "[/%]"))
1238                 return;
1239             if (tok->astParent()->astOperand2() == tok && value.isIntValueInRange(0)) {
1240                 std::list<const Token*> callstack{tok->astParent()};
1241                 ErrorLogger::ErrorMessage errmsg(callstack, &tokenizer->list, Severity::SeverityType::error, "verificationDivByZero", "There is division, cannot determine that there can't be a division by zero.", CWE(369), false);
1242                 errorLogger->reportErr(errmsg);
1243             }
1244         };
1245     
1246         std::function<void(const Token *, const ExprEngine::Value &)> nullPointerDereference = [=](const Token *tok, const ExprEngine::Value &value) {
1247             if (!tok->astParent())
1248                 return;
1249     
1250             // Is pointer dereferenced?
1251             bool deref = false;
1252             deref |= tok->astParent()->isUnaryOp("*");
1253             deref |= Token::simpleMatch(tok->astParent(), "[");
1254             deref |= Token::simpleMatch(tok->astParent(), ".") && tok == tok->astParent()->astOperand1();
1255             if (!deref)
1256                 return;
1257     
1258             // Is this a null pointer value?
1259             try {
1260                 if (auto pointerValue = dynamic_cast<const ExprEngine::PointerValue*>(&value)) {
1261                     if (!pointerValue->null)
1262                         return;
1263                 } else if (!value.isIntValueInRange(0))
1264                     return;
1265             } catch (const std::exception &) {
1266                 return;
1267             }
1268     
1269             std::list<const Token*> callstack{tok->astParent()};
1270             ErrorLogger::ErrorMessage errmsg(callstack, &tokenizer->list, Severity::SeverityType::error, "verificationNullPointerDereference", "There is pointer dereference, cannot determine that the pointer can't be NULL.", CWE(476), false);
1271             errorLogger->reportErr(errmsg);
1272         };
1273     
1274         std::function<void(const Token *, const ExprEngine::Value &)> integerOverflow = [&](const Token *tok, const ExprEngine::Value &value) {
1275             if (!tok->isArithmeticalOp() || !tok->valueType() || !tok->valueType()->isIntegral() || tok->valueType()->pointer > 0)
1276                 return;
1277     
1278             const ExprEngine::BinOpResult *b = dynamic_cast<const ExprEngine::BinOpResult *>(&value);
1279             if (!b)
1280                 return;
1281     
1282             ExprEngine::BinOpResult::IntOrFloatValue minValue,maxValue;
1283             b->getRange(&minValue, &maxValue);
1284             if (minValue.isFloat() || maxValue.isFloat())
1285                 return;
1286     
1287             int bits = getIntBitsFromValueType(tok->valueType(), *settings);
1288             if (bits == 0)
1289                 return;
1290     
1291             if (tok->valueType()->sign == ::ValueType::Sign::SIGNED) {
1292                 int128_t v = (int128_t)1 << (bits - 1);
1293                 if (minValue.intValue >= -v && maxValue.intValue < v)
1294                     return;
1295             } else {
1296                 int128_t v = (int128_t)1 << bits;
1297                 if (minValue.intValue >= 0 && maxValue.intValue < v)
1298                     return;
1299             }
1300     
1301             std::string note;
1302             if (tok->valueType()->sign == ::ValueType::Sign::UNSIGNED)
1303                 note = " Note that unsigned integer overflow is defined and will wrap around.";
1304     
1305             std::list<const Token*> callstack{tok};
1306             ErrorLogger::ErrorMessage errmsg(callstack, &tokenizer->list, Severity::SeverityType::error, "verificationIntegerOverflow", "Integer overflow, " + tok->valueType()->str() + " result." + note, false);
1307             errorLogger->reportErr(errmsg);
1308         };
1309     
1310         std::vector<ExprEngine::Callback> callbacks;
1311         callbacks.push_back(divByZero);
1312         callbacks.push_back(nullPointerDereference);
1313     #ifdef VERIFY_INTEGEROVERFLOW
1314         callbacks.push_back(integerOverflow);
1315     #endif
1316         ExprEngine::executeAllFunctions(tokenizer, settings, callbacks, std::cout);
1317     }
