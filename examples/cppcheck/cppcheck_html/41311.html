237     void MainWindow::handleCLIParams(const QStringList &params)
238     {
239         int index;
240         if (params.contains("-p")) {
241             index = params.indexOf("-p");
242             if ((index + 1) < params.length())
243                 loadProjectFile(params[index + 1]);
244         } else if (params.contains("-l")) {
245             QString logFile;
246             index = params.indexOf("-l");
247             if ((index + 1) < params.length())
248                 logFile = params[index + 1];
249     
250             if (params.contains("-d")) {
251                 QString checkedDir;
252                 index = params.indexOf("-d");
253                 if ((index + 1) < params.length())
254                     checkedDir = params[index + 1];
255     
256                 loadResults(logFile, checkedDir);
257             } else {
258                 loadResults(logFile);
259             }
260         } else if ((index = params.indexOf(QRegExp(".*\\.cppcheck$", Qt::CaseInsensitive), 0)) >= 0 && index < params.length() && QFile(params[index]).exists()) {
261             loadProjectFile(params[index]);
262         } else if ((index = params.indexOf(QRegExp(".*\\.xml$", Qt::CaseInsensitive), 0)) >= 0 && index < params.length() && QFile(params[index]).exists()) {
263             loadResults(params[index],QDir::currentPath());
264         } else
265             doAnalyzeFiles(params);
266     }

402     void MainWindow::doAnalyzeProject(ImportProject p, const bool checkLibrary, const bool checkConfiguration)
403     {
404         clearResults();
405     
406         mIsLogfileLoaded = false;
407         if (mProjectFile) {
408             std::vector<std::string> v;
409             foreach (const QString &i, mProjectFile->getExcludedPaths()) {
410                 v.push_back(i.toStdString());
411             }
412             p.ignorePaths(v);
413     
414             if (!mProjectFile->getAnalyzeAllVsConfigs()) {
415                 Settings::PlatformType platform = (Settings::PlatformType) mSettings->value(SETTINGS_CHECKED_PLATFORM, 0).toInt();
416                 p.selectOneVsConfig(platform);
417             }
418         } else {
419             enableProjectActions(false);
420         }
421     
422         mUI.mResults->clear(true);
423         mThread->clearFiles();
424     
425         mUI.mResults->checkingStarted(p.fileSettings.size());
426     
427         QDir inf(mCurrentDirectory);
428         const QString checkPath = inf.canonicalPath();
429         setPath(SETTINGS_LAST_CHECK_PATH, checkPath);
430     
431         checkLockDownUI(); // lock UI while checking
432     
433         mUI.mResults->setCheckDirectory(checkPath);
434         Settings checkSettings = getCppcheckSettings();
435         checkSettings.force = false;
436         checkSettings.checkLibrary = checkLibrary;
437         checkSettings.checkConfiguration = checkConfiguration;
438     
439         if (mProjectFile)
440             qDebug() << "Checking project file" << mProjectFile->getFilename();
441     
442         if (!checkSettings.buildDir.empty()) {
443             std::list<std::string> sourcefiles;
444             AnalyzerInformation::writeFilesTxt(checkSettings.buildDir, sourcefiles, p.fileSettings);
445         }
446     
447         //mThread->SetanalyzeProject(true);
448         if (mProjectFile) {
449             mThread->setAddonsAndTools(mProjectFile->getAddonsAndTools(), mSettings->value(SETTINGS_MISRA_FILE).toString());
450             QString clangHeaders = mSettings->value(SETTINGS_VS_INCLUDE_PATHS).toString();
451             mThread->setClangIncludePaths(clangHeaders.split(";"));
452             mThread->setSuppressions(mProjectFile->getSuppressions());
453         }
454         mThread->setProject(p);
455         mThread->check(checkSettings);
456     }

458     void MainWindow::doAnalyzeFiles(const QStringList &files, const bool checkLibrary, const bool checkConfiguration)
459     {
460         if (files.isEmpty()) {
461             return;
462         }
463         clearResults();
464     
465         mIsLogfileLoaded = false;
466         FileList pathList;
467         pathList.addPathList(files);
468         if (mProjectFile) {
469             pathList.addExcludeList(mProjectFile->getExcludedPaths());
470         } else {
471             enableProjectActions(false);
472         }
473         QStringList fileNames = pathList.getFileList();
474     
475         mUI.mResults->clear(true);
476         mThread->clearFiles();
477     
478         if (fileNames.isEmpty()) {
479             QMessageBox msg(QMessageBox::Warning,
480                             tr("Cppcheck"),
481                             tr("No suitable files found to analyze!"),
482                             QMessageBox::Ok,
483                             this);
484             msg.exec();
485             return;
486         }
487     
488         mUI.mResults->checkingStarted(fileNames.count());
489     
490         mThread->setFiles(fileNames);
491         if (mProjectFile && !checkConfiguration)
492             mThread->setAddonsAndTools(mProjectFile->getAddonsAndTools(), mSettings->value(SETTINGS_MISRA_FILE).toString());
493         mThread->setSuppressions(mProjectFile ? mProjectFile->getSuppressions() : QList<Suppressions::Suppression>());
494         QDir inf(mCurrentDirectory);
495         const QString checkPath = inf.canonicalPath();
496         setPath(SETTINGS_LAST_CHECK_PATH, checkPath);
497     
498         checkLockDownUI(); // lock UI while checking
499     
500         mUI.mResults->setCheckDirectory(checkPath);
501         Settings checkSettings = getCppcheckSettings();
502         checkSettings.checkLibrary = checkLibrary;
503         checkSettings.checkConfiguration = checkConfiguration;
504     
505         if (mProjectFile)
506             qDebug() << "Checking project file" << mProjectFile->getFilename();
507     
508         if (!checkSettings.buildDir.empty()) {
509             std::list<std::string> sourcefiles;
510             foreach (QString s, fileNames)
511                 sourcefiles.push_back(s.toStdString());
512             AnalyzerInformation::writeFilesTxt(checkSettings.buildDir, sourcefiles, checkSettings.project.fileSettings);
513         }
514     
515         mThread->setCheckFiles(true);
516         mThread->check(checkSettings);
517     }

521         // Initialize dummy ThreadResult as ErrorLogger

523         result.setFiles(QStringList(filename));
524         connect(&result, SIGNAL(progress(int, const QString&)),
525                 mUI.mResults, SLOT(progress(int, const QString&)));
526         connect(&result, SIGNAL(error(const ErrorItem &)),
527                 mUI.mResults, SLOT(error(const ErrorItem &)));
528         connect(&result, SIGNAL(log(const QString &)),
529                 mUI.mResults, SLOT(log(const QString &)));
530         connect(&result, SIGNAL(debugError(const ErrorItem &)),
531                 mUI.mResults, SLOT(debugError(const ErrorItem &)));
532     
533         // Create CppCheck instance
534         CppCheck cppcheck(result, true);
535         cppcheck.settings() = getCppcheckSettings();

537         // Check
538         checkLockDownUI();
539         clearResults();

542         analysisDone();
543     
544         // Expand results
545         if (mUI.mResults->hasVisibleResults())
546             mUI.mResults->expandAllResults();

716     Library::Error MainWindow::loadLibrary(Library *library, const QString &filename)
717     {
718         Library::Error ret;
719     
720         // Try to load the library from the project folder..
721         if (mProjectFile) {
722             QString path = QFileInfo(mProjectFile->getFilename()).canonicalPath();
723             ret = library->load(nullptr, (path+"/"+filename).toLatin1());
724             if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
725                 return ret;
726         }
727     
728         // Try to load the library from the application folder..
729         const QString appPath = QFileInfo(QCoreApplication::applicationFilePath()).canonicalPath();
730         ret = library->load(nullptr, (appPath+"/"+filename).toLatin1());
731         if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
732             return ret;
733         ret = library->load(nullptr, (appPath+"/cfg/"+filename).toLatin1());
734         if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
735             return ret;
736     
737     #ifdef FILESDIR
738         // Try to load the library from FILESDIR/cfg..
739         const QString filesdir = FILESDIR;
740         if (!filesdir.isEmpty()) {
741             ret = library->load(nullptr, (filesdir+"/cfg/"+filename).toLatin1());
742             if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
743                 return ret;
744             ret = library->load(nullptr, (filesdir+filename).toLatin1());
745             if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
746                 return ret;
747         }
748     #endif
749     
750         // Try to load the library from the cfg subfolder..
751         const QString datadir = mSettings->value("DATADIR", QString()).toString();
752         if (!datadir.isEmpty()) {
753             ret = library->load(nullptr, (datadir+"/"+filename).toLatin1());
754             if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
755                 return ret;
756             ret = library->load(nullptr, (datadir+"/cfg/"+filename).toLatin1());
757             if (ret.errorcode != Library::ErrorCode::FILE_NOT_FOUND)
758                 return ret;
759         }
760     
761         return ret;
762     }

764     bool MainWindow::tryLoadLibrary(Library *library, QString filename)
765     {
766         const Library::Error error = loadLibrary(library, filename);
767         if (error.errorcode != Library::ErrorCode::OK) {
768             if (error.errorcode == Library::UNKNOWN_ELEMENT) {
769                 QMessageBox::information(this, tr("Information"), tr("The library '%1' contains unknown elements:\n%2").arg(filename).arg(error.reason.c_str()));
770                 return true;
771             }
772     
773             QString errmsg;
774             switch (error.errorcode) {
775             case Library::ErrorCode::OK:
776                 break;
777             case Library::ErrorCode::FILE_NOT_FOUND:
778                 errmsg = tr("File not found");
779                 break;
780             case Library::ErrorCode::BAD_XML:
781                 errmsg = tr("Bad XML");
782                 break;
783             case Library::ErrorCode::MISSING_ATTRIBUTE:
784                 errmsg = tr("Missing attribute");
785                 break;
786             case Library::ErrorCode::BAD_ATTRIBUTE_VALUE:
787                 errmsg = tr("Bad attribute value");
788                 break;
789             case Library::ErrorCode::UNSUPPORTED_FORMAT:
790                 errmsg = tr("Unsupported format");
791                 break;
792             case Library::ErrorCode::DUPLICATE_PLATFORM_TYPE:
793                 errmsg = tr("Duplicate platform type");
794                 break;
795             case Library::ErrorCode::PLATFORM_TYPE_REDEFINED:
796                 errmsg = tr("Platform type redefined");
797                 break;
798             case Library::ErrorCode::UNKNOWN_ELEMENT:
799                 errmsg = tr("Unknown element");
800                 break;
801             default:
802                 errmsg = tr("Unknown issue");
803                 break;
804             }
805             if (!error.reason.empty())
806                 errmsg += " '" + QString::fromStdString(error.reason) + "'";
807             QMessageBox::information(this, tr("Information"), tr("Failed to load the selected library '%1'.\n%2").arg(filename).arg(errmsg));
808             return false;
809         }
810         return true;
811     }

813     Settings MainWindow::getCppcheckSettings()
814     {
815         saveSettings(); // Save settings
816     
817         Settings result;
818     
819         const bool std = tryLoadLibrary(&result.library, "std.cfg");
820         bool posix = true;
821         if (result.posix())
822             posix = tryLoadLibrary(&result.library, "posix.cfg");
823         bool windows = true;
824         if (result.isWindowsPlatform())
825             windows = tryLoadLibrary(&result.library, "windows.cfg");
826     
827         if (!std || !posix || !windows)
828             QMessageBox::critical(this, tr("Error"), tr("Failed to load %1. Your Cppcheck installation is broken. You can use --data-dir=<directory> at the command line to specify where this file is located. Please note that --data-dir is supposed to be used by installation scripts and therefore the GUI does not start when it is used, all that happens is that the setting is configured.").arg(!std ? "std.cfg" : !posix ? "posix.cfg" : "windows.cfg"));
829     
830         // If project file loaded, read settings from it
831         if (mProjectFile) {
832             QStringList dirs = mProjectFile->getIncludeDirs();
833             addIncludeDirs(dirs, result);
834     
835             const QStringList defines = mProjectFile->getDefines();
836             foreach (QString define, defines) {
837                 if (!result.userDefines.empty())
838                     result.userDefines += ";";
839                 result.userDefines += define.toStdString();
840             }
841     
842             const QStringList undefines = mProjectFile->getUndefines();
843             foreach (QString undefine, undefines)
844                 result.userUndefs.insert(undefine.toStdString());
845     
846             const QStringList libraries = mProjectFile->getLibraries();
847             foreach (QString library, libraries) {
848                 const QString filename = library + ".cfg";
849                 tryLoadLibrary(&result.library, filename);
850             }
851     
852             foreach (const Suppressions::Suppression &suppression, mProjectFile->getSuppressions()) {
853                 result.nomsg.addSuppression(suppression);
854             }
855     
856             // Only check the given -D configuration
857             if (!defines.isEmpty())
858                 result.maxConfigs = 1;
859     
860             const QString &buildDir = mProjectFile->getBuildDir();
861             if (!buildDir.isEmpty()) {
862                 if (QDir(buildDir).isAbsolute()) {
863                     result.buildDir = buildDir.toStdString();
864                 } else {
865                     QString prjpath = QFileInfo(mProjectFile->getFilename()).absolutePath();
866                     result.buildDir = (prjpath + '/' + buildDir).toStdString();
867                 }
868             }
869     
870             const QString platform = mProjectFile->getPlatform();
871             if (platform.endsWith(".xml")) {
872                 const QString applicationFilePath = QCoreApplication::applicationFilePath();
873                 result.loadPlatformFile(applicationFilePath.toStdString().c_str(), platform.toStdString());
874             } else {
875                 for (int i = cppcheck::Platform::Native; i <= cppcheck::Platform::Unix64; i++) {
876                     const cppcheck::Platform::PlatformType p = (cppcheck::Platform::PlatformType)i;
877                     if (platform == cppcheck::Platform::platformString(p)) {
878                         result.platform(p);
879                         break;
880                     }
881                 }
882             }
883     
884             result.maxCtuDepth = mProjectFile->getMaxCtuDepth();
885             result.checkHeaders = mProjectFile->getCheckHeaders();
886             result.checkUnusedTemplates = mProjectFile->getCheckUnusedTemplates();
887             result.safeChecks.classes = mProjectFile->getSafeChecks().classes;
888             result.safeChecks.externalFunctions = mProjectFile->getSafeChecks().externalFunctions;
889             result.safeChecks.internalFunctions = mProjectFile->getSafeChecks().internalFunctions;
890             result.safeChecks.externalVariables = mProjectFile->getSafeChecks().externalVariables;
891             foreach (QString s, mProjectFile->getCheckUnknownFunctionReturn())
892                 result.checkUnknownFunctionReturn.insert(s.toStdString());
893         }
894     
895         // Include directories (and files) are searched in listed order.
896         // Global include directories must be added AFTER the per project include
897         // directories so per project include directories can override global ones.
898         const QString globalIncludes = mSettings->value(SETTINGS_GLOBAL_INCLUDE_PATHS).toString();
899         if (!globalIncludes.isEmpty()) {
900             QStringList includes = globalIncludes.split(";");
901             addIncludeDirs(includes, result);
902         }
903     
904         result.addEnabled("warning");
905         result.addEnabled("style");
906         result.addEnabled("performance");
907         result.addEnabled("portability");
908         result.addEnabled("information");
909         result.addEnabled("missingInclude");
910         if (!result.buildDir.empty())
911             result.addEnabled("unusedFunction");
912         result.debugwarnings = mSettings->value(SETTINGS_SHOW_DEBUG_WARNINGS, false).toBool();
913         result.quiet = false;
914         result.verbose = true;
915         result.force = mSettings->value(SETTINGS_CHECK_FORCE, 1).toBool();
916         result.xml = false;
917         result.jobs = mSettings->value(SETTINGS_CHECK_THREADS, 1).toInt();
918         result.inlineSuppressions = mSettings->value(SETTINGS_INLINE_SUPPRESSIONS, false).toBool();
919         result.inconclusive = mSettings->value(SETTINGS_INCONCLUSIVE_ERRORS, false).toBool();
920         if (!mProjectFile || result.platformType == cppcheck::Platform::Unspecified)
921             result.platform((cppcheck::Platform::PlatformType) mSettings->value(SETTINGS_CHECKED_PLATFORM, 0).toInt());
922         result.standards.setCPP(mSettings->value(SETTINGS_STD_CPP, QString()).toString().toStdString());
923         result.standards.setC(mSettings->value(SETTINGS_STD_C, QString()).toString().toStdString());
924         result.enforcedLang = (Settings::Language)mSettings->value(SETTINGS_ENFORCED_LANGUAGE, 0).toInt();
925     
926         if (result.jobs <= 1) {
927             result.jobs = 1;
928         }
929     
930         result.terminate(false);
931     
932         return result;
933     }

1472     void MainWindow::analyzeProject(const ProjectFile *projectFile, const bool checkLibrary, const bool checkConfiguration)
1473     {
1474         Settings::terminate(false);
1475     
1476         QFileInfo inf(projectFile->getFilename());
1477         const QString rootpath = projectFile->getRootPath();
1478     
1479         QDir::setCurrent(inf.absolutePath());
1480     
1481         mThread->setAddonsAndTools(projectFile->getAddonsAndTools(), mSettings->value(SETTINGS_MISRA_FILE).toString());
1482         mUI.mResults->setTags(projectFile->getTags());
1483     
1484         // If the root path is not given or is not "current dir", use project
1485         // file's location directory as root path
1486         if (rootpath.isEmpty() || rootpath == ".")
1487             mCurrentDirectory = inf.canonicalPath();
1488         else if (rootpath.startsWith("./"))
1489             mCurrentDirectory = inf.canonicalPath() + rootpath.mid(1);
1490         else
1491             mCurrentDirectory = rootpath;
1492     
1493         if (!projectFile->getBuildDir().isEmpty()) {
1494             QString buildDir = projectFile->getBuildDir();
1495             if (!QDir::isAbsolutePath(buildDir))
1496                 buildDir = inf.canonicalPath() + '/' + buildDir;
1497             if (!QDir(buildDir).exists()) {
1498                 QMessageBox msg(QMessageBox::Critical,
1499                                 tr("Cppcheck"),
1500                                 tr("Build dir '%1' does not exist, create it?").arg(buildDir),
1501                                 QMessageBox::Yes | QMessageBox::No,
1502                                 this);
1503                 if (msg.exec() == QMessageBox::Yes) {
1504                     QDir().mkpath(buildDir);
1505                 }
1506             }
1507         }
1508     
1509         if (!projectFile->getImportProject().isEmpty()) {
1510             ImportProject p;
1511             QString prjfile;
1512     
1513             if (QFileInfo(projectFile->getImportProject()).isAbsolute()) {
1514                 prjfile = projectFile->getImportProject();
1515             } else {
1516                 prjfile = inf.canonicalPath() + '/' + projectFile->getImportProject();
1517             }
1518             try {
1519                 p.import(prjfile.toStdString());
1520             } catch (InternalError &e) {
1521                 QMessageBox msg(QMessageBox::Critical,
1522                                 tr("Cppcheck"),
1523                                 tr("Failed to import '%1', analysis is stopped").arg(prjfile),
1524                                 QMessageBox::Ok,
1525                                 this);
1526                 msg.exec();
1527                 return;
1528             }
1529             doAnalyzeProject(p, checkLibrary, checkConfiguration);
1530             return;
1531         }
1532     
1533         QStringList paths = projectFile->getCheckPaths();
1534     
1535         // If paths not given then check the root path (which may be the project
1536         // file's location, see above). This is to keep the compatibility with
1537         // old "silent" project file loading when we checked the director where the
1538         // project file was located.
1539         if (paths.isEmpty()) {
1540             paths << mCurrentDirectory;
1541         }
1542         doAnalyzeFiles(paths, checkLibrary, checkConfiguration);
1543     }

1681     {
1682         for (int i = 0; i < MaxRecentProjects + 1; i++) {
1683             if (mRecentProjectActs[i] != nullptr)

1686     
1687         QStringList projects = mSettings->value(SETTINGS_MRU_PROJECTS).toStringList();
1688     
1689         // Do a sanity check - remove duplicates and non-existing projects
1690         int removed = projects.removeDuplicates();
1691         for (int i = projects.size() - 1; i >= 0; i--) {
1692             if (!QFileInfo(projects[i]).exists()) {
1693                 projects.removeAt(i);

1696         }
1697     
1698         if (removed)
1699             mSettings->setValue(SETTINGS_MRU_PROJECTS, projects);
1700     
1701         const int numRecentProjects = qMin(projects.size(), (int)MaxRecentProjects);
1702         for (int i = 0; i < numRecentProjects; i++) {
1703             const QString filename = QFileInfo(projects[i]).fileName();
1704             const QString text = QString("&%1 %2").arg(i + 1).arg(filename);

1752     void MainWindow::suppressIds(QStringList ids)
1753     {

1761             std::string id2 = id.toStdString();
1762             for (int i = 0; i < suppressions.size();) {
1763                 if (suppressions[i].errorId == id2)
1764                     suppressions.removeAt(i);
1765                 else
1766                     ++i;
1767             }
1768     
1769             Suppressions::Suppression newSuppression;
1770             newSuppression.errorId = id2;
1771             suppressions << newSuppression;
1772         }
1773     
1774         mProjectFile->setSuppressions(suppressions);
1775         mProjectFile->write();
1776     }
