55         void setDefines() const {
56             ImportProject::FileSettings fs;
57     
58             fs.setDefines("A");
59             ASSERT_EQUALS("A=1", fs.defines);
60     
61             fs.setDefines("A;B;");
62             ASSERT_EQUALS("A=1;B=1", fs.defines);
63     
64             fs.setDefines("A;;B;");
65             ASSERT_EQUALS("A=1;B=1", fs.defines);
66     
67             fs.setDefines("A;;B");
68             ASSERT_EQUALS("A=1;B=1", fs.defines);
69         }

71         void setIncludePaths1() const {
72             ImportProject::FileSettings fs;
73             std::list<std::string> in(1, "../include");
74             std::map<std::string, std::string, cppcheck::stricmp> variables;
75             fs.setIncludePaths("abc/def/", in, variables);
76             ASSERT_EQUALS(1U, fs.includePaths.size());
77             ASSERT_EQUALS("abc/include/", fs.includePaths.front());
78         }

100         void importCompileCommands1() const {
101             const char json[] = "[ { \"directory\": \"/tmp\","
102                                 "\"command\": \"gcc -I/tmp -DFILESDIR=\\\\\\\"/usr/local/share/Cppcheck\\\\\\\" -DTEST1 -DTEST2=2 -o /tmp/src.o -c /tmp/src.c\","
103                                 "\"file\": \"/tmp/src.c\" } ]";
104             std::istringstream istr(json);
105             TestImporter importer;
106             importer.importCompileCommands(istr);
107             ASSERT_EQUALS(1, importer.fileSettings.size());
108             ASSERT_EQUALS("FILESDIR=\"/usr/local/share/Cppcheck\";TEST1=1;TEST2=2", importer.fileSettings.begin()->defines);
109         }

163         void importCppcheckGuiProject() const {
164             const char xml[] = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
165                                "<project version=\"1\">\n"
166                                "    <root name=\".\"/>\n"
167                                "    <builddir>out1</builddir>\n"
168                                "    <analyze-all-vs-configs>true</analyze-all-vs-configs>\n"
169                                "    <includedir>\n"
170                                "        <dir name=\"lib/\"/>\n"
171                                "    </includedir>\n"
172                                "    <paths>\n"
173                                "        <dir name=\"cli/\"/>\n"
174                                "    </paths>\n"
175                                "    <exclude>\n"
176                                "        <path name=\"gui/temp/\"/>\n"
177                                "    </exclude>\n"
178                                "</project>\n";
179             std::istringstream istr(xml);
180             Settings s;
181             TestImporter project;
182             ASSERT_EQUALS(true, project.importCppcheckGuiProject(istr, &s));
183             ASSERT_EQUALS(1, project.guiProject.pathNames.size());
184             ASSERT_EQUALS("cli/", project.guiProject.pathNames[0]);
185             ASSERT_EQUALS(1, s.includePaths.size());
186             ASSERT_EQUALS("lib/", s.includePaths.front());
187         }
