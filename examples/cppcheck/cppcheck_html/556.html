35     static std::vector<std::string> getnames(const char *names)
36     {
37         std::vector<std::string> ret;
38         while (const char *p = std::strchr(names,',')) {
39             ret.emplace_back(names, p-names);
40             names = p + 1;
41         }
42         ret.emplace_back(names);
43         return ret;
44     }

46     static void gettokenlistfromvalid(const std::string& valid, TokenList& tokenList)
47     {
48         std::istringstream istr(valid + ',');
49         tokenList.createTokens(istr);
50         for (Token *tok = tokenList.front(); tok; tok = tok->next()) {
51             if (Token::Match(tok,"- %num%")) {
52                 tok->str("-" + tok->strAt(1));
53                 tok->deleteNext();
54             }
55         }
56     }

62     Library::Error Library::load(const char exename[], const char path[])
63     {
64         if (std::strchr(path,',') != nullptr) {
65             std::string p(path);
66             for (;;) {
67                 const std::string::size_type pos = p.find(',');
68                 if (pos == std::string::npos)
69                     break;
70                 const Error &e = load(exename, p.substr(0,pos).c_str());
71                 if (e.errorcode != OK)
72                     return e;
73                 p = p.substr(pos+1);
74             }
75             if (!p.empty())
76                 return load(exename, p.c_str());
77             return Error();
78         }
79     
80         std::string absolute_path;
81         // open file..
82         tinyxml2::XMLDocument doc;
83         tinyxml2::XMLError error = doc.LoadFile(path);
84         if (error == tinyxml2::XML_ERROR_FILE_READ_ERROR && Path::getFilenameExtension(path).empty())
85             // Reading file failed, try again...
86             error = tinyxml2::XML_ERROR_FILE_NOT_FOUND;
87         if (error == tinyxml2::XML_ERROR_FILE_NOT_FOUND) {
88             // failed to open file.. is there no extension?
89             std::string fullfilename(path);
90             if (Path::getFilenameExtension(fullfilename).empty()) {
91                 fullfilename += ".cfg";
92                 error = doc.LoadFile(fullfilename.c_str());
93                 if (error != tinyxml2::XML_ERROR_FILE_NOT_FOUND)
94                     absolute_path = Path::getAbsoluteFilePath(fullfilename);
95             }
96     
97             std::list<std::string> cfgfolders;
98     #ifdef FILESDIR
99             cfgfolders.emplace_back(FILESDIR "/cfg");
100     #endif
101             if (exename) {
102                 const std::string exepath(Path::fromNativeSeparators(Path::getPathFromFilename(exename)));
103                 cfgfolders.push_back(exepath + "cfg");
104                 cfgfolders.push_back(exepath);
105             }
106     
107             while (error == tinyxml2::XML_ERROR_FILE_NOT_FOUND && !cfgfolders.empty()) {
108                 const std::string cfgfolder(cfgfolders.front());
109                 cfgfolders.pop_front();
110                 const char *sep = (!cfgfolder.empty() && endsWith(cfgfolder,'/') ? "" : "/");
111                 const std::string filename(cfgfolder + sep + fullfilename);
112                 error = doc.LoadFile(filename.c_str());
113                 if (error != tinyxml2::XML_ERROR_FILE_NOT_FOUND)
114                     absolute_path = Path::getAbsoluteFilePath(filename);
115             }
116         } else
117             absolute_path = Path::getAbsoluteFilePath(path);
118     
119         if (error == tinyxml2::XML_SUCCESS) {
120             if (mFiles.find(absolute_path) == mFiles.end()) {
121                 Error err = load(doc);
122                 if (err.errorcode == OK)
123                     mFiles.insert(absolute_path);
124                 return err;
125             }
126     
127             return Error(OK); // ignore duplicates
128         }
129     
130         if (error == tinyxml2::XML_ERROR_FILE_NOT_FOUND)
131             return Error(FILE_NOT_FOUND);
132         else {
133             doc.PrintError();
134             return Error(BAD_XML);
135         }
136     }

144     Library::Error Library::load(const tinyxml2::XMLDocument &doc)
145     {
146         const tinyxml2::XMLElement * const rootnode = doc.FirstChildElement();
147     
148         if (rootnode == nullptr) {
149             doc.PrintError();
150             return Error(BAD_XML);
151         }
152     
153         if (strcmp(rootnode->Name(),"def") != 0)
154             return Error(UNSUPPORTED_FORMAT, rootnode->Name());
155     
156         const char* format_string = rootnode->Attribute("format");
157         int format = 1; // Assume format version 1 if nothing else is specified (very old .cfg files had no 'format' attribute)
158         if (format_string)
159             format = atoi(format_string);
160     
161         if (format > 2 || format <= 0)
162             return Error(UNSUPPORTED_FORMAT);
163     
164         std::set<std::string> unknown_elements;
165     
166         for (const tinyxml2::XMLElement *node = rootnode->FirstChildElement(); node; node = node->NextSiblingElement()) {
167             const std::string nodename = node->Name();
168             if (nodename == "memory" || nodename == "resource") {
169                 // get allocationId to use..
170                 int allocationId = 0;
171                 for (const tinyxml2::XMLElement *memorynode = node->FirstChildElement(); memorynode; memorynode = memorynode->NextSiblingElement()) {
172                     if (strcmp(memorynode->Name(),"dealloc")==0) {
173                         const std::map<std::string, AllocFunc>::const_iterator it = mDealloc.find(memorynode->GetText());
174                         if (it != mDealloc.end()) {
175                             allocationId = it->second.groupId;
176                             break;
177                         }
178                     }
179                 }
180                 if (allocationId == 0) {
181                     if (nodename == "memory")
182                         while (!ismemory(++mAllocId));
183                     else
184                         while (!isresource(++mAllocId));
185                     allocationId = mAllocId;
186                 }
187     
188                 // add alloc/dealloc/use functions..
189                 for (const tinyxml2::XMLElement *memorynode = node->FirstChildElement(); memorynode; memorynode = memorynode->NextSiblingElement()) {
190                     const std::string memorynodename = memorynode->Name();
191                     if (memorynodename == "alloc" || memorynodename == "realloc") {
192                         AllocFunc temp = {0};
193                         temp.groupId = allocationId;
194     
195                         if (memorynode->Attribute("init", "false"))
196                             returnuninitdata.insert(memorynode->GetText());
197     
198                         const char *arg = memorynode->Attribute("arg");
199                         if (arg)
200                             temp.arg = atoi(arg);
201                         else
202                             temp.arg = -1;
203     
204                         const char *bufferSize = memorynode->Attribute("buffer-size");
205                         if (!bufferSize)
206                             temp.bufferSize = AllocFunc::BufferSize::none;
207                         else {
208                             if (std::strncmp(bufferSize, "malloc", 6) == 0)
209                                 temp.bufferSize = AllocFunc::BufferSize::malloc;
210                             else if (std::strncmp(bufferSize, "calloc", 6) == 0)
211                                 temp.bufferSize = AllocFunc::BufferSize::calloc;
212                             else if (std::strncmp(bufferSize, "strdup", 6) == 0)
213                                 temp.bufferSize = AllocFunc::BufferSize::strdup;
214                             else
215                                 return Error(BAD_ATTRIBUTE_VALUE, bufferSize);
216                             temp.bufferSizeArg1 = 1;
217                             temp.bufferSizeArg2 = 2;
218                             if (bufferSize[6] == 0) {
219                                 // use default values
220                             } else if (bufferSize[6] == ':' && bufferSize[7] >= '1' && bufferSize[7] <= '5') {
221                                 temp.bufferSizeArg1 = bufferSize[7] - '0';
222                                 if (bufferSize[8] == ',' && bufferSize[9] >= '1' && bufferSize[9] <= '5')
223                                     temp.bufferSizeArg2 = bufferSize[9] - '0';
224                             } else
225                                 return Error(BAD_ATTRIBUTE_VALUE, bufferSize);
226                         }
227     
228                         if (memorynodename == "realloc") {
229                             const char *reallocArg =  memorynode->Attribute("realloc-arg");
230                             if (reallocArg)
231                                 temp.reallocArg = atoi(reallocArg);
232                             else
233                                 temp.reallocArg = 1;
234                         }
235     
236                         if (memorynodename != "realloc")
237                             mAlloc[memorynode->GetText()] = temp;
238                         else
239                             mRealloc[memorynode->GetText()] = temp;
240                     } else if (memorynodename == "dealloc") {
241                         AllocFunc temp = {0};
242                         temp.groupId = allocationId;
243                         const char *arg = memorynode->Attribute("arg");
244                         if (arg)
245                             temp.arg = atoi(arg);
246                         else
247                             temp.arg = 1;
248                         mDealloc[memorynode->GetText()] = temp;
249                     } else if (memorynodename == "use")
250                         functions[memorynode->GetText()].use = true;
251                     else
252                         unknown_elements.insert(memorynodename);
253                 }
254             }
255     
256             else if (nodename == "define") {
257                 const char *name = node->Attribute("name");
258                 if (name == nullptr)
259                     return Error(MISSING_ATTRIBUTE, "name");
260                 const char *value = node->Attribute("value");
261                 if (value == nullptr)
262                     return Error(MISSING_ATTRIBUTE, "value");
263                 defines.push_back(std::string(name) +
264                                   " " +
265                                   value);
266             }
267     
268             else if (nodename == "function") {
269                 const char *name = node->Attribute("name");
270                 if (name == nullptr)
271                     return Error(MISSING_ATTRIBUTE, "name");
272                 for (const std::string &s : getnames(name)) {
273                     const Error &err = loadFunction(node, s, unknown_elements);
274                     if (err.errorcode != ErrorCode::OK)
275                         return err;
276                 }
277             }
278     
279             else if (nodename == "reflection") {
280                 for (const tinyxml2::XMLElement *reflectionnode = node->FirstChildElement(); reflectionnode; reflectionnode = reflectionnode->NextSiblingElement()) {
281                     if (strcmp(reflectionnode->Name(), "call") != 0) {
282                         unknown_elements.insert(reflectionnode->Name());
283                         continue;
284                     }
285     
286                     const char * const argString = reflectionnode->Attribute("arg");
287                     if (!argString)
288                         return Error(MISSING_ATTRIBUTE, "arg");
289     
290                     mReflection[reflectionnode->GetText()] = atoi(argString);
291                 }
292             }
293     
294             else if (nodename == "markup") {
295                 const char * const extension = node->Attribute("ext");
296                 if (!extension)
297                     return Error(MISSING_ATTRIBUTE, "ext");
298                 mMarkupExtensions.insert(extension);
299     
300                 mReportErrors[extension] = (node->Attribute("reporterrors", "true") != nullptr);
301                 mProcessAfterCode[extension] = (node->Attribute("aftercode", "true") != nullptr);
302     
303                 for (const tinyxml2::XMLElement *markupnode = node->FirstChildElement(); markupnode; markupnode = markupnode->NextSiblingElement()) {
304                     const std::string markupnodename = markupnode->Name();
305                     if (markupnodename == "keywords") {
306                         for (const tinyxml2::XMLElement *librarynode = markupnode->FirstChildElement(); librarynode; librarynode = librarynode->NextSiblingElement()) {
307                             if (strcmp(librarynode->Name(), "keyword") == 0) {
308                                 const char* nodeName = librarynode->Attribute("name");
309                                 if (nodeName == nullptr)
310                                     return Error(MISSING_ATTRIBUTE, "name");
311                                 mKeywords[extension].insert(nodeName);
312                             } else
313                                 unknown_elements.insert(librarynode->Name());
314                         }
315                     }
316     
317                     else if (markupnodename == "exported") {
318                         for (const tinyxml2::XMLElement *exporter = markupnode->FirstChildElement(); exporter; exporter = exporter->NextSiblingElement()) {
319                             if (strcmp(exporter->Name(), "exporter") != 0) {
320                                 unknown_elements.insert(exporter->Name());
321                                 continue;
322                             }
323     
324                             const char * const prefix = exporter->Attribute("prefix");
325                             if (!prefix)
326                                 return Error(MISSING_ATTRIBUTE, "prefix");
327     
328                             for (const tinyxml2::XMLElement *e = exporter->FirstChildElement(); e; e = e->NextSiblingElement()) {
329                                 const std::string ename = e->Name();
330                                 if (ename == "prefix")
331                                     mExporters[prefix].addPrefix(e->GetText());
332                                 else if (ename == "suffix")
333                                     mExporters[prefix].addSuffix(e->GetText());
334                                 else
335                                     unknown_elements.insert(ename);
336                             }
337                         }
338                     }
339     
340                     else if (markupnodename == "imported") {
341                         for (const tinyxml2::XMLElement *librarynode = markupnode->FirstChildElement(); librarynode; librarynode = librarynode->NextSiblingElement()) {
342                             if (strcmp(librarynode->Name(), "importer") == 0)
343                                 mImporters[extension].insert(librarynode->GetText());
344                             else
345                                 unknown_elements.insert(librarynode->Name());
346                         }
347                     }
348     
349                     else if (markupnodename == "codeblocks") {
350                         for (const tinyxml2::XMLElement *blocknode = markupnode->FirstChildElement(); blocknode; blocknode = blocknode->NextSiblingElement()) {
351                             const std::string blocknodename = blocknode->Name();
352                             if (blocknodename == "block") {
353                                 const char * blockName = blocknode->Attribute("name");
354                                 if (blockName)
355                                     mExecutableBlocks[extension].addBlock(blockName);
356                             } else if (blocknodename == "structure") {
357                                 const char * start = blocknode->Attribute("start");
358                                 if (start)
359                                     mExecutableBlocks[extension].setStart(start);
360                                 const char * end = blocknode->Attribute("end");
361                                 if (end)
362                                     mExecutableBlocks[extension].setEnd(end);
363                                 const char * offset = blocknode->Attribute("offset");
364                                 if (offset)
365                                     mExecutableBlocks[extension].setOffset(atoi(offset));
366                             }
367     
368                             else
369                                 unknown_elements.insert(blocknodename);
370                         }
371                     }
372     
373                     else
374                         unknown_elements.insert(markupnodename);
375                 }
376             }
377     
378             else if (nodename == "container") {
379                 const char* const id = node->Attribute("id");
380                 if (!id)
381                     return Error(MISSING_ATTRIBUTE, "id");
382     
383                 Container& container = containers[id];
384     
385                 const char* const inherits = node->Attribute("inherits");
386                 if (inherits) {
387                     const std::map<std::string, Container>::const_iterator i = containers.find(inherits);
388                     if (i != containers.end())
389                         container = i->second; // Take values from parent and overwrite them if necessary
390                     else
391                         return Error(BAD_ATTRIBUTE_VALUE, inherits);
392                 }
393     
394                 const char* const startPattern = node->Attribute("startPattern");
395                 if (startPattern) {
396                     container.startPattern = startPattern;
397                     container.startPattern2 = container.startPattern + " !!::";
398                 }
399                 const char* const endPattern = node->Attribute("endPattern");
400                 if (endPattern)
401                     container.endPattern = endPattern;
402                 const char* const itEndPattern = node->Attribute("itEndPattern");
403                 if (itEndPattern)
404                     container.itEndPattern = itEndPattern;
405                 const char* const opLessAllowed = node->Attribute("opLessAllowed");
406                 if (opLessAllowed)
407                     container.opLessAllowed = std::string(opLessAllowed) == "true";
408     
409                 for (const tinyxml2::XMLElement *containerNode = node->FirstChildElement(); containerNode; containerNode = containerNode->NextSiblingElement()) {
410                     const std::string containerNodeName = containerNode->Name();
411                     if (containerNodeName == "size" || containerNodeName == "access" || containerNodeName == "other") {
412                         for (const tinyxml2::XMLElement *functionNode = containerNode->FirstChildElement(); functionNode; functionNode = functionNode->NextSiblingElement()) {
413                             if (std::string(functionNode->Name()) != "function") {
414                                 unknown_elements.insert(functionNode->Name());
415                                 continue;
416                             }
417     
418                             const char* const functionName = functionNode->Attribute("name");
419                             if (!functionName)
420                                 return Error(MISSING_ATTRIBUTE, "name");
421     
422                             const char* const action_ptr = functionNode->Attribute("action");
423                             Container::Action action = Container::Action::NO_ACTION;
424                             if (action_ptr) {
425                                 std::string actionName = action_ptr;
426                                 if (actionName == "resize")
427                                     action = Container::Action::RESIZE;
428                                 else if (actionName == "clear")
429                                     action = Container::Action::CLEAR;
430                                 else if (actionName == "push")
431                                     action = Container::Action::PUSH;
432                                 else if (actionName == "pop")
433                                     action = Container::Action::POP;
434                                 else if (actionName == "find")
435                                     action = Container::Action::FIND;
436                                 else if (actionName == "insert")
437                                     action = Container::Action::INSERT;
438                                 else if (actionName == "erase")
439                                     action = Container::Action::ERASE;
440                                 else if (actionName == "change-content")
441                                     action = Container::Action::CHANGE_CONTENT;
442                                 else if (actionName == "change-internal")
443                                     action = Container::Action::CHANGE_INTERNAL;
444                                 else if (actionName == "change")
445                                     action = Container::Action::CHANGE;
446                                 else
447                                     return Error(BAD_ATTRIBUTE_VALUE, actionName);
448                             }
449     
450                             const char* const yield_ptr = functionNode->Attribute("yields");
451                             Container::Yield yield = Container::Yield::NO_YIELD;
452                             if (yield_ptr) {
453                                 std::string yieldName = yield_ptr;
454                                 if (yieldName == "at_index")
455                                     yield = Container::Yield::AT_INDEX;
456                                 else if (yieldName == "item")
457                                     yield = Container::Yield::ITEM;
458                                 else if (yieldName == "buffer")
459                                     yield = Container::Yield::BUFFER;
460                                 else if (yieldName == "buffer-nt")
461                                     yield = Container::Yield::BUFFER_NT;
462                                 else if (yieldName == "start-iterator")
463                                     yield = Container::Yield::START_ITERATOR;
464                                 else if (yieldName == "end-iterator")
465                                     yield = Container::Yield::END_ITERATOR;
466                                 else if (yieldName == "iterator")
467                                     yield = Container::Yield::ITERATOR;
468                                 else if (yieldName == "size")
469                                     yield = Container::Yield::SIZE;
470                                 else if (yieldName == "empty")
471                                     yield = Container::Yield::EMPTY;
472                                 else
473                                     return Error(BAD_ATTRIBUTE_VALUE, yieldName);
474                             }
475     
476                             container.functions[functionName].action = action;
477                             container.functions[functionName].yield = yield;
478                         }
479     
480                         if (containerNodeName == "size") {
481                             const char* const templateArg = containerNode->Attribute("templateParameter");
482                             if (templateArg)
483                                 container.size_templateArgNo = atoi(templateArg);
484                         } else if (containerNodeName == "access") {
485                             const char* const indexArg = containerNode->Attribute("indexOperator");
486                             if (indexArg)
487                                 container.arrayLike_indexOp = std::string(indexArg) == "array-like";
488                         }
489                     } else if (containerNodeName == "type") {
490                         const char* const templateArg = containerNode->Attribute("templateParameter");
491                         if (templateArg)
492                             container.type_templateArgNo = atoi(templateArg);
493     
494                         const char* const string = containerNode->Attribute("string");
495                         if (string)
496                             container.stdStringLike = std::string(string) == "std-like";
497                         const char* const associative = containerNode->Attribute("associative");
498                         if (associative)
499                             container.stdAssociativeLike = std::string(associative) == "std-like";
500                     } else
501                         unknown_elements.insert(containerNodeName);
502                 }
503             }
504     
505             else if (nodename == "smart-pointer") {
506                 const char *className = node->Attribute("class-name");
507                 if (className)
508                     smartPointers.insert(className);
509             }
510     
511             else if (nodename == "podtype") {
512                 const char * const name = node->Attribute("name");
513                 if (!name)
514                     return Error(MISSING_ATTRIBUTE, "name");
515                 PodType podType = {0};
516                 podType.stdtype = PodType::NO;
517                 const char * const stdtype = node->Attribute("stdtype");
518                 if (stdtype) {
519                     if (std::strcmp(stdtype, "bool") == 0)
520                         podType.stdtype = PodType::BOOL;
521                     else if (std::strcmp(stdtype, "char") == 0)
522                         podType.stdtype = PodType::CHAR;
523                     else if (std::strcmp(stdtype, "short") == 0)
524                         podType.stdtype = PodType::SHORT;
525                     else if (std::strcmp(stdtype, "int") == 0)
526                         podType.stdtype = PodType::INT;
527                     else if (std::strcmp(stdtype, "long") == 0)
528                         podType.stdtype = PodType::LONG;
529                     else if (std::strcmp(stdtype, "long long") == 0)
530                         podType.stdtype = PodType::LONGLONG;
531                 }
532                 const char * const size = node->Attribute("size");
533                 if (size)
534                     podType.size = atoi(size);
535                 const char * const sign = node->Attribute("sign");
536                 if (sign)
537                     podType.sign = *sign;
538                 for (const std::string &s : getnames(name))
539                     mPodTypes[s] = podType;
540             }
541     
542             else if (nodename == "platformtype") {
543                 const char * const type_name = node->Attribute("name");
544                 if (type_name == nullptr)
545                     return Error(MISSING_ATTRIBUTE, "name");
546                 const char *value = node->Attribute("value");
547                 if (value == nullptr)
548                     return Error(MISSING_ATTRIBUTE, "value");
549                 PlatformType type;
550                 type.mType = value;
551                 std::set<std::string> platform;
552                 for (const tinyxml2::XMLElement *typenode = node->FirstChildElement(); typenode; typenode = typenode->NextSiblingElement()) {
553                     const std::string typenodename = typenode->Name();
554                     if (typenodename == "platform") {
555                         const char * const type_attribute = typenode->Attribute("type");
556                         if (type_attribute == nullptr)
557                             return Error(MISSING_ATTRIBUTE, "type");
558                         platform.insert(type_attribute);
559                     } else if (typenodename == "signed")
560                         type._signed = true;
561                     else if (typenodename == "unsigned")
562                         type._unsigned = true;
563                     else if (typenodename == "long")
564                         type._long = true;
565                     else if (typenodename == "pointer")
566                         type._pointer= true;
567                     else if (typenodename == "ptr_ptr")
568                         type._ptr_ptr = true;
569                     else if (typenodename == "const_ptr")
570                         type._const_ptr = true;
571                     else
572                         unknown_elements.insert(typenodename);
573                 }
574                 if (platform.empty()) {
575                     const PlatformType * const type_ptr = platform_type(type_name, emptyString);
576                     if (type_ptr) {
577                         if (*type_ptr == type)
578                             return Error(DUPLICATE_PLATFORM_TYPE, type_name);
579                         return Error(PLATFORM_TYPE_REDEFINED, type_name);
580                     }
581                     mPlatformTypes[type_name] = type;
582                 } else {
583                     for (const std::string &p : platform) {
584                         const PlatformType * const type_ptr = platform_type(type_name, p);
585                         if (type_ptr) {
586                             if (*type_ptr == type)
587                                 return Error(DUPLICATE_PLATFORM_TYPE, type_name);
588                             return Error(PLATFORM_TYPE_REDEFINED, type_name);
589                         }
590                         mPlatforms[p].mPlatformTypes[type_name] = type;
591                     }
592                 }
593             }
594     
595             else
596                 unknown_elements.insert(nodename);
597         }
598         if (!unknown_elements.empty()) {
599             std::string str;
600             for (std::set<std::string>::const_iterator i = unknown_elements.begin(); i != unknown_elements.end();) {
601                 str += *i;
602                 if (++i != unknown_elements.end())
603                     str += ", ";
604             }
605             return Error(UNKNOWN_ELEMENT, str);
606         }
607         return Error(OK);
608     }

610     Library::Error Library::loadFunction(const tinyxml2::XMLElement * const node, const std::string &name, std::set<std::string> &unknown_elements)
611     {
612         if (name.empty())
613             return Error(OK);
614     
615         Function& func = functions[name];
616     
617         for (const tinyxml2::XMLElement *functionnode = node->FirstChildElement(); functionnode; functionnode = functionnode->NextSiblingElement()) {
618             const std::string functionnodename = functionnode->Name();
619             if (functionnodename == "noreturn")
620                 mNoReturn[name] = (strcmp(functionnode->GetText(), "true") == 0);
621             else if (functionnodename == "pure")
622                 func.ispure = true;
623             else if (functionnodename == "const") {
624                 func.ispure = true;
625                 func.isconst = true; // a constant function is pure
626             } else if (functionnodename == "leak-ignore")
627                 func.leakignore = true;
628             else if (functionnodename == "use-retval")
629                 func.useretval = true;
630             else if (functionnodename == "returnValue") {
631                 if (const char *expr = functionnode->GetText())
632                     mReturnValue[name] = expr;
633                 if (const char *type = functionnode->Attribute("type"))
634                     mReturnValueType[name] = type;
635                 if (const char *container = functionnode->Attribute("container"))
636                     mReturnValueContainer[name] = std::atoi(container);
637                 if (const char *unknownReturnValues = functionnode->Attribute("unknownValues")) {
638                     if (std::strcmp(unknownReturnValues, "all") == 0) {
639                         std::vector<MathLib::bigint> values{LLONG_MIN, LLONG_MAX};
640                         mUnknownReturnValues[name] = values;
641                     }
642                 }
643             } else if (functionnodename == "arg") {
644                 const char* argNrString = functionnode->Attribute("nr");
645                 if (!argNrString)
646                     return Error(MISSING_ATTRIBUTE, "nr");
647                 const bool bAnyArg = strcmp(argNrString, "any") == 0;
648                 const bool bVariadicArg = strcmp(argNrString, "variadic") == 0;
649                 const int nr = (bAnyArg || bVariadicArg) ? -1 : std::atoi(argNrString);
650                 ArgumentChecks &ac = func.argumentChecks[nr];
651                 ac.optional  = functionnode->Attribute("default") != nullptr;
652                 ac.variadic = bVariadicArg;
653                 const char * const argDirection = functionnode->Attribute("direction");
654                 if (argDirection) {
655                     const size_t argDirLen = strlen(argDirection);
656                     if (!strncmp(argDirection, "in", argDirLen)) {
657                         ac.direction = ArgumentChecks::Direction::DIR_IN;
658                     } else if (!strncmp(argDirection, "out", argDirLen)) {
659                         ac.direction = ArgumentChecks::Direction::DIR_OUT;
660                     } else if (!strncmp(argDirection, "inout", argDirLen)) {
661                         ac.direction = ArgumentChecks::Direction::DIR_INOUT;
662                     }
663                 }
664                 for (const tinyxml2::XMLElement *argnode = functionnode->FirstChildElement(); argnode; argnode = argnode->NextSiblingElement()) {
665                     const std::string argnodename = argnode->Name();
666                     if (argnodename == "not-bool")
667                         ac.notbool = true;
668                     else if (argnodename == "not-null")
669                         ac.notnull = true;
670                     else if (argnodename == "not-uninit")
671                         ac.notuninit = true;
672                     else if (argnodename == "formatstr")
673                         ac.formatstr = true;
674                     else if (argnodename == "strz")
675                         ac.strz = true;
676                     else if (argnodename == "valid") {
677                         // Validate the validation expression
678                         const char *p = argnode->GetText();
679                         bool error = false;
680                         bool range = false;
681                         bool has_dot = false;
682     
683                         if (!p)
684                             return Error(BAD_ATTRIBUTE_VALUE, "\"\"");
685     
686                         error = *p == '.';
687                         for (; *p; p++) {
688                             if (std::isdigit(*p))
689                                 error |= (*(p+1) == '-');
690                             else if (*p == ':') {
691                                 error |= range | (*(p+1) == '.');
692                                 range = true;
693                                 has_dot = false;
694                             } else if (*p == '-')
695                                 error |= (!std::isdigit(*(p+1)));
696                             else if (*p == ',') {
697                                 range = false;
698                                 error |= *(p+1) == '.';
699                                 has_dot = false;
700                             } else if (*p == '.') {
701                                 error |= has_dot | (!std::isdigit(*(p+1)));
702                                 has_dot = true;
703                             } else
704                                 error = true;
705                         }
706                         if (error)
707                             return Error(BAD_ATTRIBUTE_VALUE, argnode->GetText());
708     
709                         // Set validation expression
710                         ac.valid = argnode->GetText();
711                     }
712     
713                     else if (argnodename == "minsize") {
714                         const char *typeattr = argnode->Attribute("type");
715                         if (!typeattr)
716                             return Error(MISSING_ATTRIBUTE, "type");
717     
718                         ArgumentChecks::MinSize::Type type;
719                         if (strcmp(typeattr,"strlen")==0)
720                             type = ArgumentChecks::MinSize::Type::STRLEN;
721                         else if (strcmp(typeattr,"argvalue")==0)
722                             type = ArgumentChecks::MinSize::Type::ARGVALUE;
723                         else if (strcmp(typeattr,"sizeof")==0)
724                             type = ArgumentChecks::MinSize::Type::SIZEOF;
725                         else if (strcmp(typeattr,"mul")==0)
726                             type = ArgumentChecks::MinSize::Type::MUL;
727                         else if (strcmp(typeattr,"value")==0)
728                             type = ArgumentChecks::MinSize::Type::VALUE;
729                         else
730                             return Error(BAD_ATTRIBUTE_VALUE, typeattr);
731     
732                         if (type == ArgumentChecks::MinSize::Type::VALUE) {
733                             const char *valueattr = argnode->Attribute("value");
734                             if (!valueattr)
735                                 return Error(MISSING_ATTRIBUTE, "value");
736                             long long minsizevalue = 0;
737                             try {
738                                 minsizevalue = MathLib::toLongNumber(valueattr);
739                             } catch (const InternalError&) {
740                                 return Error(BAD_ATTRIBUTE_VALUE, valueattr);
741                             }
742                             if (minsizevalue <= 0)
743                                 return Error(BAD_ATTRIBUTE_VALUE, valueattr);
744                             ac.minsizes.emplace_back(type, 0);
745                             ac.minsizes.back().value = minsizevalue;
746                         } else {
747                             const char *argattr = argnode->Attribute("arg");
748                             if (!argattr)
749                                 return Error(MISSING_ATTRIBUTE, "arg");
750                             if (strlen(argattr) != 1 || argattr[0]<'0' || argattr[0]>'9')
751                                 return Error(BAD_ATTRIBUTE_VALUE, argattr);
752     
753                             ac.minsizes.reserve(type == ArgumentChecks::MinSize::Type::MUL ? 2 : 1);
754                             ac.minsizes.emplace_back(type, argattr[0] - '0');
755                             if (type == ArgumentChecks::MinSize::Type::MUL) {
756                                 const char *arg2attr = argnode->Attribute("arg2");
757                                 if (!arg2attr)
758                                     return Error(MISSING_ATTRIBUTE, "arg2");
759                                 if (strlen(arg2attr) != 1 || arg2attr[0]<'0' || arg2attr[0]>'9')
760                                     return Error(BAD_ATTRIBUTE_VALUE, arg2attr);
761                                 ac.minsizes.back().arg2 = arg2attr[0] - '0';
762                             }
763                         }
764                     }
765     
766                     else if (argnodename == "iterator") {
767                         ac.iteratorInfo.it = true;
768                         const char* str = argnode->Attribute("type");
769                         ac.iteratorInfo.first = str ? (std::strcmp(str, "first") == 0) : false;
770                         ac.iteratorInfo.last = str ? (std::strcmp(str, "last") == 0) : false;
771                         str = argnode->Attribute("container");
772                         ac.iteratorInfo.container = str ? std::atoi(str) : 0;
773                     }
774     
775                     else
776                         unknown_elements.insert(argnodename);
777                 }
778             } else if (functionnodename == "ignorefunction") {
779                 func.ignore = true;
780             } else if (functionnodename == "formatstr") {
781                 func.formatstr = true;
782                 const tinyxml2::XMLAttribute* scan = functionnode->FindAttribute("scan");
783                 const tinyxml2::XMLAttribute* secure = functionnode->FindAttribute("secure");
784                 func.formatstr_scan = scan && scan->BoolValue();
785                 func.formatstr_secure = secure && secure->BoolValue();
786             } else if (functionnodename == "warn") {
787                 WarnInfo wi;
788                 const char* const severity = functionnode->Attribute("severity");
789                 if (severity == nullptr)
790                     return Error(MISSING_ATTRIBUTE, "severity");
791                 wi.severity = Severity::fromString(severity);
792     
793                 const char* const cstd = functionnode->Attribute("cstd");
794                 if (cstd) {
795                     if (!wi.standards.setC(cstd))
796                         return Error(BAD_ATTRIBUTE_VALUE, cstd);
797                 } else
798                     wi.standards.c = Standards::C89;
799     
800                 const char* const cppstd = functionnode->Attribute("cppstd");
801                 if (cppstd) {
802                     if (!wi.standards.setCPP(cppstd))
803                         return Error(BAD_ATTRIBUTE_VALUE, cppstd);
804                 } else
805                     wi.standards.cpp = Standards::CPP03;
806     
807                 const char* const reason = functionnode->Attribute("reason");
808                 const char* const alternatives = functionnode->Attribute("alternatives");
809                 if (reason && alternatives) {
810                     // Construct message
811                     wi.message = std::string(reason) + " function '" + name + "' called. It is recommended to use ";
812                     std::vector<std::string> alt = getnames(alternatives);
813                     for (std::size_t i = 0; i < alt.size(); ++i) {
814                         wi.message += "'" + alt[i] + "'";
815                         if (i == alt.size() - 1)
816                             wi.message += " instead.";
817                         else if (i == alt.size() - 2)
818                             wi.message += " or ";
819                         else
820                             wi.message += ", ";
821                     }
822                 } else {
823                     const char * const message = functionnode->GetText();
824                     if (!message) {
825                         return Error(MISSING_ATTRIBUTE, "\"reason\" and \"alternatives\" or some text.");
826                     } else
827                         wi.message = message;
828                 }
829     
830                 functionwarn[name] = wi;
831             } else
832                 unknown_elements.insert(functionnodename);
833         }
834         return Error(OK);
835     }

837     bool Library::isIntArgValid(const Token *ftok, int argnr, const MathLib::bigint argvalue) const
838     {
839         const ArgumentChecks *ac = getarg(ftok, argnr);
840         if (!ac || ac->valid.empty())
841             return true;
842         else if (ac->valid.find('.') != std::string::npos)
843             return isFloatArgValid(ftok, argnr, argvalue);
844         TokenList tokenList(nullptr);
845         gettokenlistfromvalid(ac->valid, tokenList);
846         for (const Token *tok = tokenList.front(); tok; tok = tok->next()) {
847             if (tok->isNumber() && argvalue == MathLib::toLongNumber(tok->str()))
848                 return true;
849             if (Token::Match(tok, "%num% : %num%") && argvalue >= MathLib::toLongNumber(tok->str()) && argvalue <= MathLib::toLongNumber(tok->strAt(2)))
850                 return true;
851             if (Token::Match(tok, "%num% : ,") && argvalue >= MathLib::toLongNumber(tok->str()))
852                 return true;
853             if ((!tok->previous() || tok->previous()->str() == ",") && Token::Match(tok,": %num%") && argvalue <= MathLib::toLongNumber(tok->strAt(1)))
854                 return true;
855         }
856         return false;
857     }

872                 return true;
873         }
874         return false;

877     std::string Library::getFunctionName(const Token *ftok, bool *error) const
878     {
879         if (!ftok) {
880             *error = true;
881             return "";
882         }
883         if (ftok->isName()) {
884             for (const Scope *scope = ftok->scope(); scope; scope = scope->nestedIn) {
885                 if (!scope->isClassOrStruct())
886                     continue;
887                 const std::vector<Type::BaseInfo> &derivedFrom = scope->definedType->derivedFrom;
888                 for (const Type::BaseInfo & baseInfo : derivedFrom) {
889                     const std::string name(baseInfo.name + "::" + ftok->str());
890                     if (functions.find(name) != functions.end() && matchArguments(ftok, name))
891                         return name;
892                 }
893             }
894             return ftok->str();
895         }
896         if (ftok->str() == "::") {
897             if (!ftok->astOperand2())
898                 return getFunctionName(ftok->astOperand1(), error);
899             return getFunctionName(ftok->astOperand1(),error) + "::" + getFunctionName(ftok->astOperand2(),error);
900         }
901         if (ftok->str() == "." && ftok->astOperand1()) {
902             const std::string type = astCanonicalType(ftok->astOperand1());
903             if (type.empty()) {
904                 *error = true;
905                 return "";
906             }
907     
908             return type + "::" + getFunctionName(ftok->astOperand2(),error);
909         }
910         *error = true;
911         return "";
912     }
913     
914     std::string Library::getFunctionName(const Token *ftok) const
915     {
916         if (!Token::Match(ftok, "%name% (") && (ftok->strAt(-1) != "&" || ftok->previous()->astOperand2()))
917             return "";
918     
919         // Lookup function name using AST..
920         if (ftok->astParent()) {
921             bool error = false;
922             const Token * tok = ftok->astParent()->isUnaryOp("&") ? ftok->astParent()->astOperand1() : ftok->next()->astOperand1();
923             const std::string ret = getFunctionName(tok, &error);
924             return error ? std::string() : ret;
925         }
926     
927         // Lookup function name without using AST..
928         if (Token::simpleMatch(ftok->previous(), "."))
929             return "";
930         if (!Token::Match(ftok->tokAt(-2), "%name% ::"))
931             return ftok->str();
932         std::string ret(ftok->str());
933         ftok = ftok->tokAt(-2);
934         while (Token::Match(ftok, "%name% ::")) {
935             ret = ftok->str() + "::" + ret;
936             ftok = ftok->tokAt(-2);
937         }
938         return ret;
939     }

1027     bool Library::isScopeNoReturn(const Token *end, std::string *unknownFunc) const
1028     {
1029         if (unknownFunc)
1030             unknownFunc->clear();
1031     
1032         if (Token::Match(end->tokAt(-2), "!!{ ; }")) {
1033             const Token *lastTop = end->tokAt(-2)->astTop();
1034             if (Token::simpleMatch(lastTop, "<<") &&
1035                 Token::simpleMatch(lastTop->astOperand1(), "(") &&
1036                 Token::Match(lastTop->astOperand1()->previous(), "%name% ("))
1037                 return isnoreturn(lastTop->astOperand1()->previous());
1038         }
1039     
1040         if (!Token::simpleMatch(end->tokAt(-2), ") ; }"))
1041             return false;
1042     
1043         const Token *funcname = end->linkAt(-2)->previous();
1044         const Token *start = funcname;
1045         if (Token::Match(funcname->tokAt(-3),"( * %name% )")) {
1046             funcname = funcname->previous();
1047             start = funcname->tokAt(-3);
1048         } else if (funcname->isName()) {
1049             while (Token::Match(start, "%name%|.|::"))
1050                 start = start->previous();
1051         } else {
1052             return false;
1053         }
1054         if (Token::Match(start,"[;{}]") && Token::Match(funcname, "%name% )| (")) {
1055             if (funcname->str() == "exit")
1056                 return true;
1057             if (!isnotnoreturn(funcname)) {
1058                 if (unknownFunc && !isnoreturn(funcname))
1059                     *unknownFunc = funcname->str();
1060                 return true;
1061             }
1062         }
1063         return false;
1064     }

1066     const Library::Container* Library::detectContainer(const Token* typeStart, bool iterator) const
1067     {
1068         for (std::map<std::string, Container>::const_iterator i = containers.begin(); i != containers.end(); ++i) {
1069             const Container& container = i->second;
1070             if (container.startPattern.empty())
1071                 continue;
1072     
1073             if (!Token::Match(typeStart, container.startPattern2.c_str()))
1074                 continue;
1075     
1076             if (!iterator && container.endPattern.empty()) // If endPattern is undefined, it will always match, but itEndPattern has to be defined.
1077                 return &container;
1078     
1079             for (const Token* tok = typeStart; tok && !tok->varId(); tok = tok->next()) {
1080                 if (tok->link()) {
1081                     const std::string& endPattern = iterator ? container.itEndPattern : container.endPattern;
1082                     if (Token::Match(tok->link(), endPattern.c_str()))
1083                         return &container;
1084                     break;
1085                 }
1086             }
1087         }
1088         return nullptr;
1089     }

1091     bool Library::isContainerYield(const Token * const cond, Library::Container::Yield y, const std::string& fallback)
1092     {
1093         if (!cond)
1094             return false;
1095         if (cond->str() == "(") {
1096             const Token* tok = cond->astOperand1();
1097             if (tok && tok->str() == ".") {
1098                 if (tok->astOperand1() && tok->astOperand1()->valueType()) {
1099                     if (const Library::Container *container = tok->astOperand1()->valueType()->container) {
1100                         return tok->astOperand2() && y == container->getYield(tok->astOperand2()->str());
1101                     }
1102                 } else if (!fallback.empty()) {
1103                     return Token::simpleMatch(cond, "( )") && cond->previous()->str() == fallback;
1104                 }
1105             }
1106         }
1107         return false;
1108     }

1110     // returns true if ftok is not a library function
1111     bool Library::isNotLibraryFunction(const Token *ftok) const
1112     {
1113         if (ftok->function() && ftok->function()->nestedIn && ftok->function()->nestedIn->type != Scope::eGlobal)
1114             return true;
1115     
1116         // variables are not library functions.
1117         if (ftok->varId())
1118             return true;
1119     
1120         return !matchArguments(ftok, getFunctionName(ftok));
1121     }

1123     bool Library::matchArguments(const Token *ftok, const std::string &functionName) const
1124     {
1125         const int callargs = numberOfArguments(ftok);
1126         const std::map<std::string, Function>::const_iterator it = functions.find(functionName);
1127         if (it == functions.cend())
1128             return (callargs == 0);
1129         int args = 0;
1130         int firstOptionalArg = -1;
1131         for (std::map<int, ArgumentChecks>::const_iterator it2 = it->second.argumentChecks.cbegin(); it2 != it->second.argumentChecks.cend(); ++it2) {
1132             if (it2->first > args)
1133                 args = it2->first;
1134             if (it2->second.optional && (firstOptionalArg == -1 || firstOptionalArg > it2->first))
1135                 firstOptionalArg = it2->first;
1136     
1137             if (it2->second.formatstr || it2->second.variadic)
1138                 return args <= callargs;
1139         }
1140         return (firstOptionalArg < 0) ? args == callargs : (callargs >= firstOptionalArg-1 && callargs <= args);
1141     }

1379     CPPCHECKLIB const Library::Container * getLibraryContainer(const Token * tok)
1380     {
1381         if (!tok)
1382             return nullptr;
1383         // TODO: Support dereferencing iterators
1384         // TODO: Support dereferencing with ->
1385         if (tok->isUnaryOp("*") && astIsPointer(tok->astOperand1())) {
1386             for (const ValueFlow::Value& v:tok->astOperand1()->values()) {
1387                 if (!v.isLocalLifetimeValue())
1388                     continue;
1389                 if (v.lifetimeKind != ValueFlow::Value::LifetimeKind::Address)
1390                     continue;
1391                 return getLibraryContainer(v.tokvalue);
1392             }
1393         }
1394         if (!tok->valueType())
1395             return nullptr;
1396         return tok->valueType()->container;
1397     }
