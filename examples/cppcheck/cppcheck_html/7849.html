20     #ifndef CHECKTHREAD_H
21     #define CHECKTHREAD_H
22     
23     #include <QThread>
24     #include "cppcheck.h"
25     #include "threadresult.h"
26     #include "suppressions.h"
27     
28     class Settings;
29     
30     /// @addtogroup GUI
31     /// @{
32     
33     /**
34     * @brief Thread to run cppcheck
35     *
36     */
37     class CheckThread : public QThread {
38         Q_OBJECT
39     public:
40         explicit CheckThread(ThreadResult &result);
41         virtual ~CheckThread();
42     
43         /**
44         * @brief Set settings for cppcheck
45         *
46         * @param settings settings for cppcheck
47         */
48         void check(const Settings &settings);
49     
50         /**
51         * @brief Run whole program analysis
52         * @param files    All files
53         */
54         void analyseWholeProgram(const QStringList &files);
55     
56         void setAddonsAndTools(const QStringList &addonsAndTools) {
57             mAddonsAndTools = addonsAndTools;
58         }
59     
60         void setMisraFile(const QString &misraFile) {
61             mMisraFile = misraFile;
62         }
63     
64         void setDataDir(const QString &dataDir) {
65             mDataDir = dataDir;
66         }
67     
68         void setClangIncludePaths(const QStringList &s) {
69             mClangIncludePaths = s;
70         }
71     
72         void setSuppressions(const QList<Suppressions::Suppression> &s) {
73             mSuppressions = s;
74         }
75     
76         /**
77         * @brief method that is run in a thread
78         *
79         */
80         void run();
81     
82         void stop();
83     
84         /**
85          * Determine command to run clang
86          * \return Command to run clang, empty if it is not found
87          */
88         static QString clangCmd();
89     
90         /**
91          * Determine command to run clang-tidy
92          * \return Command to run clang-tidy, empty if it is not found
93          */
94         static QString clangTidyCmd();
95     
96         /**
97          * Determine command to run python
98          * \return Command to run python, empty if it is not found
99          */
100         static QString pythonCmd();
101     
102         /**
103          * Look for addon and return path
104          * \return path to addon if found, empty if it is not found
105          */
106         static QString getAddonFilePath(const QString &dataDir, const QString &addonFile);
107     
108     signals:
109     
110         /**
111         * @brief cpp checking is done
112         *
113         */
114         void done();
115     
116         void fileChecked(const QString &file);
117     protected:
118     
119         /**
120         * @brief States for the check thread.
121         * Whole purpose of these states is to allow stopping of the checking. When
122         * stopping we say for the thread (Stopping) that stop when current check
123         * has been completed. Thread must be stopped cleanly, just terminating thread
124         * likely causes unpredictable side-effects.
125         */
126         enum State {
127             Running, /**< The thread is checking. */
128             Stopping, /**< The thread will stop after current work. */
129             Stopped, /**< The thread has been stopped. */
130             Ready, /**< The thread is ready. */
131         };
132     
133         /**
134         * @brief Thread's current execution state.
135         */
136         State mState;
137     
138         ThreadResult &mResult;
139         /**
140         * @brief Cppcheck itself
141         */
142         CppCheck mCppcheck;
143     
144     private:
145         void runAddonsAndTools(const ImportProject::FileSettings *fileSettings, const QString &fileName);
146     
147         void parseAddonErrors(QString err, const QString &tool);
148         void parseClangErrors(const QString &tool, const QString &file0, QString err);
149     
150         bool isSuppressed(const Suppressions::ErrorMessage &errorMessage) const;
151     
152         QStringList mFiles;
153         bool mAnalyseWholeProgram;
154         QStringList mAddonsAndTools;
155         QString mDataDir;
156         QStringList mClangIncludePaths;
157         QList<Suppressions::Suppression> mSuppressions;
158         QString mMisraFile;
159     };
160     /// @}
161     #endif // CHECKTHREAD_H
