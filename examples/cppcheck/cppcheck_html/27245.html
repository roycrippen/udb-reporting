82     void XmlReportV2::writeHeader()
83     {
84         mXmlWriter->setAutoFormatting(true);
85         mXmlWriter->writeStartDocument();
86         mXmlWriter->writeStartElement(ResultElementName);
87         mXmlWriter->writeAttribute(VersionAttribute, QString::number(2));
88         mXmlWriter->writeStartElement(CppcheckElementName);
89         mXmlWriter->writeAttribute(VersionAttribute, QString(CppCheck::version()));
90         mXmlWriter->writeEndElement();
91         mXmlWriter->writeStartElement(ErrorsElementName);
92     }
93     
94     void XmlReportV2::writeFooter()
95     {
96         mXmlWriter->writeEndElement(); // errors
97         mXmlWriter->writeEndElement(); // results
98         mXmlWriter->writeEndDocument();
99     }

101     void XmlReportV2::writeError(const ErrorItem &error)
102     {
103         /*
104         Error example from the core program in xml
105         <error id="mismatchAllocDealloc" severity="error" msg="Mismatching allocation and deallocation: k"
106                   verbose="Mismatching allocation and deallocation: k">
107           <location file="..\..\test\test.cxx" line="16"/>
108           <location file="..\..\test\test.cxx" line="32"/>
109         </error>
110         */
111     
112         mXmlWriter->writeStartElement(ErrorElementName);
113         mXmlWriter->writeAttribute(IdAttribute, error.errorId);
114     
115         // Don't localize severity so we can read these files
116         mXmlWriter->writeAttribute(SeverityAttribute, GuiSeverity::toString(error.severity));
117         const QString summary = XmlReport::quoteMessage(error.summary);
118         mXmlWriter->writeAttribute(MsgAttribute, summary);
119         const QString message = XmlReport::quoteMessage(error.message);
120         mXmlWriter->writeAttribute(VerboseAttribute, message);
121         if (error.inconclusive)
122             mXmlWriter->writeAttribute(InconclusiveAttribute, "true");
123         if (error.cwe > 0)
124             mXmlWriter->writeAttribute(CWEAttribute, QString::number(error.cwe));
125         if (!error.sinceDate.isEmpty())
126             mXmlWriter->writeAttribute(SinceDateAttribute, error.sinceDate);
127         if (!error.tags.isEmpty())
128             mXmlWriter->writeAttribute(TagsAttribute, error.tags);
129     
130         for (int i = error.errorPath.count() - 1; i >= 0; i--) {
131             mXmlWriter->writeStartElement(LocationElementName);
132     
133             QString file = QDir::toNativeSeparators(error.errorPath[i].file);
134             if (!error.file0.isEmpty() && file != error.file0) {
135                 mXmlWriter->writeAttribute(IncludedFromFilenameAttribute, quoteMessage(error.file0));
136             }
137             mXmlWriter->writeAttribute(FilenameAttribute, XmlReport::quoteMessage(file));
138             mXmlWriter->writeAttribute(LineAttribute, QString::number(error.errorPath[i].line));
139             if (error.errorPath[i].column > 0)
140                 mXmlWriter->writeAttribute(ColumnAttribute, QString::number(error.errorPath[i].column));
141             if (error.errorPath.count() > 1)
142                 mXmlWriter->writeAttribute(InfoAttribute, XmlReport::quoteMessage(error.errorPath[i].info));
143     
144             mXmlWriter->writeEndElement();
145         }
146     
147         mXmlWriter->writeEndElement();
148     }

150     QList<ErrorItem> XmlReportV2::read()
151     {
152         QList<ErrorItem> errors;
153         bool insideResults = false;
154         if (!mXmlReader) {
155             qDebug() << "You must Open() the file before reading it!";
156             return errors;
157         }
158         while (!mXmlReader->atEnd()) {
159             switch (mXmlReader->readNext()) {
160             case QXmlStreamReader::StartElement:
161                 if (mXmlReader->name() == ResultElementName)
162                     insideResults = true;
163     
164                 // Read error element from inside result element
165                 if (insideResults && mXmlReader->name() == ErrorElementName) {
166                     ErrorItem item = readError(mXmlReader);
167                     errors.append(item);
168                 }
169                 break;
170     
171             case QXmlStreamReader::EndElement:
172                 if (mXmlReader->name() == ResultElementName)
173                     insideResults = false;
174                 break;
175     
176             // Not handled
177             case QXmlStreamReader::NoToken:
178             case QXmlStreamReader::Invalid:
179             case QXmlStreamReader::StartDocument:
180             case QXmlStreamReader::EndDocument:
181             case QXmlStreamReader::Characters:
182             case QXmlStreamReader::Comment:
183             case QXmlStreamReader::DTD:
184             case QXmlStreamReader::EntityReference:
185             case QXmlStreamReader::ProcessingInstruction:
186                 break;
187             }
188         }
189         return errors;
190     }

192     ErrorItem XmlReportV2::readError(QXmlStreamReader *reader)
193     {
194         /*
195         Error example from the core program in xml
196         <error id="mismatchAllocDealloc" severity="error" msg="Mismatching allocation and deallocation: k"
197                   verbose="Mismatching allocation and deallocation: k">
198           <location file="..\..\test\test.cxx" line="16"/>
199           <location file="..\..\test\test.cxx" line="32"/>
200         </error>
201         */
202     
203         ErrorItem item;
204     
205         // Read error element from inside errors element
206         if (mXmlReader->name() == ErrorElementName) {
207             QXmlStreamAttributes attribs = reader->attributes();
208             item.errorId = attribs.value(QString(), IdAttribute).toString();
209             item.severity = GuiSeverity::fromString(attribs.value(QString(), SeverityAttribute).toString());
210             const QString summary = attribs.value(QString(), MsgAttribute).toString();
211             item.summary = XmlReport::unquoteMessage(summary);
212             const QString message = attribs.value(QString(), VerboseAttribute).toString();
213             item.message = XmlReport::unquoteMessage(message);
214             if (attribs.hasAttribute(QString(), InconclusiveAttribute))
215                 item.inconclusive = true;
216             if (attribs.hasAttribute(QString(), CWEAttribute))
217                 item.cwe = attribs.value(QString(), CWEAttribute).toString().toInt();
218             if (attribs.hasAttribute(QString(), SinceDateAttribute))
219                 item.sinceDate = attribs.value(QString(), SinceDateAttribute).toString();
220             if (attribs.hasAttribute(QString(), TagsAttribute))
221                 item.tags = attribs.value(QString(), TagsAttribute).toString();
222         }
223     
224         bool errorRead = false;
225         while (!errorRead && !mXmlReader->atEnd()) {
226             switch (mXmlReader->readNext()) {
227             case QXmlStreamReader::StartElement:
228     
229                 // Read location element from inside error element
230                 if (mXmlReader->name() == LocationElementName) {
231                     QXmlStreamAttributes attribs = mXmlReader->attributes();
232                     QString file0 = attribs.value(QString(), IncludedFromFilenameAttribute).toString();
233                     if (!file0.isEmpty())
234                         item.file0 = XmlReport::unquoteMessage(file0);
235                     QErrorPathItem loc;
236                     loc.file = XmlReport::unquoteMessage(attribs.value(QString(), FilenameAttribute).toString());
237                     loc.line = attribs.value(QString(), LineAttribute).toString().toUInt();
238                     if (attribs.hasAttribute(QString(), ColumnAttribute))
239                         loc.column = attribs.value(QString(), ColumnAttribute).toString().toInt();
240                     if (attribs.hasAttribute(QString(), InfoAttribute))
241                         loc.info = XmlReport::unquoteMessage(attribs.value(QString(), InfoAttribute).toString());
242                     item.errorPath.push_front(loc);
243                 }
244                 break;
245     
246             case QXmlStreamReader::EndElement:
247                 if (mXmlReader->name() == ErrorElementName)
248                     errorRead = true;
249                 break;
250     
251             // Not handled
252             case QXmlStreamReader::NoToken:
253             case QXmlStreamReader::Invalid:
254             case QXmlStreamReader::StartDocument:
255             case QXmlStreamReader::EndDocument:
256             case QXmlStreamReader::Characters:
257             case QXmlStreamReader::Comment:
258             case QXmlStreamReader::DTD:
259             case QXmlStreamReader::EntityReference:
260             case QXmlStreamReader::ProcessingInstruction:
261                 break;
262             }
263         }
264     
265         if (item.errorPath.size() == 1 && item.errorPath[0].info.isEmpty())
266             item.errorPath[0].info = item.message;
267     
268         return item;
269     }
