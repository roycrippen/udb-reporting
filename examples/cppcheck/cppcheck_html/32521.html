68         ret.notuninit = mUi->notuninit->isChecked();

71         if (mUi->minsize1type->currentIndex() != 0) {

79             if (mUi->minsize2type->currentIndex() != 0) {

81                 minsize2.type = mUi->minsize2type->currentText();
