70     static const Token* skipScopeIdentifiers(const Token* tok)
71     {
72         if (tok && tok->str() == "::") {
73             tok = tok->next();
74         }
75         while (Token::Match(tok, "%name% ::") ||
76                (Token::Match(tok, "%name% <") && Token::simpleMatch(tok->linkAt(1), "> ::"))) {
77             if (tok->strAt(1) == "::")
78                 tok = tok->tokAt(2);
79             else
80                 tok = tok->linkAt(1)->tokAt(2);
81         }
82     
83         return tok;
84     }

86     static bool isExecutableScope(const Token* tok)
87     {
88         if (!Token::simpleMatch(tok, "{"))
89             return false;
90         const Token * tok2 = tok->link()->previous();
91         if (Token::simpleMatch(tok2, "; }"))
92             return true;
93         if (Token::Match(tok2, "{|} }")) {
94             const Token* startTok = tok2->str() == "{" ? tok2 : tok2->link();
95             if (Token::Match(startTok->previous(), "do|try|else {"))
96                 return true;
97             if (Token::simpleMatch(startTok->previous(), ") {"))
98                 return !findLambdaStartToken(tok2);
99             if (tok->str() == "{")
100                 return false;
101             else
102                 return isExecutableScope(startTok);
103         }
104         return false;
105     }

107     void SymbolDatabase::createSymbolDatabaseFindAllScopes()
108     {
109         // create global scope
110         scopeList.emplace_back(this, nullptr, nullptr);
111     
112         // pointer to current scope
113         Scope *scope = &scopeList.back();
114     
115         // Store current access in each scope (depends on evaluation progress)
116         std::map<const Scope*, AccessControl> access;
117     
118         // find all scopes
119         for (const Token *tok = mTokenizer->tokens(); tok; tok = tok ? tok->next() : nullptr) {
120             // #5593 suggested to add here:
121             if (mErrorLogger)
122                 mErrorLogger->reportProgress(mTokenizer->list.getSourceFilePath(),
123                                              "SymbolDatabase",
124                                              tok->progressValue());
125             // Locate next class
126             if ((mTokenizer->isCPP() && ((Token::Match(tok, "class|struct|union|namespace ::| %name% {|:|::|<") &&
127                                           !Token::Match(tok->previous(), "new|friend|const|enum|typedef|mutable|volatile|using|)|(|<")) ||
128                                          (Token::Match(tok, "enum class| %name% {") ||
129                                           Token::Match(tok, "enum class| %name% : %name% {"))))
130                 || (mTokenizer->isC() && Token::Match(tok, "struct|union|enum %name% {"))) {
131                 const Token *tok2 = tok->tokAt(2);
132     
133                 if (tok->strAt(1) == "::")
134                     tok2 = tok2->next();
135                 else if (mTokenizer->isCPP() && tok->strAt(1) == "class")
136                     tok2 = tok2->next();
137     
138                 while (Token::Match(tok2, ":: %name%"))
139                     tok2 = tok2->tokAt(2);
140                 while (Token::Match(tok2, "%name% :: %name%"))
141                     tok2 = tok2->tokAt(2);
142     
143                 // skip over template args
144                 while (tok2 && tok2->str() == "<" && tok2->link()) {
145                     tok2 = tok2->link()->next();
146                     while (Token::Match(tok2, ":: %name%"))
147                         tok2 = tok2->tokAt(2);
148                 }
149     
150                 // skip over final
151                 if (mTokenizer->isCPP() && Token::simpleMatch(tok2, "final"))
152                     tok2 = tok2->next();
153     
154                 // make sure we have valid code
155                 if (!Token::Match(tok2, "{|:")) {
156                     // check for qualified variable
157                     if (tok2 && tok2->next()) {
158                         if (tok2->next()->str() == ";")
159                             tok = tok2->next();
160                         else if (Token::simpleMatch(tok2->next(), "= {") &&
161                                  Token::simpleMatch(tok2->linkAt(2), "} ;"))
162                             tok = tok2->linkAt(2)->next();
163                         else if (Token::Match(tok2->next(), "(|{") &&
164                                  tok2->next()->link()->strAt(1) == ";")
165                             tok = tok2->next()->link()->next();
166                         // skip variable declaration
167                         else if (Token::Match(tok2, "*|&|>"))
168                             continue;
169                         else if (Token::Match(tok2, "%name% (") && mTokenizer->isFunctionHead(tok2->next(), "{;"))
170                             continue;
171                         else if (Token::Match(tok2, "%name% ["))
172                             continue;
173                         // skip template
174                         else if (Token::simpleMatch(tok2, ";") &&
175                                  Token::Match(tok->previous(), "template|> class|struct")) {
176                             tok = tok2;
177                             continue;
178                         }
179                         // forward declaration
180                         else if (Token::simpleMatch(tok2, ";") &&
181                                  Token::Match(tok, "class|struct|union")) {
182                             // TODO: see if it can be used
183                             tok = tok2;
184                             continue;
185                         }
186                         // skip constructor
187                         else if (Token::simpleMatch(tok2, "(") &&
188                                  Token::simpleMatch(tok2->link(), ") ;")) {
189                             tok = tok2->link()->next();
190                             continue;
191                         } else
192                             throw InternalError(tok2, "SymbolDatabase bailout; unhandled code", InternalError::SYNTAX);
193                         continue;
194                     }
195                     break; // bail
196                 }
197     
198                 const Token * name = tok->next();
199     
200                 if (name->str() == "class" && name->strAt(-1) == "enum")
201                     name = name->next();
202     
203                 Scope *new_scope = findScope(name, scope);
204     
205                 if (new_scope) {
206                     // only create base list for classes and structures
207                     if (new_scope->isClassOrStruct()) {
208                         // goto initial '{'
209                         if (!new_scope->definedType)
210                             mTokenizer->syntaxError(nullptr); // #6808
211                         tok2 = new_scope->definedType->initBaseInfo(tok, tok2);
212                         // make sure we have valid code
213                         if (!tok2) {
214                             break;
215                         }
216                     }
217     
218                     // definition may be different than declaration
219                     if (mTokenizer->isCPP() && tok->str() == "class") {
220                         access[new_scope] = AccessControl::Private;
221                         new_scope->type = Scope::eClass;
222                     } else if (tok->str() == "struct") {
223                         access[new_scope] = AccessControl::Public;
224                         new_scope->type = Scope::eStruct;
225                     }
226     
227                     new_scope->classDef = tok;
228                     new_scope->bodyStart = tok2;
229                     new_scope->bodyEnd = tok2->link();
230                     // make sure we have valid code
231                     if (!new_scope->bodyEnd) {
232                         mTokenizer->syntaxError(tok);
233                     }
234                     scope = new_scope;
235                     tok = tok2;
236                 } else {
237                     scopeList.emplace_back(this, tok, scope);
238                     new_scope = &scopeList.back();
239     
240                     if (tok->str() == "class")
241                         access[new_scope] = AccessControl::Private;
242                     else if (tok->str() == "struct" || tok->str() == "union")
243                         access[new_scope] = AccessControl::Public;
244     
245                     // fill typeList...
246                     if (new_scope->isClassOrStructOrUnion() || new_scope->type == Scope::eEnum) {
247                         Type* new_type = findType(name, scope);
248                         if (!new_type) {
249                             typeList.emplace_back(new_scope->classDef, new_scope, scope);
250                             new_type = &typeList.back();
251                             scope->definedTypesMap[new_type->name()] = new_type;
252                         } else
253                             new_type->classScope = new_scope;
254                         new_scope->definedType = new_type;
255                     }
256     
257                     // only create base list for classes and structures
258                     if (new_scope->isClassOrStruct()) {
259                         // goto initial '{'
260                         tok2 = new_scope->definedType->initBaseInfo(tok, tok2);
261     
262                         // make sure we have valid code
263                         if (!tok2) {
264                             mTokenizer->syntaxError(tok);
265                         }
266                     } else if (new_scope->type == Scope::eEnum) {
267                         if (tok2->str() == ":")
268                             tok2 = tok2->tokAt(2);
269                     }
270     
271                     new_scope->bodyStart = tok2;
272                     new_scope->bodyEnd = tok2->link();
273     
274                     // make sure we have valid code
275                     if (!new_scope->bodyEnd) {
276                         mTokenizer->syntaxError(tok);
277                     }
278     
279                     if (new_scope->type == Scope::eEnum) {
280                         tok2 = new_scope->addEnum(tok, mTokenizer->isCPP());
281                         scope->nestedList.push_back(new_scope);
282     
283                         if (!tok2)
284                             mTokenizer->syntaxError(tok);
285                     } else {
286                         // make the new scope the current scope
287                         scope->nestedList.push_back(new_scope);
288                         scope = new_scope;
289                     }
290     
291                     tok = tok2;
292                 }
293             }
294     
295             // Namespace and unknown macro (#3854)
296             else if (mTokenizer->isCPP() &&
297                      Token::Match(tok, "namespace %name% %type% (") &&
298                      tok->tokAt(2)->isUpperCaseName() &&
299                      Token::simpleMatch(tok->linkAt(3), ") {")) {
300                 scopeList.emplace_back(this, tok, scope);
301     
302                 Scope *new_scope = &scopeList.back();
303                 access[new_scope] = AccessControl::Public;
304     
305                 const Token *tok2 = tok->linkAt(3)->next();
306     
307                 new_scope->bodyStart = tok2;
308                 new_scope->bodyEnd = tok2->link();
309     
310                 // make sure we have valid code
311                 if (!new_scope->bodyEnd) {
312                     scopeList.pop_back();
313                     break;
314                 }
315     
316                 // make the new scope the current scope
317                 scope->nestedList.push_back(new_scope);
318                 scope = &scopeList.back();
319     
320                 tok = tok2;
321             }
322     
323             // forward declaration
324             else if (Token::Match(tok, "class|struct|union %name% ;") &&
325                      tok->strAt(-1) != "friend") {
326                 if (!findType(tok->next(), scope)) {
327                     // fill typeList..
328                     typeList.emplace_back(tok, nullptr, scope);
329                     Type* new_type = &typeList.back();
330                     scope->definedTypesMap[new_type->name()] = new_type;
331                 }
332                 tok = tok->tokAt(2);
333             }
334     
335             // using namespace
336             else if (mTokenizer->isCPP() && Token::Match(tok, "using namespace ::| %type% ;|::")) {
337                 Scope::UsingInfo using_info;
338     
339                 using_info.start = tok; // save location
340                 using_info.scope = findNamespace(tok->tokAt(2), scope);
341     
342                 scope->usingList.push_back(using_info);
343     
344                 // check for global namespace
345                 if (tok->strAt(2) == "::")
346                     tok = tok->tokAt(4);
347                 else
348                     tok = tok->tokAt(3);
349     
350                 // skip over qualification
351                 while (Token::Match(tok, "%type% ::"))
352                     tok = tok->tokAt(2);
353             }
354     
355             // using type alias
356             else if (mTokenizer->isCPP() && Token::Match(tok, "using %name% =")) {
357                 if (tok->strAt(-1) != ">" && !findType(tok->next(), scope)) {
358                     // fill typeList..
359                     typeList.emplace_back(tok, nullptr, scope);
360                     Type* new_type = &typeList.back();
361                     scope->definedTypesMap[new_type->name()] = new_type;
362                 }
363     
364                 tok = tok->tokAt(3);
365     
366                 while (tok && tok->str() != ";")
367                     tok = tok->next();
368             }
369     
370             // unnamed struct and union
371             else if (Token::Match(tok, "struct|union {") &&
372                      Token::Match(tok->next()->link(), "} *|&| %name% ;|[|=")) {
373                 scopeList.emplace_back(this, tok, scope);
374     
375                 Scope *new_scope = &scopeList.back();
376                 access[new_scope] = AccessControl::Public;
377     
378                 const Token* varNameTok = tok->next()->link()->next();
379                 if (varNameTok->str() == "*") {
380                     varNameTok = varNameTok->next();
381                 } else if (varNameTok->str() == "&") {
382                     varNameTok = varNameTok->next();
383                 }
384     
385                 typeList.emplace_back(tok, new_scope, scope);
386                 {
387                     Type* new_type = &typeList.back();
388                     new_scope->definedType = new_type;
389                     scope->definedTypesMap[new_type->name()] = new_type;
390                 }
391     
392                 scope->addVariable(varNameTok, tok, tok, access[scope], new_scope->definedType, scope, mSettings);
393     
394                 const Token *tok2 = tok->next();
395     
396                 new_scope->bodyStart = tok2;
397                 new_scope->bodyEnd = tok2->link();
398     
399                 // make sure we have valid code
400                 if (!new_scope->bodyEnd) {
401                     scopeList.pop_back();
402                     break;
403                 }
404     
405                 // make the new scope the current scope
406                 scope->nestedList.push_back(new_scope);
407                 scope = new_scope;
408     
409                 tok = tok2;
410             }
411     
412             // anonymous struct, union and namespace
413             else if ((Token::Match(tok, "struct|union {") &&
414                       Token::simpleMatch(tok->next()->link(), "} ;")) ||
415                      Token::simpleMatch(tok, "namespace {")) {
416                 scopeList.emplace_back(this, tok, scope);
417     
418                 Scope *new_scope = &scopeList.back();
419                 access[new_scope] = AccessControl::Public;
420     
421                 const Token *tok2 = tok->next();
422     
423                 new_scope->bodyStart = tok2;
424                 new_scope->bodyEnd = tok2->link();
425     
426                 typeList.emplace_back(tok, new_scope, scope);
427                 {
428                     Type* new_type = &typeList.back();
429                     new_scope->definedType = new_type;
430                     scope->definedTypesMap[new_type->name()] = new_type;
431                 }
432     
433                 // make sure we have valid code
434                 if (!new_scope->bodyEnd) {
435                     scopeList.pop_back();
436                     break;
437                 }
438     
439                 // make the new scope the current scope
440                 scope->nestedList.push_back(new_scope);
441                 scope = new_scope;
442     
443                 tok = tok2;
444             }
445     
446             // forward declared enum
447             else if (Token::Match(tok, "enum class| %name% ;") || Token::Match(tok, "enum class| %name% : %name% ;")) {
448                 typeList.emplace_back(tok, nullptr, scope);
449                 Type* new_type = &typeList.back();
450                 scope->definedTypesMap[new_type->name()] = new_type;
451                 tok = tok->tokAt(2);
452             }
453     
454             // check for end of scope
455             else if (tok == scope->bodyEnd) {
456                 access.erase(scope);
457                 scope = const_cast<Scope*>(scope->nestedIn);
458                 continue;
459             }
460     
461             // check if in class or structure or union
462             else if (scope->isClassOrStructOrUnion()) {
463                 const Token *funcStart = nullptr;
464                 const Token *argStart = nullptr;
465                 const Token *declEnd = nullptr;
466     
467                 // What section are we in..
468                 if (tok->str() == "private:")
469                     access[scope] = AccessControl::Private;
470                 else if (tok->str() == "protected:")
471                     access[scope] = AccessControl::Protected;
472                 else if (tok->str() == "public:" || tok->str() == "__published:")
473                     access[scope] = AccessControl::Public;
474                 else if (Token::Match(tok, "public|protected|private %name% :")) {
475                     if (tok->str() == "private")
476                         access[scope] = AccessControl::Private;
477                     else if (tok->str() == "protected")
478                         access[scope] = AccessControl::Protected;
479                     else
480                         access[scope] = AccessControl::Public;
481     
482                     tok = tok->tokAt(2);
483                 }
484     
485                 // class function?
486                 else if (isFunction(tok, scope, &funcStart, &argStart, &declEnd)) {
487                     if (tok->previous()->str() != "::" || tok->strAt(-2) == scope->className) {
488                         Function function(mTokenizer, tok, scope, funcStart, argStart);
489     
490                         // save the access type
491                         function.access = access[scope];
492     
493                         const Token *end = function.argDef->link();
494     
495                         // count the number of constructors
496                         if (function.isConstructor())
497                             scope->numConstructors++;
498     
499                         // assume implementation is inline (definition and implementation same)
500                         function.token = function.tokenDef;
501                         function.arg = function.argDef;
502     
503                         // out of line function
504                         if (const Token *endTok = mTokenizer->isFunctionHead(end, ";")) {
505                             tok = endTok;
506                             scope->addFunction(function);
507                         }
508     
509                         // inline function
510                         else {
511                             // find start of function '{'
512                             bool foundInitList = false;
513                             while (end && end->str() != "{" && end->str() != ";") {
514                                 if (end->link() && Token::Match(end, "(|<")) {
515                                     end = end->link();
516                                 } else if (foundInitList &&
517                                            Token::Match(end, "%name%|> {") &&
518                                            Token::Match(end->linkAt(1), "} ,|{")) {
519                                     end = end->linkAt(1);
520                                 } else {
521                                     if (end->str() == ":")
522                                         foundInitList = true;
523                                     end = end->next();
524                                 }
525                             }
526     
527                             if (!end || end->str() == ";")
528                                 continue;
529     
530                             scope->addFunction(function);
531     
532                             Function* funcptr = &scope->functionList.back();
533                             const Token *tok2 = funcStart;
534     
535                             addNewFunction(&scope, &tok2);
536                             if (scope) {
537                                 scope->functionOf = function.nestedIn;
538                                 scope->function = funcptr;
539                                 scope->function->functionScope = scope;
540                             }
541     
542                             tok = tok2;
543                         }
544                     }
545     
546                     // nested class or friend function?
547                     else {
548                         /** @todo check entire qualification for match */
549                         const Scope * const nested = scope->findInNestedListRecursive(tok->strAt(-2));
550     
551                         if (nested)
552                             addClassFunction(&scope, &tok, argStart);
553                         else {
554                             /** @todo handle friend functions */
555                         }
556                     }
557                 }
558     
559                 // friend class declaration?
560                 else if (mTokenizer->isCPP() && Token::Match(tok, "friend class| ::| %any% ;|::")) {
561                     Type::FriendInfo friendInfo;
562     
563                     // save the name start
564                     friendInfo.nameStart = tok->strAt(1) == "class" ? tok->tokAt(2) : tok->next();
565                     friendInfo.nameEnd = friendInfo.nameStart;
566     
567                     // skip leading "::"
568                     if (friendInfo.nameEnd->str() == "::")
569                         friendInfo.nameEnd = friendInfo.nameEnd->next();
570     
571                     // skip qualification "name ::"
572                     while (friendInfo.nameEnd && friendInfo.nameEnd->strAt(1) == "::")
573                         friendInfo.nameEnd = friendInfo.nameEnd->tokAt(2);
574     
575                     // fill this in after parsing is complete
576                     friendInfo.type = nullptr;
577     
578                     if (!scope->definedType)
579                         mTokenizer->syntaxError(tok);
580     
581                     scope->definedType->friendList.push_back(friendInfo);
582                 }
583             } else if (scope->type == Scope::eNamespace || scope->type == Scope::eGlobal) {
584                 const Token *funcStart = nullptr;
585                 const Token *argStart = nullptr;
586                 const Token *declEnd = nullptr;
587     
588                 // function?
589                 if (isFunction(tok, scope, &funcStart, &argStart, &declEnd)) {
590                     // has body?
591                     if (declEnd && declEnd->str() == "{") {
592                         tok = funcStart;
593     
594                         // class function
595                         if (tok->previous() && tok->previous()->str() == "::")
596                             addClassFunction(&scope, &tok, argStart);
597     
598                         // class destructor
599                         else if (tok->previous() &&
600                                  tok->previous()->str() == "~" &&
601                                  tok->strAt(-2) == "::")
602                             addClassFunction(&scope, &tok, argStart);
603     
604                         // regular function
605                         else {
606                             const Function* const function = addGlobalFunction(scope, tok, argStart, funcStart);
607     
608                             if (!function)
609                                 mTokenizer->syntaxError(tok);
610                         }
611     
612                         // syntax error?
613                         if (!scope)
614                             mTokenizer->syntaxError(tok);
615                     }
616                     // function prototype?
617                     else if (declEnd && declEnd->str() == ";") {
618                         bool newFunc = true; // Is this function already in the database?
619                         for (std::multimap<std::string, const Function *>::const_iterator i = scope->functionMap.find(tok->str()); i != scope->functionMap.end() && i->first == tok->str(); ++i) {
620                             if (Function::argsMatch(scope, i->second->argDef, argStart, emptyString, 0)) {
621                                 newFunc = false;
622                                 break;
623                             }
624                         }
625     
626                         // save function prototype in database
627                         if (newFunc) {
628                             addGlobalFunctionDecl(scope, tok, argStart, funcStart);
629                         }
630     
631                         tok = declEnd;
632                         continue;
633                     }
634                 }
635             } else if (scope->isExecutable()) {
636                 if (Token::Match(tok, "else|try|do {")) {
637                     const Token* tok1 = tok->next();
638                     if (tok->str() == "else")
639                         scopeList.emplace_back(this, tok, scope, Scope::eElse, tok1);
640                     else if (tok->str() == "do")
641                         scopeList.emplace_back(this, tok, scope, Scope::eDo, tok1);
642                     else //if (tok->str() == "try")
643                         scopeList.emplace_back(this, tok, scope, Scope::eTry, tok1);
644     
645                     tok = tok1;
646                     scope->nestedList.push_back(&scopeList.back());
647                     scope = &scopeList.back();
648                 } else if (Token::Match(tok, "if|for|while|catch|switch (") && Token::simpleMatch(tok->next()->link(), ") {")) {
649                     const Token *scopeStartTok = tok->next()->link()->next();
650                     if (tok->str() == "if")
651                         scopeList.emplace_back(this, tok, scope, Scope::eIf, scopeStartTok);
652                     else if (tok->str() == "for") {
653                         scopeList.emplace_back(this, tok, scope, Scope::eFor, scopeStartTok);
654                     } else if (tok->str() == "while")
655                         scopeList.emplace_back(this, tok, scope, Scope::eWhile, scopeStartTok);
656                     else if (tok->str() == "catch") {
657                         scopeList.emplace_back(this, tok, scope, Scope::eCatch, scopeStartTok);
658                     } else // if (tok->str() == "switch")
659                         scopeList.emplace_back(this, tok, scope, Scope::eSwitch, scopeStartTok);
660     
661                     scope->nestedList.push_back(&scopeList.back());
662                     scope = &scopeList.back();
663                     if (scope->type == Scope::eFor)
664                         scope->checkVariable(tok->tokAt(2), AccessControl::Local, mSettings); // check for variable declaration and add it to new scope if found
665                     else if (scope->type == Scope::eCatch)
666                         scope->checkVariable(tok->tokAt(2), AccessControl::Throw, mSettings); // check for variable declaration and add it to new scope if found
667                     tok = scopeStartTok;
668                 } else if (Token::Match(tok, "%var% {")) {
669                     tok = tok->linkAt(1);
670                 } else if (const Token *lambdaEndToken = findLambdaEndToken(tok)) {
671                     const Token *lambdaStartToken = lambdaEndToken->link();
672                     const Token * argStart = lambdaStartToken->astParent();
673                     const Token * funcStart = Token::simpleMatch(argStart, "[") ? argStart : argStart->astParent();
674                     const Function * function = addGlobalFunction(scope, tok, argStart, funcStart);
675                     if (!function)
676                         mTokenizer->syntaxError(tok);
677                     tok = lambdaStartToken;
678                 } else if (tok->str() == "{") {
679                     if (isExecutableScope(tok)) {
680                         scopeList.emplace_back(this, tok, scope, Scope::eUnconditional, tok);
681                         scope->nestedList.push_back(&scopeList.back());
682                         scope = &scopeList.back();
683                     } else {
684                         tok = tok->link();
685                     }
686                 }
687                 // syntax error?
688                 if (!scope)
689                     mTokenizer->syntaxError(tok);
690             }
691         }
692     }

824     void SymbolDatabase::createSymbolDatabaseNeedInitialization()
825     {
826         if (mTokenizer->isC()) {
827             // For C code it is easy, as there are no constructors and no default values
828             for (std::list<Scope>::iterator it = scopeList.begin(); it != scopeList.end(); ++it) {
829                 Scope *scope = &(*it);
830                 if (scope->definedType)
831                     scope->definedType->needInitialization = Type::NeedInitialization::True;
832             }
833         } else {
834             // For C++, it is more difficult: Determine if user defined type needs initialization...
835             unsigned int unknowns = 0; // stop checking when there are no unknowns
836             unsigned int retry = 0;    // bail if we don't resolve all the variable types for some reason
837     
838             do {
839                 unknowns = 0;
840     
841                 for (std::list<Scope>::iterator it = scopeList.begin(); it != scopeList.end(); ++it) {
842                     Scope *scope = &(*it);
843     
844                     if (!scope->definedType) {
845                         mBlankTypes.emplace_back();
846                         scope->definedType = &mBlankTypes.back();
847                     }
848     
849                     if (scope->isClassOrStruct() && scope->definedType->needInitialization == Type::NeedInitialization::Unknown) {
850                         // check for default constructor
851                         bool hasDefaultConstructor = false;
852     
853                         std::list<Function>::const_iterator func;
854     
855                         for (func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
856                             if (func->type == Function::eConstructor) {
857                                 // check for no arguments: func ( )
858                                 if (func->argCount() == 0) {
859                                     hasDefaultConstructor = true;
860                                     break;
861                                 }
862     
863                                 /** check for arguments with default values */
864                                 else if (func->argCount() == func->initializedArgCount()) {
865                                     hasDefaultConstructor = true;
866                                     break;
867                                 }
868                             }
869                         }
870     
871                         // User defined types with user defined default constructor doesn't need initialization.
872                         // We assume the default constructor initializes everything.
873                         // Another check will figure out if the constructor actually initializes everything.
874                         if (hasDefaultConstructor)
875                             scope->definedType->needInitialization = Type::NeedInitialization::False;
876     
877                         // check each member variable to see if it needs initialization
878                         else {
879                             bool needInitialization = false;
880                             bool unknown = false;
881     
882                             std::list<Variable>::const_iterator var;
883                             for (var = scope->varlist.begin(); var != scope->varlist.end() && !needInitialization; ++var) {
884                                 if (var->isClass()) {
885                                     if (var->type()) {
886                                         // does this type need initialization?
887                                         if (var->type()->needInitialization == Type::NeedInitialization::True)
888                                             needInitialization = true;
889                                         else if (var->type()->needInitialization == Type::NeedInitialization::Unknown) {
890                                             if (!(var->valueType() && var->valueType()->type == ValueType::CONTAINER))
891                                                 unknown = true;
892                                         }
893                                     }
894                                 } else if (!var->hasDefault())
895                                     needInitialization = true;
896                             }
897     
898                             if (needInitialization)
899                                 scope->definedType->needInitialization = Type::NeedInitialization::True;
900                             else if (!unknown)
901                                 scope->definedType->needInitialization = Type::NeedInitialization::False;
902                             else {
903                                 if (scope->definedType->needInitialization == Type::NeedInitialization::Unknown)
904                                     unknowns++;
905                             }
906                         }
907                     } else if (scope->type == Scope::eUnion && scope->definedType->needInitialization == Type::NeedInitialization::Unknown)
908                         scope->definedType->needInitialization = Type::NeedInitialization::True;
909                 }
910     
911                 retry++;
912             } while (unknowns && retry < 100);
913     
914             // this shouldn't happen so output a debug warning
915             if (retry == 100 && mSettings->debugwarnings) {
916                 for (std::list<Scope>::iterator it = scopeList.begin(); it != scopeList.end(); ++it) {
917                     const Scope *scope = &(*it);
918     
919                     if (scope->isClassOrStruct() && scope->definedType->needInitialization == Type::NeedInitialization::Unknown)
920                         debugMessage(scope->classDef, "SymbolDatabase::SymbolDatabase couldn't resolve all user defined types.");
921                 }
922             }
923         }
924     }

926     void SymbolDatabase::createSymbolDatabaseVariableSymbolTable()
927     {
928         // create variable symbol table
929         mVariableList.resize(mTokenizer->varIdCount() + 1);
930         std::fill_n(mVariableList.begin(), mVariableList.size(), (const Variable*)nullptr);
931     
932         // check all scopes for variables
933         for (std::list<Scope>::iterator it = scopeList.begin(); it != scopeList.end(); ++it) {
934             Scope *scope = &(*it);
935     
936             // add all variables
937             for (std::list<Variable>::iterator var = scope->varlist.begin(); var != scope->varlist.end(); ++var) {
938                 const unsigned int varId = var->declarationId();
939                 if (varId)
940                     mVariableList[varId] = &(*var);
941                 // fix up variables without type
942                 if (!var->type() && !var->typeStartToken()->isStandardType()) {
943                     const Type *type = findType(var->typeStartToken(), scope);
944                     if (type)
945                         var->type(type);
946                 }
947             }
948     
949             // add all function parameters
950             for (std::list<Function>::iterator func = scope->functionList.begin(); func != scope->functionList.end(); ++func) {
951                 for (std::list<Variable>::iterator arg = func->argumentList.begin(); arg != func->argumentList.end(); ++arg) {
952                     // check for named parameters
953                     if (arg->nameToken() && arg->declarationId()) {
954                         const unsigned int declarationId = arg->declarationId();
955                         if (declarationId > 0U)
956                             mVariableList[declarationId] = &(*arg);
957                         // fix up parameters without type
958                         if (!arg->type() && !arg->typeStartToken()->isStandardType()) {
959                             const Type *type = findTypeInNested(arg->typeStartToken(), scope);
960                             if (type)
961                                 arg->type(type);
962                         }
963                     }
964                 }
965             }
966         }
967     
968         // fill in missing variables if possible
969         const std::size_t functions = functionScopes.size();
970         for (std::size_t i = 0; i < functions; ++i) {
971             const Scope *func = functionScopes[i];
972             for (const Token *tok = func->bodyStart->next(); tok && tok != func->bodyEnd; tok = tok->next()) {
973                 // check for member variable
974                 if (tok->varId() && tok->next() &&
975                     (tok->next()->str() == "." ||
976                      (tok->next()->str() == "[" && tok->linkAt(1)->strAt(1) == "."))) {
977                     const Token *tok1 = tok->next()->str() == "." ? tok->tokAt(2) : tok->linkAt(1)->tokAt(2);
978                     if (tok1 && tok1->varId() && mVariableList[tok1->varId()] == nullptr) {
979                         const Variable *var = mVariableList[tok->varId()];
980                         if (var && var->typeScope()) {
981                             // find the member variable of this variable
982                             const Variable *var1 = var->typeScope()->getVariable(tok1->str());
983                             if (var1) {
984                                 // add this variable to the look up table
985                                 mVariableList[tok1->varId()] = var1;
986                             }
987                         }
988                     }
989                 }
990             }
991         }
992     }

995     {
996         // Set scope pointers
997         for (std::list<Scope>::iterator it = scopeList.begin(); it != scopeList.end(); ++it) {
998             Token* start = const_cast<Token*>(it->bodyStart);
999             Token* end = const_cast<Token*>(it->bodyEnd);

1005             assert(end);
1006     
1007             end->scope(&*it);
1008     
1009             for (Token* tok = start; tok != end; tok = tok->next()) {
1010                 if (start != end && tok->str() == "{") {
1011                     bool isEndOfScope = false;
1012                     for (std::list<Scope*>::const_iterator innerScope = it->nestedList.begin(); innerScope != it->nestedList.end(); ++innerScope) {

1016                                 isEndOfScope = true;
1017                                 break;
1018                             }
1019                             tok = tok->next();
1020                             break;
1021                         }
1022                     }
1023                     if (isEndOfScope)
1024                         break;
1025                 }
1026                 tok->scope(&*it);
1027             }
1028         }

1240                     if (!enumerator.end)
1241                         mTokenizer->syntaxError(enumerator.start);
1242                     for (const Token * tok3 = enumerator.start; tok3 && tok3 != enumerator.end->next(); tok3 = tok3->next()) {
1243                         if (tok3->tokType() == Token::eName) {
1244                             const Enumerator * e = findEnumerator(tok3);
1245                             if (e)
1246                                 const_cast<Token *>(tok3)->enumerator(e);
1247                         }
1248                     }
1249     
1250                     // look for possible constant folding expressions
1251                     // rhs of operator:
1252                     Token *rhs = enumerator.start->previous()->astOperand2();
1253     
1254                     // constant folding of expression:
1255                     ValueFlow::valueFlowConstantFoldAST(rhs, mSettings);
1256     
1257                     // get constant folded value:
1258                     if (rhs && rhs->hasKnownIntValue()) {
1259                         enumerator.value = rhs->values().front().intvalue;
1260                         enumerator.value_known = true;
1261                         value = enumerator.value + 1;
1262                     }
1263                 }
1264     
1265                 // not initialized so use default value
1266                 else {
1267                     enumerator.value = value++;
1268                     enumerator.value_known = true;
1269                 }
1270             }
1271         }
1272     
1273         // find enumerators
1274         for (const Token* tok = mTokenizer->list.front(); tok != mTokenizer->list.back(); tok = tok->next()) {
1275             if (tok->tokType() != Token::eName)
1276                 continue;
1277             const Enumerator * enumerator = findEnumerator(tok);
1278             if (enumerator)
1279                 const_cast<Token *>(tok)->enumerator(enumerator);
1280         }
1281     }

1395     {
1396         // set all unknown array dimensions
1397         for (const Variable *var : mVariableList) {
1398             // check each array variable
1399             if (!var || !var->isArray())
1400                 continue;
1401             // check each array dimension
1402             for (const Dimension &const_dimension : var->dimensions()) {
1403                 Dimension &dimension = const_cast<Dimension &>(const_dimension);
1404                 if (dimension.num != 0 || !dimension.tok)
1405                     continue;
1406                 dimension.known = false;
1407     
1408                 // check for a single token dimension
1409                 if (dimension.tok->hasKnownIntValue()) {
1410                     dimension.known = true;
1411                     dimension.num = dimension.tok->getKnownIntValue();
1412                     continue;
1413                 }
1414     
1415                 else if (dimension.tok->valueType() && dimension.tok->valueType()->pointer == 0) {
1416                     int bits = 0;
1417                     switch (dimension.tok->valueType()->type) {
1418                     case ValueType::Type::CHAR:
1419                         bits = mSettings->char_bit;
1420                         break;
1421                     case ValueType::Type::SHORT:
1422                         bits = mSettings->short_bit;
1423                         break;
1424                     case ValueType::Type::INT:
1425                         bits = mSettings->int_bit;
1426                         break;
1427                     case ValueType::Type::LONG:
1428                         bits = mSettings->long_bit;
1429                         break;
1430                     case ValueType::Type::LONGLONG:
1431                         bits = mSettings->long_long_bit;
1432                         break;
1433                     default:
1434                         break;
1435                     };
1436     
1437                     if (bits > 0 && bits <= 62) {
1438                         if (dimension.tok->valueType()->sign == ValueType::Sign::UNSIGNED)
1439                             dimension.num = 1LL << bits;
1440                         else
1441                             dimension.num = 1LL << (bits - 1);
1442                     }
1443                 }
1444             }
1445         }
1446     }

1461     bool SymbolDatabase::isFunction(const Token *tok, const Scope* outerScope, const Token **funcStart, const Token **argStart, const Token** declEnd) const
1462     {
1463         if (tok->varId())
1464             return false;
1465     
1466         // function returning function pointer? '... ( ... %name% ( ... ))( ... ) {'
1467         // function returning reference to array '... ( & %name% ( ... ))[ ... ] {'
1468         // TODO: Activate this again
1469         if (false && tok->str() == "(" && tok->strAt(1) != "*" &&
1470             (tok->link()->previous()->str() == ")" || Token::simpleMatch(tok->link()->tokAt(-2), ") const"))) {
1471             const Token* tok2 = tok->link()->next();
1472             if (tok2 && tok2->str() == "(" && Token::Match(tok2->link()->next(), "{|;|const|=")) {
1473                 const Token* argStartTok;
1474                 if (tok->link()->previous()->str() == "const")
1475                     argStartTok = tok->link()->linkAt(-2);
1476                 else
1477                     argStartTok = tok->link()->linkAt(-1);
1478                 *funcStart = argStartTok->previous();
1479                 *argStart = argStartTok;
1480                 *declEnd = Token::findmatch(tok2->link()->next(), "{|;");
1481                 return true;
1482             } else if (tok2 && tok2->str() == "[") {
1483                 while (tok2 && tok2->str() == "[")
1484                     tok2 = tok2->link()->next();
1485                 if (Token::Match(tok2, "{|;|const|=")) {
1486                     const Token* argStartTok;
1487                     if (tok->link()->previous()->str() == "const")
1488                         argStartTok = tok->link()->linkAt(-2);
1489                     else
1490                         argStartTok = tok->link()->linkAt(-1);
1491                     *funcStart = argStartTok->previous();
1492                     *argStart = argStartTok;
1493                     *declEnd = Token::findmatch(tok2, "{|;");
1494                     return true;
1495                 }
1496             }
1497         }
1498     
1499         // regular function?
1500         else if (Token::Match(tok, "%name% (") && !isReservedName(tok->str()) && tok->previous() &&
1501                  (Token::Match(tok->previous(), "%name%|>|&|*|::|~") || // Either a return type or scope qualifier in front of tok
1502                   outerScope->isClassOrStructOrUnion())) { // or a ctor/dtor
1503             const Token* tok1 = tok->previous();
1504             const Token* tok2 = tok->next()->link()->next();
1505     
1506             if (!mTokenizer->isFunctionHead(tok->next(), ";:{"))
1507                 return false;
1508     
1509             // skip over destructor "~"
1510             if (tok1->str() == "~")
1511                 tok1 = tok1->previous();
1512     
1513             // skip over qualification
1514             while (Token::simpleMatch(tok1, "::")) {
1515                 tok1 = tok1->previous();
1516                 if (Token::Match(tok1, "%name%"))
1517                     tok1 = tok1->previous();
1518                 else if (tok1 && tok1->str() == ">" && tok1->link() && Token::Match(tok1->link()->previous(), "%name%"))
1519                     tok1 = tok1->link()->tokAt(-2);
1520             }
1521     
1522             // skip over const, noexcept, throw, override, final and volatile specifiers
1523             while (Token::Match(tok2, "const|noexcept|throw|override|final|volatile|&|&&")) {
1524                 tok2 = tok2->next();
1525                 if (tok2 && tok2->str() == "(")
1526                     tok2 = tok2->link()->next();
1527             }
1528     
1529             // skip over trailing return type
1530             if (tok2 && tok2->str() == ".") {
1531                 for (tok2 = tok2->next(); tok2; tok2 = tok2->next()) {
1532                     if (Token::Match(tok2, ";|{|=|override|final"))
1533                         break;
1534                     if (tok2->link() && Token::Match(tok2, "<|[|("))
1535                         tok2 = tok2->link();
1536                 }
1537             }
1538     
1539             // done if constructor or destructor
1540             if (!Token::Match(tok1, "{|}|;|public:|protected:|private:") && tok1) {
1541                 // skip over pointers and references
1542                 while (Token::Match(tok1, "%type%|*|&") && !endsWith(tok1->str(), ':') && (!isReservedName(tok1->str()) || tok1->str() == "const"))
1543                     tok1 = tok1->previous();
1544     
1545                 // skip over template
1546                 if (tok1 && tok1->str() == ">") {
1547                     if (tok1->link())
1548                         tok1 = tok1->link()->previous();
1549                     else
1550                         return false;
1551                 }
1552     
1553                 // function can't have number or variable as return type
1554                 if (tok1 && (tok1->isNumber() || tok1->varId()))
1555                     return false;
1556     
1557                 // skip over return type
1558                 if (Token::Match(tok1, "%name%")) {
1559                     if (tok1->str() == "return")
1560                         return false;
1561                     tok1 = tok1->previous();
1562                 }
1563     
1564                 // skip over qualification
1565                 while (Token::simpleMatch(tok1, "::")) {
1566                     tok1 = tok1->previous();
1567                     if (Token::Match(tok1, "%name%"))
1568                         tok1 = tok1->previous();
1569                     else if (tok1 && tok1->str() == ">" && tok1->link() && Token::Match(tok1->link()->previous(), "%name%"))
1570                         tok1 = tok1->link()->tokAt(-2);
1571                 }
1572     
1573                 // skip over modifiers and other stuff
1574                 while (Token::Match(tok1, "const|static|extern|template|virtual|struct|class|enum|%name%")) {
1575                     // friend type func(); is not a function
1576                     if (isCPP() && tok1->str() == "friend" && tok2->str() == ";")
1577                         return false;
1578                     tok1 = tok1->previous();
1579                 }
1580     
1581                 // should be at a sequence point if this is a function
1582                 if (!Token::Match(tok1, ">|{|}|;|public:|protected:|private:") && tok1)
1583                     return false;
1584             }
1585     
1586             if (tok2 &&
1587                 (Token::Match(tok2, ";|{|=") ||
1588                  (tok2->isUpperCaseName() && Token::Match(tok2, "%name% ;|{")) ||
1589                  (tok2->isUpperCaseName() && Token::Match(tok2, "%name% (") && tok2->next()->link()->strAt(1) == "{") ||
1590                  Token::Match(tok2, ": ::| %name% (|::|<|{") ||
1591                  Token::Match(tok2, "&|&&| ;|{") ||
1592                  Token::Match(tok2, "= delete|default ;"))) {
1593                 *funcStart = tok;
1594                 *argStart = tok->next();
1595                 *declEnd = Token::findmatch(tok2, "{|;");
1596                 return true;
1597             }
1598         }
1599     
1600         // UNKNOWN_MACRO(a,b) { ... }
1601         else if (outerScope->type == Scope::eGlobal &&
1602                  Token::Match(tok, "%name% (") &&
1603                  tok->isUpperCaseName() &&
1604                  Token::simpleMatch(tok->linkAt(1), ") {") &&
1605                  (!tok->previous() || Token::Match(tok->previous(), "[;{}]"))) {
1606             *funcStart = tok;
1607             *argStart = tok->next();
1608             *declEnd = tok->linkAt(1)->next();
1609             return true;
1610         }
1611     
1612         // template constructor?
1613         else if (Token::Match(tok, "%name% <") && Token::simpleMatch(tok->next()->link(), "> (")) {
1614             const Token* tok2 = tok->next()->link()->next()->link();
1615             if (Token::Match(tok2, ") const| ;|{|=") ||
1616                 Token::Match(tok2, ") : ::| %name% (|::|<|{") ||
1617                 Token::Match(tok2, ") const| noexcept {|;|(")) {
1618                 *funcStart = tok;
1619                 *argStart = tok2->link();
1620                 *declEnd = Token::findmatch(tok2->next(), "{|;");
1621                 return true;
1622             }
1623         }
1624     
1625         // regular C function with missing return or invalid C++ ?
1626         else if (Token::Match(tok, "%name% (") && !isReservedName(tok->str()) &&
1627                  Token::simpleMatch(tok->linkAt(1), ") {") &&
1628                  (!tok->previous() || Token::Match(tok->previous(), ";|}"))) {
1629             if (mTokenizer->isC()) {
1630                 debugMessage(tok, "SymbolDatabase::isFunction found C function '" + tok->str() + "' without a return type.");
1631                 *funcStart = tok;
1632                 *argStart = tok->next();
1633                 *declEnd = tok->linkAt(1)->next();
1634                 return true;
1635             }
1636             mTokenizer->syntaxError(tok);
1637         }
1638     
1639         return false;
1640     }

1723     void Variable::evaluate(const Settings* settings)
1724     {
1725         const Library * const lib = settings ? &settings->library : nullptr;
1726     
1727         if (mNameToken)
1728             setFlag(fIsArray, arrayDimensions(settings));
1729     
1730         if (mTypeStartToken)
1731             setValueType(ValueType::parseDecl(mTypeStartToken,settings));
1732     
1733         const Token* tok = mTypeStartToken;
1734         while (tok && tok->previous() && tok->previous()->isName())
1735             tok = tok->previous();
1736         const Token* end = mTypeEndToken;
1737         if (end)
1738             end = end->next();
1739         while (tok != end) {
1740             if (tok->str() == "static")
1741                 setFlag(fIsStatic, true);
1742             else if (tok->str() == "extern")
1743                 setFlag(fIsExtern, true);
1744             else if (tok->str() == "volatile")
1745                 setFlag(fIsVolatile, true);
1746             else if (tok->str() == "mutable")
1747                 setFlag(fIsMutable, true);
1748             else if (tok->str() == "const")
1749                 setFlag(fIsConst, true);
1750             else if (tok->str() == "*") {
1751                 setFlag(fIsPointer, !isArray() || Token::Match(tok->previous(), "( * %name% )"));
1752                 setFlag(fIsConst, false); // Points to const, isn't necessarily const itself
1753             } else if (tok->str() == "&") {
1754                 if (isReference())
1755                     setFlag(fIsRValueRef, true);
1756                 setFlag(fIsReference, true);
1757             } else if (tok->str() == "&&") { // Before simplification, && isn't split up
1758                 setFlag(fIsRValueRef, true);
1759                 setFlag(fIsReference, true); // Set also fIsReference
1760             }
1761     
1762             if (tok->str() == "<" && tok->link())
1763                 tok = tok->link();
1764             else
1765                 tok = tok->next();
1766         }
1767     
1768         while (Token::Match(mTypeStartToken, "static|const|volatile %any%"))
1769             mTypeStartToken = mTypeStartToken->next();
1770         while (mTypeEndToken && mTypeEndToken->previous() && Token::Match(mTypeEndToken, "const|volatile"))
1771             mTypeEndToken = mTypeEndToken->previous();
1772     
1773         if (mTypeStartToken) {
1774             std::string strtype = mTypeStartToken->str();
1775             for (const Token *typeToken = mTypeStartToken; Token::Match(typeToken, "%type% :: %type%"); typeToken = typeToken->tokAt(2))
1776                 strtype += "::" + typeToken->strAt(2);
1777             setFlag(fIsClass, !lib->podtype(strtype) && !mTypeStartToken->isStandardType() && !isEnumType() && !isPointer() && !isReference() && strtype != "...");
1778             setFlag(fIsStlType, Token::simpleMatch(mTypeStartToken, "std ::"));
1779             setFlag(fIsStlString, isStlType() && (Token::Match(mTypeStartToken->tokAt(2), "string|wstring|u16string|u32string !!::") || (Token::simpleMatch(mTypeStartToken->tokAt(2), "basic_string <") && !Token::simpleMatch(mTypeStartToken->linkAt(3), "> ::"))));
1780             setFlag(fIsSmartPointer, lib->isSmartPointer(mTypeStartToken));
1781         }
1782         if (mAccess == AccessControl::Argument) {
1783             tok = mNameToken;
1784             if (!tok) {
1785                 // Argument without name
1786                 tok = mTypeEndToken;
1787                 // back up to start of array dimensions
1788                 while (tok && tok->str() == "]")
1789                     tok = tok->link()->previous();
1790                 // add array dimensions if present
1791                 if (tok && tok->next()->str() == "[")
1792                     setFlag(fIsArray, arrayDimensions(settings));
1793             }
1794             if (!tok)
1795                 return;
1796             tok = tok->next();
1797             while (tok->str() == "[")
1798                 tok = tok->link();
1799             setFlag(fHasDefault, tok->str() == "=");
1800         }
1801         // check for C++11 member initialization
1802         if (mScope && mScope->isClassOrStruct()) {
1803             // type var = x or
1804             // type var = {x}
1805             // type var = x; gets simplified to: type var ; var = x ;
1806             Token const * declEnd = declEndToken();
1807             if ((Token::Match(declEnd, "; %name% =") && declEnd->strAt(1) == mNameToken->str()) ||
1808                 Token::Match(declEnd, "=|{"))
1809                 setFlag(fHasDefault, true);
1810         }
1811     
1812         if (mTypeStartToken) {
1813             if (Token::Match(mTypeStartToken, "float|double"))
1814                 setFlag(fIsFloatType, true);
1815         }
1816     }

1846     Function::Function(const Tokenizer *mTokenizer,
1847                        const Token *tok,
1848                        const Scope *scope,
1849                        const Token *tokDef,
1850                        const Token *tokArgDef)
1851         : tokenDef(tokDef),
1852           argDef(tokArgDef),
1853           token(nullptr),
1854           arg(nullptr),
1855           retDef(nullptr),
1856           retType(nullptr),
1857           functionScope(nullptr),
1858           nestedIn(scope),
1859           initArgCount(0),
1860           type(eFunction),
1861           access(AccessControl::Public),
1862           noexceptArg(nullptr),
1863           throwArg(nullptr),
1864           templateDef(nullptr),
1865           mFlags(0)
1866     {
1867         // operator function
1868         if (tokenDef->isOperatorKeyword()) {
1869             isOperator(true);
1870     
1871             // 'operator =' is special
1872             if (tokenDef->str() == "operator=")
1873                 type = Function::eOperatorEqual;
1874         }
1875     
1876         else if (tokenDef->str() == "[") {
1877             type = Function::eLambda;
1878         }
1879     
1880         // class constructor/destructor
1881         else if (tokenDef->str() == scope->className) {
1882             // destructor
1883             if (tokenDef->previous()->str() == "~")
1884                 type = Function::eDestructor;
1885             // constructor of any kind
1886             else
1887                 type = Function::eConstructor;
1888     
1889             isExplicit(tokenDef->previous()->str() == "explicit");
1890         }
1891     
1892         const Token *tok1 = tok;
1893     
1894         // look for end of previous statement
1895         while (tok1->previous() && !Token::Match(tok1->previous(), ";|}|{|public:|protected:|private:")) {
1896             tok1 = tok1->previous();
1897     
1898             // extern function
1899             if (tok1->str() == "extern") {
1900                 isExtern(true);
1901             }
1902     
1903             // virtual function
1904             else if (tok1->str() == "virtual") {
1905                 hasVirtualSpecifier(true);
1906             }
1907     
1908             // static function
1909             else if (tok1->str() == "static") {
1910                 isStatic(true);
1911                 if (scope->type == Scope::eNamespace || scope->type == Scope::eGlobal)
1912                     isStaticLocal(true);
1913             }
1914     
1915             // friend function
1916             else if (tok1->str() == "friend") {
1917                 isFriend(true);
1918             }
1919     
1920             // Function template
1921             else if (tok1->link() && tok1->str() == ">" && Token::simpleMatch(tok1->link()->previous(), "template <")) {
1922                 templateDef = tok1->link()->previous();
1923                 break;
1924             }
1925         }
1926     
1927         // find the return type
1928         if (!isConstructor() && !isDestructor() && !isLambda()) {
1929             // @todo auto type deduction should be checked
1930             // @todo attributes and exception specification can also precede trailing return type
1931             if (Token::Match(argDef->link()->next(), "const|volatile| &|&&| .")) { // Trailing return type
1932                 hasTrailingReturnType(true);
1933                 if (argDef->link()->strAt(1) == ".")
1934                     retDef = argDef->link()->tokAt(2);
1935                 else if (argDef->link()->strAt(2) == ".")
1936                     retDef = argDef->link()->tokAt(3);
1937                 else if (argDef->link()->strAt(3) == ".")
1938                     retDef = argDef->link()->tokAt(4);
1939             } else {
1940                 if (tok1->str() == ">")
1941                     tok1 = tok1->next();
1942                 while (Token::Match(tok1, "extern|virtual|static|friend|struct|union|enum"))
1943                     tok1 = tok1->next();
1944                 retDef = tok1;
1945             }
1946         }
1947     
1948         const Token *end = argDef->link();
1949     
1950         // parse function attributes..
1951         tok = end->next();
1952         while (tok) {
1953             if (tok->str() == "const")
1954                 isConst(true);
1955             else if (tok->str() == "&")
1956                 hasLvalRefQualifier(true);
1957             else if (tok->str() == "&&")
1958                 hasRvalRefQualifier(true);
1959             else if (tok->str() == "override")
1960                 setFlag(fHasOverrideSpecifier, true);
1961             else if (tok->str() == "final")
1962                 setFlag(fHasFinalSpecifier, true);
1963             else if (tok->str() == "volatile")
1964                 isVolatile(true);
1965             else if (tok->str() == "noexcept") {
1966                 isNoExcept(!Token::simpleMatch(tok->next(), "( false )"));
1967                 if (tok->next()->str() == "(")
1968                     tok = tok->linkAt(1);
1969             } else if (Token::simpleMatch(tok, "throw (")) {
1970                 isThrow(true);
1971                 if (tok->strAt(2) != ")")
1972                     throwArg = tok->next();
1973                 tok = tok->linkAt(1);
1974             } else if (Token::Match(tok, "= 0|default|delete ;")) {
1975                 const std::string& modifier = tok->strAt(1);
1976                 isPure(modifier == "0");
1977                 isDefault(modifier == "default");
1978                 isDelete(modifier == "delete");
1979             } else if (tok->str() == ".") { // trailing return type
1980                 // skip over return type
1981                 while (tok && !Token::Match(tok->next(), ";|{|override|final"))
1982                     tok = tok->next();
1983             } else
1984                 break;
1985             if (tok)
1986                 tok = tok->next();
1987         }
1988     
1989         if (mTokenizer->isFunctionHead(end, ":{")) {
1990             // assume implementation is inline (definition and implementation same)
1991             token = tokenDef;
1992             arg = argDef;
1993             isInline(true);
1994             hasBody(true);
1995         }
1996     }

1998     bool Function::argsMatch(const Scope *scope, const Token *first, const Token *second, const std::string &path, nonneg int path_length)
1999     {
2000         const bool isCPP = scope->check->isCPP();
2001         if (!isCPP) // C does not support overloads
2002             return true;
2003     
2004         int arg_path_length = path_length;
2005     
2006         while (first->str() == second->str() &&
2007                first->isLong() == second->isLong() &&
2008                first->isUnsigned() == second->isUnsigned()) {
2009     
2010             // skip optional type information
2011             if (Token::Match(first->next(), "struct|enum|union|class"))
2012                 first = first->next();
2013             if (Token::Match(second->next(), "struct|enum|union|class"))
2014                 second = second->next();
2015     
2016             // skip const on type passed by value
2017             if (Token::Match(first->next(), "const %type% %name%|,|)") &&
2018                 !Token::Match(first->next(), "const %type% %name%| ["))
2019                 first = first->next();
2020             if (Token::Match(second->next(), "const %type% %name%|,|)") &&
2021                 !Token::Match(second->next(), "const %type% %name%| ["))
2022                 second = second->next();
2023     
2024             // at end of argument list
2025             if (first->str() == ")") {
2026                 return true;
2027             }
2028     
2029             // skip default value assignment
2030             else if (first->next()->str() == "=") {
2031                 first = first->nextArgument();
2032                 if (first)
2033                     first = first->tokAt(-2);
2034                 if (second->next()->str() == "=") {
2035                     second = second->nextArgument();
2036                     if (second)
2037                         second = second->tokAt(-2);
2038                     if (!first || !second) { // End of argument list (first or second)
2039                         return !first && !second;
2040                     }
2041                 } else if (!first) { // End of argument list (first)
2042                     return !second->nextArgument(); // End of argument list (second)
2043                 }
2044             } else if (second->next()->str() == "=") {
2045                 second = second->nextArgument();
2046                 if (second)
2047                     second = second->tokAt(-2);
2048                 if (!second) { // End of argument list (second)
2049                     return false;
2050                 }
2051             }
2052     
2053             // definition missing variable name
2054             else if ((first->next()->str() == "," && second->next()->str() != ",") ||
2055                      (first->next()->str() == ")" && second->next()->str() != ")")) {
2056                 second = second->next();
2057                 // skip default value assignment
2058                 if (second->next()->str() == "=") {
2059                     do {
2060                         second = second->next();
2061                     } while (!Token::Match(second->next(), ",|)"));
2062                 }
2063             } else if (first->next()->str() == "[" && second->next()->str() != "[")
2064                 second = second->next();
2065     
2066             // function missing variable name
2067             else if ((second->next()->str() == "," && first->next()->str() != ",") ||
2068                      (second->next()->str() == ")" && first->next()->str() != ")")) {
2069                 first = first->next();
2070                 // skip default value assignment
2071                 if (first->next()->str() == "=") {
2072                     do {
2073                         first = first->next();
2074                     } while (!Token::Match(first->next(), ",|)"));
2075                 }
2076             } else if (second->next()->str() == "[" && first->next()->str() != "[")
2077                 first = first->next();
2078     
2079             // argument list has different number of arguments
2080             else if (second->str() == ")")
2081                 break;
2082     
2083             // ckeck for type * x == type x[]
2084             else if (Token::Match(first->next(), "* %name%| ,|)|=") &&
2085                      Token::Match(second->next(), "%name%| [ ] ,|)")) {
2086                 do {
2087                     first = first->next();
2088                 } while (!Token::Match(first->next(), ",|)"));
2089                 do {
2090                     second = second->next();
2091                 } while (!Token::Match(second->next(), ",|)"));
2092             }
2093     
2094             // const after *
2095             else if (first->next()->str() == "*" && first->strAt(2) != "const" &&
2096                      second->next()->str() == "*" && second->strAt(2) == "const") {
2097                 first = first->next();
2098                 second = second->tokAt(2);
2099             }
2100     
2101             // variable names are different
2102             else if ((Token::Match(first->next(), "%name% ,|)|=|[") &&
2103                       Token::Match(second->next(), "%name% ,|)|[")) &&
2104                      (first->next()->str() != second->next()->str())) {
2105                 // skip variable names
2106                 first = first->next();
2107                 second = second->next();
2108     
2109                 // skip default value assignment
2110                 if (first->next()->str() == "=") {
2111                     do {
2112                         first = first->next();
2113                     } while (!Token::Match(first->next(), ",|)"));
2114                 }
2115             }
2116     
2117             // variable with class path
2118             else if (arg_path_length && Token::Match(first->next(), "%name%") && first->strAt(1) != "const") {
2119                 std::string param = path;
2120     
2121                 if (Token::simpleMatch(second->next(), param.c_str())) {
2122                     second = second->tokAt(int(arg_path_length));
2123                     arg_path_length = 0;
2124                 }
2125     
2126                 // nested or base class variable
2127                 else if (arg_path_length <= 2 && Token::Match(first->next(), "%name%") &&
2128                          (Token::Match(second->next(), "%name% :: %name%") ||
2129                           (Token::Match(second->next(), "%name% <") &&
2130                            Token::Match(second->linkAt(1), "> :: %name%"))) &&
2131                          ((second->next()->str() == scope->className) ||
2132                           (scope->definedType && scope->definedType->isDerivedFrom(second->next()->str()))) &&
2133                          (first->next()->str() == second->strAt(3))) {
2134                     if (Token::Match(second->next(), "%name% <"))
2135                         second = second->linkAt(1)->next();
2136                     else
2137                         second = second->tokAt(2);
2138                 }
2139     
2140                 // remove class name
2141                 else if (arg_path_length > 2 && first->strAt(1) != second->strAt(1)) {
2142                     std::string short_path = path;
2143                     unsigned int short_path_length = arg_path_length;
2144     
2145                     // remove last " :: "
2146                     short_path.resize(short_path.size() - 4);
2147                     short_path_length--;
2148     
2149                     // remove last name
2150                     std::string::size_type lastSpace = short_path.find_last_of(' ');
2151                     if (lastSpace != std::string::npos) {
2152                         short_path.resize(lastSpace+1);
2153                         short_path_length--;
2154                         if (short_path[short_path.size() - 1] == '>') {
2155                             short_path.resize(short_path.size() - 3);
2156                             while (short_path[short_path.size() - 1] == '<') {
2157                                 lastSpace = short_path.find_last_of(' ');
2158                                 short_path.resize(lastSpace+1);
2159                                 short_path_length--;
2160                             }
2161                         }
2162                     }
2163     
2164                     param = short_path;
2165                     if (Token::simpleMatch(second->next(), param.c_str())) {
2166                         second = second->tokAt(int(short_path_length));
2167                         arg_path_length = 0;
2168                     }
2169                 }
2170             }
2171     
2172             first = first->next();
2173             second = second->next();
2174     
2175             // reset path length
2176             if (first->str() == "," || second->str() == ",")
2177                 arg_path_length = path_length;
2178         }
2179     
2180         return false;
2181     }

2264     {
2265         const bool destructor((*tok)->previous()->str() == "~");
2266         const bool has_const(argStart->link()->strAt(1) == "const");
2267         const bool lval(argStart->link()->strAt(has_const ? 2 : 1) == "&");
2268         const bool rval(argStart->link()->strAt(has_const ? 2 : 1) == "&&");
2269         int count = 0;
2270         std::string path;
2271         unsigned int path_length = 0;
2272         const Token *tok1 = (*tok);
2273     
2274         if (destructor)
2275             tok1 = tok1->previous();
2276     
2277         // back up to head of path
2278         while (tok1 && tok1->previous() && tok1->previous()->str() == "::" && tok1->tokAt(-2) &&
2279                (tok1->tokAt(-2)->isName() ||
2280                 (tok1->strAt(-2) == ">" && tok1->linkAt(-2) && Token::Match(tok1->linkAt(-2)->previous(), "%name%")))) {
2281             count++;
2282             const Token * tok2 = tok1->tokAt(-2);
2283             if (tok2->str() == ">")
2284                 tok2 = tok2->link()->previous();
2285     
2286             if (tok2) {
2287                 do {
2288                     path = tok1->previous()->str() + " " + path;
2289                     tok1 = tok1->previous();
2290                     path_length++;
2291                 } while (tok1 != tok2);
2292             } else
2293                 return; // syntax error ?
2294         }
2295     
2296         // syntax error?
2297         if (!tok1)
2298             return;
2299     
2300         std::list<Scope>::iterator it1;
2301     
2302         // search for match

2313                         Function * func = findFunctionInScope(tok1, it2->scope, path, path_length);

2315                             if (!func->hasBody()) {
2316                                 func->hasBody(true);
2317                                 func->token = *tok;
2318                                 func->arg = argStart;
2319                                 addNewFunction(scope, tok);
2320                                 if (*scope) {
2321                                     (*scope)->functionOf = func->nestedIn;
2322                                     (*scope)->function = func;
2323                                     (*scope)->function->functionScope = *scope;
2324                                 }
2325                                 return;
2326                             }
2327                         }
2328                     }
2329                 }
2330             }
2331

2335                                                      (*scope)->className == scope1->nestedIn->className &&

2344                             if (s1->className != s2->className)

2347                             s2 = s2->nestedIn;
2348                         }

2369                 }
2370             }
2371

2386                                         func->hasBody(true);
2387                                     }
2388                                 }
2389                             }
2390     
2391                             if (func->hasBody()) {
2392                                 func->token = *tok;
2393                                 func->arg = argStart;

2398                                     (*scope)->function->functionScope = *scope;
2399                                 }
2400                                 return;
2401                             }
2402                         }
2403                     }
2404                 }

2421             if (tok1->link() && Token::Match(tok1, "(|<")) {
2422                 tok1 = tok1->link();

2424                        Token::Match(tok1, "%name%|> {") &&
2425                        Token::Match(tok1->linkAt(1), "} ,|{")) {
2426                 tok1 = tok1->linkAt(1);
2427             } else {
2428                 if (tok1->str() == ":")
2429                     foundInitList = true;
2430                 tok1 = tok1->next();
2431             }
2432         }
2433     
2434         if (tok1 && tok1->str() == "{") {
2435             newScope->bodyStart = tok1;
2436             newScope->bodyEnd = tok1->link();

2451         } else {
2452             scopeList.pop_back();

4262         // check each function against the arguments in the function call for a match
4263         for (std::size_t i = 0; i < matches.size();) {
4264             bool constFallback = false;
4265             const Function * func = matches[i];
4266             size_t same = 0;
4267     
4268             if (requireConst && !func->isConst()) {
4269                 i++;
4270                 continue;
4271             }
4272     
4273             if (!requireConst || !func->isConst()) {
4274                 // get the function this call is in
4275                 const Scope * scope = tok->scope();
4276     
4277                 // check if this function is a member function
4278                 if (scope && scope->functionOf && scope->functionOf->isClassOrStruct() && scope->function) {
4279                     // check if isConst mismatches
4280                     if (scope->function->isConst() != func->isConst()) {
4281                         if (scope->function->isConst()) {
4282                             ++i;
4283                             continue;
4284                         }
4285                         constFallback = true;
4286                     }
4287                 }
4288             }
4289     
4290             size_t fallback1 = 0;
4291             size_t fallback2 = 0;
4292             bool erased = false;
4293             for (std::size_t j = 0; j < args; ++j) {
4294     
4295                 // don't check variadic arguments
4296                 if (func->isVariadic() && j > (func->argCount() - 1)) {
4297                     break;
4298                 }
4299                 const Variable *funcarg = func->getArgumentVar(j);
4300     
4301                 if (!arguments[j]->valueType()) {
4302                     const Token *vartok = arguments[j];
4303                     int pointer = 0;
4304                     while (vartok && (vartok->isUnaryOp("&") || vartok->isUnaryOp("*"))) {
4305                         pointer += vartok->isUnaryOp("&") ? 1 : -1;
4306                         vartok = vartok->astOperand1();
4307                     }
4308                     if (vartok && vartok->variable()) {
4309                         const Token *callArgTypeToken = vartok->variable()->typeStartToken();
4310                         const Token *funcArgTypeToken = funcarg->typeStartToken();
4311     
4312                         auto parseDecl = [](const Token *typeToken) -> ValueType {
4313                             ValueType ret;
4314                             while (Token::Match(typeToken->previous(), "%name%"))
4315                                 typeToken = typeToken->previous();
4316                             while (Token::Match(typeToken, "%name%|*|&|::|<"))
4317                             {
4318                                 if (typeToken->str() == "const")
4319                                     ret.constness |= (1 << ret.pointer);
4320                                 else if (typeToken->str() == "*")
4321                                     ret.pointer++;
4322                                 else if (typeToken->str() == "<") {
4323                                     if (!typeToken->link())
4324                                         break;
4325                                     typeToken = typeToken->link();
4326                                 }
4327                                 typeToken = typeToken->next();
4328                             }
4329                             return ret;
4330                         };
4331     
4332                         const std::string type1 = getTypeString(callArgTypeToken);
4333                         const std::string type2 = getTypeString(funcArgTypeToken);
4334                         if (!type1.empty() && type1 == type2) {
4335                             ValueType callArgType = parseDecl(callArgTypeToken);
4336                             callArgType.pointer += pointer;
4337                             ValueType funcArgType = parseDecl(funcArgTypeToken);
4338     
4339                             callArgType.sign = funcArgType.sign = ValueType::Sign::SIGNED;
4340                             callArgType.type = funcArgType.type = ValueType::Type::INT;
4341     
4342                             ValueType::MatchResult res = ValueType::matchParameter(&callArgType, &funcArgType);
4343                             if (res == ValueType::MatchResult::SAME)
4344                                 ++same;
4345                             else if (res == ValueType::MatchResult::FALLBACK1)
4346                                 ++fallback1;
4347                             else if (res == ValueType::MatchResult::FALLBACK2)
4348                                 ++fallback2;
4349                             continue;
4350                         }
4351                     }
4352                 }
4353     
4354                 // check for a match with a variable
4355                 if (Token::Match(arguments[j], "%var% ,|)")) {
4356                     const Variable * callarg = arguments[j]->variable();
4357                     checkVariableCallMatch(callarg, funcarg, same, fallback1, fallback2);
4358                 }
4359     
4360                 else if (funcarg->isStlStringType() && arguments[j]->valueType() && arguments[j]->valueType()->pointer == 1 && arguments[j]->valueType()->type == ValueType::Type::CHAR)
4361                     fallback2++;
4362     
4363                 // check for a match with nullptr
4364                 else if (funcarg->isPointer() && Token::Match(arguments[j], "nullptr|NULL ,|)"))
4365                     same++;
4366     
4367                 else if (arguments[j]->isNumber() && funcarg->isPointer() && MathLib::isNullValue(arguments[j]->str()))
4368                     fallback1++;
4369     
4370                 // Try to evaluate the apparently more complex expression
4371                 else {
4372                     const Token *vartok = arguments[j];
4373                     while (vartok->isUnaryOp("&") || vartok->isUnaryOp("*"))
4374                         vartok = vartok->astOperand1();
4375                     ValueType::MatchResult res = ValueType::matchParameter(arguments[j]->valueType(), vartok->variable(), funcarg);
4376                     if (res == ValueType::MatchResult::SAME)
4377                         ++same;
4378                     else if (res == ValueType::MatchResult::FALLBACK1)
4379                         ++fallback1;
4380                     else if (res == ValueType::MatchResult::FALLBACK2)
4381                         ++fallback2;
4382                     else if (res == ValueType::MatchResult::NOMATCH) {
4383                         // can't match so remove this function from possible matches
4384                         matches.erase(matches.begin() + i);
4385                         erased = true;
4386                         break;
4387                     }
4388                 }
4389             }
4390     
4391             const size_t hasToBe = func->isVariadic() ? (func->argCount() - 1) : args;
4392     
4393             // check if all arguments matched
4394             if (same == hasToBe) {
4395                 if (constFallback || (!requireConst && func->isConst()))
4396                     fallback1Func = func;
4397                 else
4398                     return func;
4399             }
4400     
4401             else if (!fallback1Func) {
4402                 if (same + fallback1 == hasToBe)
4403                     fallback1Func = func;
4404                 else if (!fallback2Func && same + fallback2 + fallback1 == hasToBe)
4405                     fallback2Func = func;
4406             }
4407     
4408             if (!erased)
4409                 ++i;
4410         }
4411     
4412         // Fallback cases
4413         if (fallback1Func)
4414             return fallback1Func;
4415     
4416         if (fallback2Func)
4417             return fallback2Func;
4418     
4419         // Only one candidate left
4420         if (matches.size() == 1)
4421             return matches[0];
4422     
4423         return nullptr;
4424     }

4428     const Function* SymbolDatabase::findFunction(const Token *tok) const
4429     {
4430         // find the scope this function is in
4431         const Scope *currScope = tok->scope();
4432         while (currScope && currScope->isExecutable()) {
4433             if (currScope->functionOf)
4434                 currScope = currScope->functionOf;
4435             else
4436                 currScope = currScope->nestedIn;
4437         }
4438     
4439         // check for a qualified name and use it when given
4440         if (tok->strAt(-1) == "::") {
4441             // find start of qualified function name
4442             const Token *tok1 = tok;
4443     
4444             while (Token::Match(tok1->tokAt(-2), ">|%type% ::")) {
4445                 if (tok1->strAt(-2) == ">") {
4446                     if (tok1->linkAt(-2))
4447                         tok1 = tok1->linkAt(-2)->tokAt(-1);
4448                     else {
4449                         if (mSettings->debugwarnings)
4450                             debugMessage(tok1->tokAt(-2), "SymbolDatabase::findFunction found '>' without link.");
4451                         return nullptr;
4452                     }
4453                 } else
4454                     tok1 = tok1->tokAt(-2);
4455             }
4456     
4457             // check for global scope
4458             if (tok1->strAt(-1) == "::") {
4459                 currScope = &scopeList.front();
4460     
4461                 currScope = currScope->findRecordInNestedList(tok1->str());
4462             }
4463     
4464             // find start of qualification
4465             else {
4466                 while (currScope) {
4467                     if (currScope->className == tok1->str())
4468                         break;
4469                     else {
4470                         const Scope *scope = currScope->findRecordInNestedList(tok1->str());
4471     
4472                         if (scope) {
4473                             currScope = scope;
4474                             break;
4475                         } else
4476                             currScope = currScope->nestedIn;
4477                     }
4478                 }
4479             }
4480     
4481             if (currScope) {
4482                 while (currScope && !(Token::Match(tok1, "%type% :: %any% (") ||
4483                                       (Token::Match(tok1, "%type% <") && Token::Match(tok1->linkAt(1), "> :: %any% (")))) {
4484                     if (tok1->strAt(1) == "::")
4485                         tok1 = tok1->tokAt(2);
4486                     else
4487                         tok1 = tok1->linkAt(1)->tokAt(2);
4488                     currScope = currScope->findRecordInNestedList(tok1->str());
4489                 }
4490     
4491                 tok1 = tok1->tokAt(2);
4492     
4493                 if (currScope && tok1)
4494                     return currScope->findFunction(tok1);
4495             }
4496         }
4497     
4498         // check for member function
4499         else if (Token::Match(tok->tokAt(-2), "!!this .")) {
4500             const Token *tok1 = tok->tokAt(-2);
4501             if (Token::Match(tok1, "%var% .")) {
4502                 const Variable *var = getVariableFromVarId(tok1->varId());
4503                 if (var && var->typeScope())
4504                     return var->typeScope()->findFunction(tok, var->valueType()->constness == 1);
4505                 if (var && var->smartPointerType() && var->smartPointerType()->classScope && tok1->next()->originalName() == "->")
4506                     return var->smartPointerType()->classScope->findFunction(tok, var->valueType()->constness == 1);
4507             } else if (Token::simpleMatch(tok->previous()->astOperand1(), "(")) {
4508                 const Token *castTok = tok->previous()->astOperand1();
4509                 if (castTok->isCast()) {
4510                     ValueType vt = ValueType::parseDecl(castTok->next(),mSettings);
4511                     if (vt.typeScope)
4512                         return vt.typeScope->findFunction(tok, vt.constness == 1);
4513                 }
4514             }
4515         }
4516     
4517         // check in enclosing scopes
4518         else {
4519             while (currScope) {
4520                 const Function *func = currScope->findFunction(tok);
4521                 if (func)
4522                     return func;
4523                 currScope = currScope->nestedIn;
4524             }
4525         }
4526         return nullptr;
4527     }

5281     static const Token * parsedecl(const Token *type, ValueType * const valuetype, ValueType::Sign defaultSignedness, const Settings* settings)
5282     {
5283         const Token * const previousType = type;
5284         const unsigned int pointer0 = valuetype->pointer;
5285         while (Token::Match(type->previous(), "%name%"))
5286             type = type->previous();
5287         valuetype->sign = ValueType::Sign::UNKNOWN_SIGN;
5288         if (!valuetype->typeScope && !valuetype->smartPointerType)
5289             valuetype->type = ValueType::Type::UNKNOWN_TYPE;
5290         else if (valuetype->smartPointerType)
5291             valuetype->type = ValueType::Type::NONSTD;
5292         else if (valuetype->typeScope->type == Scope::eEnum) {
5293             const Token * enum_type = valuetype->typeScope->enumType;
5294             if (enum_type) {
5295                 if (enum_type->isSigned())
5296                     valuetype->sign = ValueType::Sign::SIGNED;
5297                 else if (enum_type->isUnsigned())
5298                     valuetype->sign = ValueType::Sign::UNSIGNED;
5299                 else
5300                     valuetype->sign = defaultSignedness;
5301                 const ValueType::Type t = ValueType::typeFromString(enum_type->str(), enum_type->isLong());
5302                 if (t != ValueType::Type::UNKNOWN_TYPE)
5303                     valuetype->type = t;
5304                 else if (enum_type->isStandardType())
5305                     valuetype->fromLibraryType(enum_type->str(), settings);
5306             } else
5307                 valuetype->type = ValueType::Type::INT;
5308         } else
5309             valuetype->type = ValueType::Type::RECORD;
5310         while (Token::Match(type, "%name%|*|&|::") && !Token::Match(type, "typename|template") &&
5311                !type->variable() && !type->function()) {
5312             if (type->isSigned())
5313                 valuetype->sign = ValueType::Sign::SIGNED;
5314             else if (type->isUnsigned())
5315                 valuetype->sign = ValueType::Sign::UNSIGNED;
5316             if (valuetype->type == ValueType::Type::UNKNOWN_TYPE &&
5317                 type->type() && type->type()->isTypeAlias() && type->type()->typeStart &&
5318                 type->type()->typeStart->str() != type->str() && type->type()->typeStart != previousType)
5319                 parsedecl(type->type()->typeStart, valuetype, defaultSignedness, settings);
5320             else if (type->str() == "const")
5321                 valuetype->constness |= (1 << (valuetype->pointer - pointer0));
5322             else if (const Library::Container *container = settings->library.detectContainer(type)) {
5323                 valuetype->type = ValueType::Type::CONTAINER;
5324                 valuetype->container = container;
5325                 while (Token::Match(type, "%name%|::|<")) {
5326                     if (type->str() == "<" && type->link()) {
5327                         if (container->type_templateArgNo >= 0) {
5328                             const Token *templateType = type->next();
5329                             for (int j = 0; templateType && j < container->type_templateArgNo; j++)
5330                                 templateType = templateType->nextTemplateArgument();
5331                             valuetype->containerTypeToken = templateType;
5332                         }
5333                         type = type->link();
5334                     }
5335                     type = type->next();
5336                 }
5337                 continue;
5338             } else if (settings->library.isSmartPointer(type)) {
5339                 const Token* argTok = Token::findsimplematch(type, "<");
5340                 if (!argTok)
5341                     continue;
5342                 valuetype->smartPointerTypeToken = argTok->next();
5343                 valuetype->smartPointerType = argTok->next()->type();
5344             } else if (Token::Match(type, "%name% :: %name%")) {
5345                 std::string typestr;
5346                 const Token *end = type;
5347                 while (Token::Match(end, "%name% :: %name%")) {
5348                     typestr += end->str() + "::";
5349                     end = end->tokAt(2);
5350                 }
5351                 typestr += end->str();
5352                 if (valuetype->fromLibraryType(typestr, settings))
5353                     type = end;
5354             } else if (ValueType::Type::UNKNOWN_TYPE != ValueType::typeFromString(type->str(), type->isLong()))
5355                 valuetype->type = ValueType::typeFromString(type->str(), type->isLong());
5356             else if (type->str() == "auto") {
5357                 const ValueType *vt = type->valueType();
5358                 if (!vt)
5359                     return nullptr;
5360                 valuetype->type = vt->type;
5361                 valuetype->pointer = vt->pointer;
5362                 if (vt->sign != ValueType::Sign::UNKNOWN_SIGN)
5363                     valuetype->sign = vt->sign;
5364                 valuetype->constness = vt->constness;
5365                 valuetype->originalTypeName = vt->originalTypeName;
5366                 while (Token::Match(type, "%name%|*|&|::") && !type->variable())
5367                     type = type->next();
5368                 break;
5369             } else if (!valuetype->typeScope && (type->str() == "struct" || type->str() == "enum"))
5370                 valuetype->type = type->str() == "struct" ? ValueType::Type::RECORD : ValueType::Type::NONSTD;
5371             else if (!valuetype->typeScope && type->type() && type->type()->classScope) {
5372                 valuetype->type = ValueType::Type::RECORD;
5373                 valuetype->typeScope = type->type()->classScope;
5374             } else if (type->isName() && valuetype->sign != ValueType::Sign::UNKNOWN_SIGN && valuetype->pointer == 0U)
5375                 return nullptr;
5376             else if (type->str() == "*")
5377                 valuetype->pointer++;
5378             else if (type->isStandardType())
5379                 valuetype->fromLibraryType(type->str(), settings);
5380             else if (Token::Match(type->previous(), "!!:: %name% !!::"))
5381                 valuetype->fromLibraryType(type->str(), settings);
5382             if (!type->originalName().empty())
5383                 valuetype->originalTypeName = type->originalName();
5384             type = type->next();
5385         }
5386     
5387         // Set signedness for integral types..
5388         if (valuetype->isIntegral() && valuetype->sign == ValueType::Sign::UNKNOWN_SIGN) {
5389             if (valuetype->type == ValueType::Type::CHAR)
5390                 valuetype->sign = defaultSignedness;
5391             else if (valuetype->type >= ValueType::Type::SHORT)
5392                 valuetype->sign = ValueType::Sign::SIGNED;
5393         }
5394     
5395         return (type && (valuetype->type != ValueType::Type::UNKNOWN_TYPE || valuetype->pointer > 0)) ? type : nullptr;
5396     }

5427     void SymbolDatabase::setValueTypeInTokenList(bool reportDebugWarnings)
5428     {
5429         Token * tokens = const_cast<Tokenizer *>(mTokenizer)->list.front();
5430     
5431         for (Token *tok = tokens; tok; tok = tok->next())
5432             tok->setValueType(nullptr);
5433     
5434         for (Token *tok = tokens; tok; tok = tok->next()) {
5435             if (tok->isNumber()) {
5436                 if (MathLib::isFloat(tok->str())) {
5437                     ValueType::Type type = ValueType::Type::DOUBLE;
5438                     const char suffix = tok->str()[tok->str().size() - 1];
5439                     if (suffix == 'f' || suffix == 'F')
5440                         type = ValueType::Type::FLOAT;
5441                     else if (suffix == 'L' || suffix == 'l')
5442                         type = ValueType::Type::LONGDOUBLE;
5443                     setValueType(tok, ValueType(ValueType::Sign::UNKNOWN_SIGN, type, 0U));
5444                 } else if (MathLib::isInt(tok->str())) {
5445                     const bool unsignedSuffix = (tok->str().find_last_of("uU") != std::string::npos);
5446                     ValueType::Sign sign = unsignedSuffix ? ValueType::Sign::UNSIGNED : ValueType::Sign::SIGNED;
5447                     ValueType::Type type;
5448                     const MathLib::bigint value = MathLib::toLongNumber(tok->str());
5449                     if (mSettings->platformType == cppcheck::Platform::Unspecified)
5450                         type = ValueType::Type::INT;
5451                     else if (mSettings->isIntValue(unsignedSuffix ? (value >> 1) : value))
5452                         type = ValueType::Type::INT;
5453                     else if (mSettings->isLongValue(unsignedSuffix ? (value >> 1) : value))
5454                         type = ValueType::Type::LONG;
5455                     else
5456                         type = ValueType::Type::LONGLONG;
5457                     if (MathLib::isIntHex(tok->str()))
5458                         sign = ValueType::Sign::UNSIGNED;
5459                     for (std::size_t pos = tok->str().size() - 1U; pos > 0U; --pos) {
5460                         const char suffix = tok->str()[pos];
5461                         if (suffix == 'u' || suffix == 'U')
5462                             sign = ValueType::Sign::UNSIGNED;
5463                         else if (suffix == 'l' || suffix == 'L')
5464                             type = (type == ValueType::Type::INT) ? ValueType::Type::LONG : ValueType::Type::LONGLONG;
5465                         else if (pos > 2U && suffix == '4' && tok->str()[pos - 1] == '6' && tok->str()[pos - 2] == 'i') {
5466                             type = ValueType::Type::LONGLONG;
5467                             pos -= 2;
5468                         } else break;
5469                     }
5470                     setValueType(tok, ValueType(sign, type, 0U));
5471                 }
5472             } else if (tok->isComparisonOp() || tok->tokType() == Token::eLogicalOp) {
5473                 if (mIsCpp && tok->isComparisonOp() && (getClassScope(tok->astOperand1()) || getClassScope(tok->astOperand2()))) {
5474                     const Function *function = getOperatorFunction(tok);
5475                     if (function) {
5476                         ValueType vt;
5477                         parsedecl(function->retDef, &vt, mDefaultSignedness, mSettings);
5478                         setValueType(tok, vt);
5479                         continue;
5480                     }
5481                 }
5482                 setValueType(tok, ValueType(ValueType::Sign::UNKNOWN_SIGN, ValueType::Type::BOOL, 0U));
5483             } else if (tok->isBoolean()) {
5484                 setValueType(tok, ValueType(ValueType::Sign::UNKNOWN_SIGN, ValueType::Type::BOOL, 0U));
5485             } else if (tok->tokType() == Token::eChar) {
5486                 setValueType(tok, ValueType(ValueType::Sign::UNKNOWN_SIGN, tok->isLong() ? ValueType::Type::WCHAR_T : ValueType::Type::CHAR, 0U));
5487             } else if (tok->tokType() == Token::eString) {
5488                 ValueType valuetype(ValueType::Sign::UNKNOWN_SIGN, ValueType::Type::CHAR, 1U, 1U);
5489                 if (tok->isLong()) {
5490                     valuetype.originalTypeName = "wchar_t";
5491                     valuetype.type = ValueType::Type::WCHAR_T;
5492                 }
5493                 setValueType(tok, valuetype);
5494             } else if (tok->str() == "(") {
5495                 // cast
5496                 if (tok->isCast() && !tok->astOperand2() && Token::Match(tok, "( %name%")) {
5497                     ValueType valuetype;
5498                     if (Token::simpleMatch(parsedecl(tok->next(), &valuetype, mDefaultSignedness, mSettings), ")"))
5499                         setValueType(tok, valuetype);
5500                 }
5501     
5502                 // C++ cast
5503                 else if (tok->astOperand2() && Token::Match(tok->astOperand1(), "static_cast|const_cast|dynamic_cast|reinterpret_cast < %name%") && tok->astOperand1()->linkAt(1)) {
5504                     ValueType valuetype;
5505                     if (Token::simpleMatch(parsedecl(tok->astOperand1()->tokAt(2), &valuetype, mDefaultSignedness, mSettings), ">"))
5506                         setValueType(tok, valuetype);
5507                 }
5508     
5509                 // function
5510                 else if (tok->previous() && tok->previous()->function() && tok->previous()->function()->retDef) {
5511                     ValueType valuetype;
5512                     if (parsedecl(tok->previous()->function()->retDef, &valuetype, mDefaultSignedness, mSettings))
5513                         setValueType(tok, valuetype);
5514                 }
5515     
5516                 else if (Token::simpleMatch(tok->previous(), "sizeof (")) {
5517                     // TODO: use specified size_t type
5518                     ValueType valuetype(ValueType::Sign::UNSIGNED, ValueType::Type::LONG, 0U);
5519                     valuetype.originalTypeName = "size_t";
5520                     setValueType(tok, valuetype);

5525                             setValueType(tok->next(), vt);
5526                         }
5527                     }
5528                 }
5529     
5530                 // function style cast
5531                 else if (tok->previous() && tok->previous()->isStandardType()) {
5532                     ValueType valuetype;
5533                     valuetype.type = ValueType::typeFromString(tok->previous()->str(), tok->previous()->isLong());

5535                         valuetype.sign = ValueType::Sign::UNSIGNED;
5536                     else if (tok->previous()->isSigned())
5537                         valuetype.sign = ValueType::Sign::SIGNED;
5538                     setValueType(tok, valuetype);
5539                 }
5540     
5541                 // constructor
5542                 else if (tok->previous() && tok->previous()->type() && tok->previous()->type()->classScope) {
5543                     ValueType valuetype;
5544                     valuetype.type = ValueType::RECORD;
5545                     valuetype.typeScope = tok->previous()->type()->classScope;
5546                     setValueType(tok, valuetype);
5547                 }

5549                 // library function
5550                 else if (tok->previous()) {
5551                     const std::string& typestr(mSettings->library.returnValueType(tok->previous()));
5552                     if (!typestr.empty()) {
5553                         ValueType valuetype;
5554                         if (valuetype.fromLibraryType(typestr, mSettings)) {
5555                             setValueType(tok, valuetype);
5556                         }
5557                     }
5558     
5559                     if (typestr.empty() || typestr == "iterator") {
5560                         if (Token::simpleMatch(tok->astOperand1(), ".") &&
5561                             tok->astOperand1()->astOperand1() &&
5562                             tok->astOperand1()->astOperand2() &&
5563                             tok->astOperand1()->astOperand1()->valueType() &&

5569                                     it->second.yield == Library::Container::Yield::END_ITERATOR ||

5572                                     vt.type = ValueType::Type::ITERATOR;
5573                                     vt.container = cont;
5574                                     setValueType(tok, vt);
5575                                 }
5576                             }
5577                         }
5578                         continue;
5579                     }
5580                     TokenList tokenList(mSettings);
5581                     std::istringstream istr(typestr+";");
5582                     if (tokenList.createTokens(istr)) {
5583                         ValueType vt;
5584                         tokenList.simplifyPlatformTypes();
5585                         tokenList.simplifyStdType();
5586                         if (parsedecl(tokenList.front(), &vt, mDefaultSignedness, mSettings)) {
5587                             setValueType(tok, vt);
5588                         }
5589                     }
5590                 }
5591             } else if (tok->variable()) {
5592                 setValueType(tok, *tok->variable());
5593             } else if (tok->enumerator()) {
5594                 setValueType(tok, *tok->enumerator());
5595             } else if (mIsCpp && tok->str() == "new") {
5596                 const Token *typeTok = tok->next();
5597                 if (Token::Match(typeTok, "( std| ::| nothrow )"))
5598                     typeTok = typeTok->link()->next();
5599                 if (const Library::Container *c = mSettings->library.detectContainer(typeTok)) {
5600                     ValueType vt;
5601                     vt.pointer = 1;
5602                     vt.container = c;
5603                     vt.type = ValueType::Type::CONTAINER;
5604                     setValueType(tok, vt);
5605                     continue;
5606                 }
5607                 std::string typestr;
5608                 while (Token::Match(typeTok, "%name% :: %name%")) {
5609                     typestr += typeTok->str() + "::";
5610                     typeTok = typeTok->tokAt(2);
5611                 }
5612                 if (!Token::Match(typeTok, "%type% ;|[|("))
5613                     continue;
5614                 typestr += typeTok->str();
5615                 ValueType vt;
5616                 vt.pointer = 1;
5617                 if (typeTok->type() && typeTok->type()->classScope) {
5618                     vt.type = ValueType::Type::RECORD;
5619                     vt.typeScope = typeTok->type()->classScope;
5620                 } else {
5621                     vt.type = ValueType::typeFromString(typestr, typeTok->isLong());

5623                         vt.fromLibraryType(typestr, mSettings);
5624                     if (vt.type == ValueType::Type::UNKNOWN_TYPE)
5625                         continue;
5626                     if (typeTok->isUnsigned())
5627                         vt.sign = ValueType::Sign::UNSIGNED;
5628                     else if (typeTok->isSigned())

5631                         vt.sign = (vt.type == ValueType::Type::CHAR) ? mDefaultSignedness : ValueType::Sign::SIGNED;
5632                 }
5633                 setValueType(tok, vt);
5634             } else if (tok->str() == "return" && tok->scope()) {
5635                 const Function *function = tok->scope()->function;
5636                 if (function && function->retDef) {
5637                     ValueType vt;
5638                     parsedecl(function->retDef, &vt, mDefaultSignedness, mSettings);
5639                     setValueType(tok, vt);
5640                 }
5641             }
5642         }
5643     
5644         if (reportDebugWarnings && mSettings->debugwarnings) {
5645             for (Token *tok = tokens; tok; tok = tok->next()) {
5646                 if (tok->str() == "auto" && !tok->valueType())
5647                     debugMessage(tok, "auto token with no type.");
5648             }
5649         }
5650     
5651         // Update functions with new type information.
5652         createSymbolDatabaseSetFunctionPointers(false);
5653     
5654         // Update auto variables with new type information.
5655         createSymbolDatabaseSetVariablePointers();
5656     }

5688     bool ValueType::fromLibraryType(const std::string &typestr, const Settings *settings)
5689     {
5690         const Library::PodType* podtype = settings->library.podtype(typestr);
5691         if (podtype && (podtype->sign == 's' || podtype->sign == 'u')) {
5692             if (podtype->size == 1)
5693                 type = ValueType::Type::CHAR;
5694             else if (podtype->size == settings->sizeof_int)
5695                 type = ValueType::Type::INT;
5696             else if (podtype->size == settings->sizeof_short)
5697                 type = ValueType::Type::SHORT;
5698             else if (podtype->size == settings->sizeof_long)
5699                 type = ValueType::Type::LONG;
5700             else if (podtype->size == settings->sizeof_long_long)
5701                 type = ValueType::Type::LONGLONG;
5702             else if (podtype->stdtype == Library::PodType::BOOL)
5703                 type = ValueType::Type::BOOL;
5704             else if (podtype->stdtype == Library::PodType::CHAR)
5705                 type = ValueType::Type::CHAR;
5706             else if (podtype->stdtype == Library::PodType::SHORT)
5707                 type = ValueType::Type::SHORT;
5708             else if (podtype->stdtype == Library::PodType::INT)
5709                 type = ValueType::Type::INT;
5710             else if (podtype->stdtype == Library::PodType::LONG)
5711                 type = ValueType::Type::LONG;
5712             else if (podtype->stdtype == Library::PodType::LONGLONG)
5713                 type = ValueType::Type::LONGLONG;
5714             else
5715                 type = ValueType::Type::UNKNOWN_INT;
5716             sign = (podtype->sign == 'u') ? ValueType::UNSIGNED : ValueType::SIGNED;
5717             return true;
5718         }
5719     
5720         const Library::PlatformType *platformType = settings->library.platform_type(typestr, settings->platformString());
5721         if (platformType) {
5722             if (platformType->mType == "char")
5723                 type = ValueType::Type::CHAR;
5724             else if (platformType->mType == "short")
5725                 type = ValueType::Type::SHORT;
5726             else if (platformType->mType == "wchar_t")
5727                 type = ValueType::Type::WCHAR_T;
5728             else if (platformType->mType == "int")
5729                 type = platformType->_long ? ValueType::Type::LONG : ValueType::Type::INT;
5730             else if (platformType->mType == "long")
5731                 type = platformType->_long ? ValueType::Type::LONGLONG : ValueType::Type::LONG;
5732             if (platformType->_signed)
5733                 sign = ValueType::SIGNED;
5734             else if (platformType->_unsigned)
5735                 sign = ValueType::UNSIGNED;
5736             if (platformType->_pointer)
5737                 pointer = 1;
5738             if (platformType->_ptr_ptr)
5739                 pointer = 2;
5740             if (platformType->_const_ptr)
5741                 constness = 1;
5742             return true;
5743         } else if (!podtype && (typestr == "size_t" || typestr == "std::size_t")) {
5744             originalTypeName = "size_t";
5745             sign = ValueType::UNSIGNED;
5746             if (settings->sizeof_size_t == settings->sizeof_long)
5747                 type = ValueType::Type::LONG;
5748             else if (settings->sizeof_size_t == settings->sizeof_long_long)
5749                 type = ValueType::Type::LONGLONG;
5750             else if (settings->sizeof_size_t == settings->sizeof_int)
5751                 type = ValueType::Type::INT;
5752             else
5753                 type = ValueType::Type::UNKNOWN_INT;
5754             return true;
5755         }
5756     
5757         return false;
5758     }

5932     ValueType::MatchResult ValueType::matchParameter(const ValueType *call, const ValueType *func)
5933     {
5934         if (!call || !func)
5935             return ValueType::MatchResult::UNKNOWN;
5936         if (call->pointer != func->pointer) {
5937             if (call->pointer > 1 && func->pointer == 1 && func->type == ValueType::Type::VOID)
5938                 return ValueType::MatchResult::FALLBACK1;
5939             if (call->pointer == 1 && call->type == ValueType::Type::CHAR && func->pointer == 0 && func->container && func->container->stdStringLike)
5940                 return ValueType::MatchResult::FALLBACK2;
5941             return ValueType::MatchResult::NOMATCH; // TODO
5942         }
5943         if (call->pointer > 0 && ((call->constness | func->constness) != func->constness))
5944             return ValueType::MatchResult::NOMATCH;
5945         if (call->type != func->type) {
5946             if (call->type == ValueType::Type::VOID || func->type == ValueType::Type::VOID)
5947                 return ValueType::MatchResult::FALLBACK1;
5948             if (call->pointer > 0 && func->pointer > 0)
5949                 return func->type == ValueType::UNKNOWN_TYPE ? ValueType::MatchResult::UNKNOWN : ValueType::MatchResult::NOMATCH;
5950             if (call->isIntegral() && func->isIntegral())
5951                 return call->type < func->type ?
5952                        ValueType::MatchResult::FALLBACK1 :
5953                        ValueType::MatchResult::FALLBACK2;
5954             else if (call->isFloat() && func->isFloat())
5955                 return ValueType::MatchResult::FALLBACK1;
5956             else if (call->isIntegral() && func->isFloat())
5957                 return ValueType::MatchResult::FALLBACK2;
5958             else if (call->isFloat() && func->isIntegral())
5959                 return ValueType::MatchResult::FALLBACK2;
5960             return ValueType::MatchResult::UNKNOWN; // TODO
5961         }
5962     
5963         if (call->typeScope != nullptr || func->typeScope != nullptr)
5964             return call->typeScope == func->typeScope ? ValueType::MatchResult::SAME : ValueType::MatchResult::NOMATCH;
5965     
5966         if (call->container != nullptr || func->container != nullptr) {
5967             if (call->container != func->container)
5968                 return ValueType::MatchResult::NOMATCH;
