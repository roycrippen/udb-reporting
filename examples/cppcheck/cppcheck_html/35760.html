79         std::string getRange(const char code[], const std::string &str, int linenr = 0) {
80             Settings settings;
81             settings.platform(cppcheck::Platform::Unix64);
82             settings.library.smartPointers.insert("std::shared_ptr");
83             Tokenizer tokenizer(&settings, this);
84             std::istringstream istr(code);
85             tokenizer.tokenize(istr, "test.cpp");
86             std::string ret;
87             std::function<void(const Token *, const ExprEngine::Value &)> f = [&](const Token *tok, const ExprEngine::Value &value) {
88                 if ((linenr == 0 || linenr == tok->linenr()) && tok->expressionString() == str) {
89                     if (!ret.empty())
90                         ret += ",";
91                     ret += value.getRange();
92                 }
93             };
94             std::vector<ExprEngine::Callback> callbacks;
95             callbacks.push_back(f);
96             std::ostringstream dummy;
97             ExprEngine::executeAllFunctions(&tokenizer, &settings, callbacks, dummy);
98             return ret;
99         }

101         std::string trackExecution(const char code[]) {
102             Settings settings;
103             settings.debugVerification = true;
104             settings.platform(cppcheck::Platform::Unix64);
105             settings.library.smartPointers.insert("std::shared_ptr");
106             Tokenizer tokenizer(&settings, this);
107             std::istringstream istr(code);
108             tokenizer.tokenize(istr, "test.cpp");
109             std::vector<ExprEngine::Callback> dummy;
110             std::ostringstream ret;
111             ExprEngine::executeAllFunctions(&tokenizer, &settings, dummy, ret);
112             return ret.str();
113         }
