251         void preprocess(const char* code, std::map<std::string, std::string>& actual, const char filename[] = "file.c") {
252             errout.str("");
253             std::istringstream istr(code);
254             simplecpp::OutputList outputList;
255             std::vector<std::string> files;
256             simplecpp::TokenList tokens(istr, files, filename, &outputList);
257             tokens.removeComments();
258             preprocessor0.simplifyPragmaAsm(&tokens);
259             const std::set<std::string> configs(preprocessor0.getConfigs(tokens));
260             preprocessor0.setDirectives(tokens);
261             for (const std::string & config : configs) {
262                 try {
263                     const std::string &cfgcode = preprocessor0.getcode(tokens, config, files, std::string(code).find("#file") != std::string::npos);
264                     actual[config] = cfgcode;
265                 } catch (const simplecpp::Output &) {
266                     actual[config] = "";
267                 } catch (...) {
268                 }
269             }
270         }

272         std::string getConfigsStr(const char filedata[], const char *arg = nullptr) {
273             Settings settings;
274             if (arg && std::strncmp(arg,"-D",2)==0)
275                 settings.userDefines = arg + 2;
276             if (arg && std::strncmp(arg,"-U",2)==0)
277                 settings.userUndefs.insert(arg+2);
278             Preprocessor preprocessor(settings, this);
279             std::vector<std::string> files;
280             std::istringstream istr(filedata);
281             simplecpp::TokenList tokens(istr,files);
282             tokens.removeComments();
283             const std::set<std::string> configs = preprocessor.getConfigs(tokens);
284             std::string ret;
285             for (const std::string & config : configs)
286                 ret += config + '\n';
287             return ret;
288         }
