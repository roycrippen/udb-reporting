50     static bool isPtrArg(const Token *tok)
51     {
52         const Variable *var = tok->variable();
53         return (var && var->isArgument() && var->isPointer());
54     }

56     static bool isArrayArg(const Token *tok)
57     {
58         const Variable *var = tok->variable();
59         return (var && var->isArgument() && var->isArray());
60     }

63     {
64         const Variable *var = tok->variable();
65         return (var && var->isArray());
66     }

69     {
70         const Variable *var = tok->variable();
71         return (var && var->isArgument() && var->isReference() && var->isPointer());

74     static bool isNonReferenceArg(const Token *tok)
75     {
76         const Variable *var = tok->variable();
77         return (var && var->isArgument() && !var->isReference() && (var->isPointer() || var->valueType()->type >= ValueType::Type::CONTAINER || var->type()));

80     static bool isAutoVar(const Token *tok)
81     {
82         const Variable *var = tok->variable();
83     
84         if (!var || !var->isLocal() || var->isStatic())
85             return false;
86     
87         if (var->isReference()) {
88             // address of reference variable can be taken if the address
89             // of the variable it points at is not a auto-var
90             // TODO: check what the reference variable references.
91             return false;
92         }
93     
94         if (Token::Match(tok, "%name% .|::")) {
95             do {
96                 tok = tok->tokAt(2);
97             } while (Token::Match(tok, "%name% .|::"));
98             if (Token::Match(tok, "%name% ("))
99                 return false;
100         }
101         return true;
102     }

104     static bool isAutoVarArray(const Token *tok)
105     {
106         if (!tok)
107             return false;
108     
109         // &x[..]
110         if (tok->isUnaryOp("&") && Token::simpleMatch(tok->astOperand1(), "["))
111             return isAutoVarArray(tok->astOperand1()->astOperand1());
112     
113         // x+y
114         if (tok->str() == "+")
115             return isAutoVarArray(tok->astOperand1()) || isAutoVarArray(tok->astOperand2());
116     
117         // x-intexpr
118         if (tok->str() == "-")
119             return isAutoVarArray(tok->astOperand1()) &&
120                    tok->astOperand2() &&
121                    tok->astOperand2()->valueType() &&
122                    tok->astOperand2()->valueType()->isIntegral();
123     
124         const Variable *var = tok->variable();
125         if (!var)
126             return false;
127     
128         // Variable
129         if (var->isLocal() && !var->isStatic() && var->isArray() && !var->isPointer())
130             return true;
131     
132         // ValueFlow
133         if (var->isPointer() && !var->isArgument()) {
134             for (std::list<ValueFlow::Value>::const_iterator it = tok->values().begin(); it != tok->values().end(); ++it) {
135                 const ValueFlow::Value &val = *it;
136                 if (val.isTokValue() && isAutoVarArray(val.tokvalue))
137                     return true;
138             }
139         }
140     
141         return false;
142     }

144     // Verification that we really take the address of a local variable
145     static bool checkRvalueExpression(const Token * const vartok)
146     {
147         const Variable * const var = vartok->variable();
148         if (var == nullptr)
149             return false;
150     
151         if (Token::Match(vartok->previous(), "& %name% [") && var->isPointer())
152             return false;
153     
154         const Token * const next = vartok->next();
155         // &a.b[0]
156         if (Token::Match(vartok, "%name% . %var% [") && !var->isPointer()) {
157             const Variable *var2 = next->next()->variable();
158             return var2 && !var2->isPointer();
159         }
160     
161         return ((next->str() != "." || (!var->isPointer() && (!var->isClass() || var->type()))) && next->strAt(2) != ".");
162     }

164     static bool isAddressOfLocalVariable(const Token *expr)
165     {
166         if (!expr)
167             return false;
168         if (Token::Match(expr, "+|-"))
169             return isAddressOfLocalVariable(expr->astOperand1()) || isAddressOfLocalVariable(expr->astOperand2());
170         if (expr->isCast())
171             return isAddressOfLocalVariable(expr->astOperand2() ? expr->astOperand2() : expr->astOperand1());
172         if (expr->isUnaryOp("&")) {
173             const Token *op = expr->astOperand1();
174             bool deref = false;
175             while (Token::Match(op, ".|[")) {
176                 if (op->originalName() == "->")
177                     return false;
178                 if (op->str() == "[")
179                     deref = true;
180                 op = op->astOperand1();
181             }
182             return op && isAutoVar(op) && (!deref || !op->variable()->isPointer());
183         }
184         return false;
185     }

187     static bool variableIsUsedInScope(const Token* start, nonneg int varId, const Scope *scope)
188     {
189         if (!start) // Ticket #5024
190             return false;
191     
192         for (const Token *tok = start; tok && tok != scope->bodyEnd; tok = tok->next()) {
193             if (tok->varId() == varId)
194                 return true;
195             const Scope::ScopeType scopeType = tok->scope()->type;
196             if (scopeType == Scope::eFor || scopeType == Scope::eDo || scopeType == Scope::eWhile) // In case of loops, better checking would be necessary
197                 return true;
198             if (Token::simpleMatch(tok, "asm ("))
199                 return true;
200         }
201         return false;
202     }

204     void CheckAutoVariables::assignFunctionArg()
205     {
206         const bool printStyle = mSettings->isEnabled(Settings::STYLE);
207         const bool printWarning = mSettings->isEnabled(Settings::WARNING);
208         if (!printStyle && !printWarning)
209             return;
210     
211         const SymbolDatabase *symbolDatabase = mTokenizer->getSymbolDatabase();
212         for (const Scope * scope : symbolDatabase->functionScopes) {
213             for (const Token *tok = scope->bodyStart; tok && tok != scope->bodyEnd; tok = tok->next()) {
214                 // TODO: What happens if this is removed?
215                 if (tok->astParent())
216                     continue;
217                 if (!(tok->isAssignmentOp() || Token::Match(tok, "++|--")) || !Token::Match(tok->astOperand1(), "%var%"))
218                     continue;
219                 const Token* const vartok = tok->astOperand1();
220                 if (isNonReferenceArg(vartok) &&
221                     !Token::Match(vartok->next(), "= %varid% ;", vartok->varId()) &&
222                     !variableIsUsedInScope(Token::findsimplematch(vartok->next(), ";"), vartok->varId(), scope) &&
223                     !Token::findsimplematch(vartok, "goto", scope->bodyEnd)) {
224                     if (vartok->variable()->isPointer() && printWarning)
225                         errorUselessAssignmentPtrArg(vartok);
226                     else if (printStyle)
227                         errorUselessAssignmentArg(vartok);
228                 }
229             }
230         }
231     }

233     void CheckAutoVariables::autoVariables()
234     {
235         const bool printInconclusive = mSettings->inconclusive;
236         const SymbolDatabase *symbolDatabase = mTokenizer->getSymbolDatabase();
237         for (const Scope * scope : symbolDatabase->functionScopes) {
238             for (const Token *tok = scope->bodyStart; tok && tok != scope->bodyEnd; tok = tok->next()) {
239                 // Skip lambda..
240                 if (const Token *lambdaEndToken = findLambdaEndToken(tok)) {
241                     tok = lambdaEndToken;
242                     continue;
243                 }
244                 // Critical assignment
245                 if (Token::Match(tok, "[;{}] %var% = & %var%") && isRefPtrArg(tok->next()) && isAutoVar(tok->tokAt(4))) {
246                     if (checkRvalueExpression(tok->tokAt(4)))
247                         errorAutoVariableAssignment(tok->next(), false);
248                 } else if (Token::Match(tok, "[;{}] * %var% =") && isPtrArg(tok->tokAt(2)) && isAddressOfLocalVariable(tok->tokAt(3)->astOperand2())) {
249                     errorAutoVariableAssignment(tok->next(), false);
250                 } else if (Token::Match(tok, "[;{}] %var% . %var% =") && isPtrArg(tok->next()) && isAddressOfLocalVariable(tok->tokAt(4)->astOperand2())) {
251                     errorAutoVariableAssignment(tok->next(), false);
252                 } else if (Token::Match(tok, "[;{}] %var% . %var% = %var% ;")) {
253                     // TODO: check if the parameter is only changed temporarily (#2969)
254                     if (printInconclusive && isPtrArg(tok->next())) {
255                         if (isAutoVarArray(tok->tokAt(5)))
256                             errorAutoVariableAssignment(tok->next(), true);
257                     }
258                     tok = tok->tokAt(5);
259                 } else if (Token::Match(tok, "[;{}] * %var% = %var% ;")) {
260                     const Variable * var1 = tok->tokAt(2)->variable();
261                     if (var1 && var1->isArgument() && Token::Match(var1->nameToken()->tokAt(-3), "%type% * *")) {
262                         if (isAutoVarArray(tok->tokAt(4)))
263                             errorAutoVariableAssignment(tok->next(), false);
264                     }
265                     tok = tok->tokAt(4);
266                 } else if (Token::Match(tok, "[;{}] %var% [") && Token::simpleMatch(tok->linkAt(2), "] =") &&
267                            (isPtrArg(tok->next()) || isArrayArg(tok->next())) && isAddressOfLocalVariable(tok->linkAt(2)->next()->astOperand2())) {
268                     errorAutoVariableAssignment(tok->next(), false);
269                 }
270                 // Invalid pointer deallocation
271                 else if ((Token::Match(tok, "%name% ( %var% ) ;") && mSettings->library.getDeallocFuncInfo(tok)) ||
272                          (mTokenizer->isCPP() && Token::Match(tok, "delete [| ]| (| %var% !!["))) {
273                     tok = Token::findmatch(tok->next(), "%var%");
274                     if (isArrayVar(tok))
275                         errorInvalidDeallocation(tok, nullptr);
276                     else if (tok && tok->variable() && tok->variable()->isPointer()) {
277                         for (const ValueFlow::Value &v : tok->values()) {
278                             if (v.isTokValue() && isArrayVar(v.tokvalue)) {
279                                 errorInvalidDeallocation(tok, &v);
280                                 break;
281                             }
282                         }
283                     }
284                 } else if ((Token::Match(tok, "%name% ( & %var% ) ;") && mSettings->library.getDeallocFuncInfo(tok)) ||
285                            (mTokenizer->isCPP() && Token::Match(tok, "delete [| ]| (| & %var% !!["))) {
286                     tok = Token::findmatch(tok->next(), "%var%");
287                     if (isAutoVar(tok))
288                         errorInvalidDeallocation(tok, nullptr);
289                 }
290             }
291         }
292     }

311     void CheckAutoVariables::errorAutoVariableAssignment(const Token *tok, bool inconclusive)
312     {
313         if (!inconclusive) {
314             reportError(tok, Severity::error, "autoVariables",
315                         "Address of local auto-variable assigned to a function parameter.\n"
316                         "Dangerous assignment - the function parameter is assigned the address of a local "
317                         "auto-variable. Local auto-variables are reserved from the stack which "
318                         "is freed when the function ends. So the pointer to a local variable "
319                         "is invalid after the function ends.", CWE562, false);
320         } else {
321             reportError(tok, Severity::error, "autoVariables",
322                         "Address of local auto-variable assigned to a function parameter.\n"
323                         "Function parameter is assigned the address of a local auto-variable. "
324                         "Local auto-variables are reserved from the stack which is freed when "
325                         "the function ends. The address is invalid after the function ends and it "
326                         "might 'leak' from the function through the parameter.",
327                         CWE562,
328                         true);
329         }
330     }

360     // return temporary?
361     bool CheckAutoVariables::returnTemporary(const Token *tok)
362     {
363         bool func = false;     // Might it be a function call?
364         bool retref = false;   // is there such a function that returns a reference?
365         bool retvalue = false; // is there such a function that returns a value?
366     
367         const Function *function = tok->function();
368         if (function) {
369             // Ticket #5478: Only functions or operator equal might return a temporary
370             if (function->type != Function::eOperatorEqual && function->type != Function::eFunction)
371                 return false;
372             retref = function->tokenDef->strAt(-1) == "&";
373             if (!retref) {
374                 const Token *start = function->retDef;
375                 if (start->str() == "const")
376                     start = start->next();
377                 if (start->str() == "::")
378                     start = start->next();
379     
380                 if (Token::simpleMatch(start, "std ::")) {
381                     if (start->strAt(3) != "<" || !Token::simpleMatch(start->linkAt(3), "> ::"))
382                         retvalue = true;
383                     else
384                         retref = true; // Assume that a reference is returned
385                 } else {
386                     if (start->type())
387                         retvalue = true;
388                     else
389                         retref = true;
390                 }
391             }
392             func = true;
393         }
394         if (!func && tok->type())
395             return true;
396     
397         return (!retref && retvalue);
398     }

402     static bool astHasAutoResult(const Token *tok)
403     {
404         if (tok->astOperand1() && !astHasAutoResult(tok->astOperand1()))
405             return false;
406         if (tok->astOperand2() && !astHasAutoResult(tok->astOperand2()))
407             return false;
408     
409         if (tok->isOp()) {
410             if (tok->tokType() == Token::eIncDecOp)
411                 return false;
412             if ((tok->str() == "<<" || tok->str() == ">>") && tok->astOperand1()) {
413                 const Token* tok2 = tok->astOperand1();
414                 while (tok2 && tok2->isUnaryOp("*"))
415                     tok2 = tok2->astOperand1();
416                 return tok2 && tok2->variable() && !tok2->variable()->isClass() && !tok2->variable()->isStlType(); // Class or unknown type on LHS: Assume it is a stream
417             }
418             return true;
419         }
420     
421         if (tok->isLiteral())
422             return true;
423     
424         if (tok->isName()) {
425             // TODO: check function calls, struct members, arrays, etc also
426             if (!tok->variable())
427                 return false;
428             if (tok->variable()->isStlType())
429                 return true;
430             if (tok->variable()->isClass() || tok->variable()->isPointer() || tok->variable()->isReference()) // TODO: Properly handle pointers/references to classes in symbol database
431                 return false;
432     
433             return true;
434         }
435     
436         return false;
437     }

439     void CheckAutoVariables::returnReference()
440     {
441         if (mTokenizer->isC())
442             return;
443     
444         const SymbolDatabase *symbolDatabase = mTokenizer->getSymbolDatabase();
445     
446         for (const Scope * scope : symbolDatabase->functionScopes) {
447             if (!scope->function)
448                 continue;
449     
450             const Token *tok = scope->function->tokenDef;
451     
452             // have we reached a function that returns a reference?
453             if (tok->previous() && tok->previous()->str() == "&") {
454                 for (const Token *tok2 = scope->bodyStart->next(); tok2 && tok2 != scope->bodyEnd; tok2 = tok2->next()) {
455                     if (!tok2->scope()->isExecutable()) {
456                         tok2 = tok2->scope()->bodyEnd;
457                         if (!tok2)
458                             break;
459                         continue;
460                     }
461     
462                     // Skip over lambdas
463                     const Token *lambdaEndToken = findLambdaEndToken(tok2);
464                     if (lambdaEndToken)
465                         tok2 = lambdaEndToken->next();
466     
467                     if (tok2->str() == "(")
468                         tok2 = tok2->link();
469     
470                     if (tok2->str() != "return")
471                         continue;
472     
473                     // return reference to temporary..
474                     if (Token::Match(tok2, "return %name% (") && Token::simpleMatch(tok2->linkAt(2), ") ;")) {
475                         if (returnTemporary(tok2->next())) {
476                             // report error..
477                             errorReturnTempReference(tok2);
478                         }
479                     }
480     
481                     // Return reference to a literal or the result of a calculation
482                     else if (tok2->astOperand1() && (tok2->astOperand1()->isCalculation() || tok2->next()->isLiteral()) &&
483                              astHasAutoResult(tok2->astOperand1())) {
484                         errorReturnTempReference(tok2);
485                     }
486                 }
487             }
488         }
489     }

491     static bool isInScope(const Token * tok, const Scope * scope)
492     {
493         if (!tok)
494             return false;
495         if (!scope)
496             return false;
497         const Variable * var = tok->variable();
498         if (var && (var->isGlobal() || var->isStatic() || var->isExtern()))
499             return false;
500         if (tok->scope() && tok->scope()->isNestedIn(scope))
501             return true;
502         if (!var)
503             return false;
504         if (var->isArgument() && !var->isReference()) {
505             const Scope * tokScope = tok->scope();
506             if (!tokScope)
507                 return false;
508             for (const Scope * argScope:tokScope->nestedList) {
509                 if (argScope && argScope->isNestedIn(scope))
510                     return true;
511             }
512         }
513         return false;
514     }

516     static bool isDeadScope(const Token * tok, const Scope * scope)
517     {
518         if (!tok)
519             return false;
520         if (!scope)
521             return false;
522         const Variable * var = tok->variable();
523         if (var && (!var->isLocal() || var->isStatic() || var->isExtern()))
524             return false;
525         if (tok->scope() && tok->scope()->bodyEnd != scope->bodyEnd && precedes(tok->scope()->bodyEnd, scope->bodyEnd))
526             return true;
527         return false;
528     }

530     static int getPointerDepth(const Token *tok)
531     {
532         if (!tok)
533             return 0;
534         return tok->valueType() ? tok->valueType()->pointer : 0;

537     void CheckAutoVariables::checkVarLifetimeScope(const Token * start, const Token * end)
538     {
539         if (!start)
540             return;
541         const Scope * scope = start->scope();
542         if (!scope)
543             return;
544         // If the scope is not set correctly then skip checking it
545         if (scope->bodyStart != start)
546             return;
547         bool returnRef = Function::returnsReference(scope->function);
548         for (const Token *tok = start; tok && tok != end; tok = tok->next()) {
549             // Return reference from function
550             if (returnRef && Token::simpleMatch(tok->astParent(), "return")) {
551                 for (const LifetimeToken& lt : getLifetimeTokens(tok)) {
552                     const Variable* var = lt.token->variable();
553                     if (!var)
554                         continue;
555                     if (!var->isGlobal() && !var->isStatic() && !var->isReference() && !var->isRValueReference() &&
556                         isInScope(var->nameToken(), tok->scope())) {
557                         errorReturnReference(tok, lt.errorPath, lt.inconclusive);
558                         break;
559                     }
560                 }
561                 // Assign reference to non-local variable
562             } else if (Token::Match(tok->previous(), "&|&& %var% =") && tok->astParent() == tok->next() &&
563                        tok->variable() && tok->variable()->nameToken() == tok &&
564                        tok->variable()->declarationId() == tok->varId() && tok->variable()->isStatic() &&
565                        !tok->variable()->isArgument()) {
566                 ErrorPath errorPath;
567                 const Variable *var = getLifetimeVariable(tok, errorPath);
568                 if (var && isInScope(var->nameToken(), tok->scope())) {
569                     errorDanglingReference(tok, var, errorPath);
570                     continue;
571                 }
572             }
573             for (const ValueFlow::Value& val:tok->values()) {
574                 if (!val.isLocalLifetimeValue())
575                     continue;
576                 if (!val.tokvalue->variable())
577                     continue;
578                 if (Token::Match(tok->astParent(), "return|throw")) {
579                     if (getPointerDepth(tok) < getPointerDepth(val.tokvalue))
580                         continue;
581                     if (!isLifetimeBorrowed(tok, mSettings))
582                         continue;
583                     if (isInScope(val.tokvalue->variable()->nameToken(), scope)) {
584                         errorReturnDanglingLifetime(tok, &val);
585                         break;
586                     }
587                 } else if (isDeadScope(val.tokvalue->variable()->nameToken(), tok->scope())) {
588                     errorInvalidLifetime(tok, &val);
589                     break;
590                 } else if (isInScope(val.tokvalue->variable()->nameToken(), tok->scope())) {
591                     const Variable * var = nullptr;
592                     const Token * tok2 = tok;
593                     if (Token::simpleMatch(tok->astParent(), "=")) {
594                         if (tok->astParent()->astOperand2() == tok) {
595                             var = getLHSVariable(tok->astParent());
596                             tok2 = tok->astParent()->astOperand1();
597                         }
598                     } else if (tok->variable() && tok->variable()->declarationId() == tok->varId()) {
599                         var = tok->variable();
600                     }
601                     if (!isLifetimeBorrowed(tok, mSettings))
602                         continue;
603                     if (var && !var->isLocal() && !var->isArgument() && !isVariableChanged(tok->next(), tok->scope()->bodyEnd, var->declarationId(), var->isGlobal(), mSettings, mTokenizer->isCPP())) {
604                         errorDanglngLifetime(tok2, &val);
605                         break;
606                     }
607                 }
608             }
609             const Token *lambdaEndToken = findLambdaEndToken(tok);
610             if (lambdaEndToken) {
611                 checkVarLifetimeScope(lambdaEndToken->link(), lambdaEndToken);
612                 tok = lambdaEndToken;
613             }
614             if (tok->str() == "{" && tok->scope()) {
615                 // Check functions in local classes
616                 if (tok->scope()->type == Scope::eClass ||
617                     tok->scope()->type == Scope::eStruct ||
618                     tok->scope()->type == Scope::eUnion) {
619                     for (const Function& f:tok->scope()->functionList) {
620                         if (f.functionScope)
621                             checkVarLifetimeScope(f.functionScope->bodyStart, f.functionScope->bodyEnd);
622                     }
623                     tok = tok->link();
624                 }
625             }
626         }

629     void CheckAutoVariables::checkVarLifetime()
630     {
631         const SymbolDatabase *symbolDatabase = mTokenizer->getSymbolDatabase();
632         for (const Scope * scope : symbolDatabase->functionScopes) {
633             if (!scope->function)
634                 continue;
635             checkVarLifetimeScope(scope->bodyStart, scope->bodyEnd);
636         }
637     }

688     void CheckAutoVariables::errorInvalidDeallocation(const Token *tok, const ValueFlow::Value *val)
689     {
690         const Variable *var = val ? val->tokvalue->variable() : (tok ? tok->variable() : nullptr);
691     
692         std::string type = "auto-variable";
693         if (var) {
694             if (var->isGlobal())
695                 type = "global variable";
696             else if (var->isStatic())
697                 type = "static variable";
698         }
699     
700         if (val)
701             type += " (" + val->tokvalue->str() + ")";
702     
703         reportError(getErrorPath(tok, val, "Deallocating memory that was not dynamically allocated"),
704                     Severity::error,
705                     "autovarInvalidDeallocation",
706                     "Deallocation of an " + type + " results in undefined behaviour.\n"
707                     "The deallocation of an " + type + " results in undefined behaviour. You should only free memory "
708                     "that has been allocated dynamically.", CWE590, false);
709     }
