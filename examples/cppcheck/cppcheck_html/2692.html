46     void StatsDialog::setProject(const ProjectFile* projectFile)
47     {
48         if (projectFile) {
49             mUI.mProject->setText(projectFile->getRootPath());
50             mUI.mPaths->setText(projectFile->getCheckPaths().join(";"));
51             mUI.mIncludePaths->setText(projectFile->getIncludeDirs().join(";"));
52             mUI.mDefines->setText(projectFile->getDefines().join(";"));
53             mUI.mUndefines->setText(projectFile->getUndefines().join(";"));
54     #ifndef HAVE_QCHART
55             mUI.mTabHistory->setVisible(false);
56     #else
57             QString statsFile;
58             if (!projectFile->getBuildDir().isEmpty()) {
59                 const QString prjpath = QFileInfo(projectFile->getFilename()).absolutePath();
60                 const QString buildDir = prjpath + '/' + projectFile->getBuildDir();
61                 if (QDir(buildDir).exists()) {
62                     statsFile = buildDir + "/statistics.txt";
63                 }
64             }
65             mUI.mLblHistoryFile->setText(tr("File: ") + (statsFile.isEmpty() ? tr("No cppcheck build dir") : statsFile));
66             if (!statsFile.isEmpty()) {
67                 QChartView *chartView;
68                 chartView = createChart(statsFile, "cppcheck");
69                 mUI.mTabHistory->layout()->addWidget(chartView);
70                 if (projectFile->getClangAnalyzer()) {
71                     chartView = createChart(statsFile, CLANG_ANALYZER);
72                     mUI.mTabHistory->layout()->addWidget(chartView);
73                 }
74                 if (projectFile->getClangTidy()) {
75                     chartView = createChart(statsFile, CLANG_TIDY);
76                     mUI.mTabHistory->layout()->addWidget(chartView);
77                 }
78             }
79     #endif
80         } else {
81             mUI.mProject->setText(QString());
82             mUI.mPaths->setText(QString());
83             mUI.mIncludePaths->setText(QString());
84             mUI.mDefines->setText(QString());
85             mUI.mUndefines->setText(QString());
86         }
87     }

99     void StatsDialog::setScanDuration(double seconds)
100     {
101         // Factor the duration into units (days/hours/minutes/seconds)
102         int secs = seconds;
103         int days = secs / (24 * 60 * 60);
104         secs -= days * (24 * 60 * 60);
105         int hours = secs / (60 * 60);
106         secs -= hours * (60 * 60);
107         int mins = secs / 60;
108         secs -= mins * 60;
109     
110         // Concatenate the two most significant units (e.g. "1 day and 3 hours")
111         QStringList parts;
112         if (days)
113             parts << ((days == 1) ? tr("1 day") : tr("%1 days").arg(days));
114         if (hours)
115             parts << ((hours == 1) ? tr("1 hour") : tr("%1 hours").arg(hours));
116         if (mins && parts.size() < 2)
117             parts << ((mins == 1) ? tr("1 minute") : tr("%1 minutes").arg(mins));
118         if (secs && parts.size() < 2)
119             parts << ((secs == 1) ? tr("1 second") : tr("%1 seconds").arg(secs));
120     
121         // For durations < 1s, show the fraction of a second (e.g. "0.7 seconds")
122         if (parts.isEmpty())
123             parts << tr("0.%1 seconds").arg(int(10.0 *(seconds - secs)));
124     
125         mUI.mScanDuration->setText(parts.join(tr(" and ")));
126     }

334     #ifdef HAVE_QCHART
335     QChartView *StatsDialog::createChart(const QString &statsFile, const QString &tool)
336     {
337         QChart *chart = new QChart;
338         chart->addSeries(numberOfReports(statsFile, tool + "-error"));
339         chart->addSeries(numberOfReports(statsFile, tool + "-warning"));
340         chart->addSeries(numberOfReports(statsFile, tool + "-style"));
341         chart->addSeries(numberOfReports(statsFile, tool + "-performance"));
342         chart->addSeries(numberOfReports(statsFile, tool + "-portability"));
343     
344         QDateTimeAxis *axisX = new QDateTimeAxis;
345         axisX->setTitleText("Date");
346         chart->addAxis(axisX, Qt::AlignBottom);
347     
348         foreach (QAbstractSeries *s, chart->series()) {
349             s->attachAxis(axisX);
350         }
351     
352         QValueAxis *axisY = new QValueAxis;
353         axisY->setLabelFormat("%i");
354         axisY->setTitleText("Count");
355         chart->addAxis(axisY, Qt::AlignLeft);
356     
357         qreal maxY = 0;
358         foreach (QAbstractSeries *s, chart->series()) {
359             s->attachAxis(axisY);
360             if (QLineSeries *ls = dynamic_cast<QLineSeries*>(s)) {
361                 foreach (QPointF p, ls->points()) {
362                     if (p.y() > maxY)
363                         maxY = p.y();
364                 }
365             }
366         }
367         axisY->setMax(maxY);
368     
369         //chart->createDefaultAxes();
370         chart->setTitle(tool);
371     
372         QChartView *chartView = new QChartView(chart);
373         chartView->setRenderHint(QPainter::Antialiasing);
374         return chartView;
375     }

377     QLineSeries *StatsDialog::numberOfReports(const QString &fileName, const QString &severity) const
378     {
379         QLineSeries *series = new QLineSeries();
380         series->setName(severity);
381         QFile f(fileName);
382         if (f.open(QIODevice::ReadOnly | QIODevice::Text)) {
383             quint64 t = 0;
384             QTextStream in(&f);
385             while (!in.atEnd()) {
386                 QString line = in.readLine();
387                 QRegExp rxdate("\\[(\\d\\d)\\.(\\d\\d)\\.(\\d\\d\\d\\d)\\]");
388                 if (rxdate.exactMatch(line)) {
389                     int y = rxdate.cap(3).toInt();
390                     int m = rxdate.cap(2).toInt();
391                     int d = rxdate.cap(1).toInt();
392                     QDateTime dt;
393                     dt.setDate(QDate(y,m,d));
394                     if (t == dt.toMSecsSinceEpoch())
395                         t += 1000;
396                     else
397                         t = dt.toMSecsSinceEpoch();
398                 }
399                 if (line.startsWith(severity + ':')) {
400                     int y = line.mid(1+severity.length()).toInt();
401                     series->append(t, y);
402                 }
403             }
404         }
405         return series;
406     }
