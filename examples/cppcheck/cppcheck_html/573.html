210     struct PathAnalysis {
211         enum class Progress {
212             Continue,
213             Break
214         };
215         PathAnalysis(const Token* start, const Library& library)
216             : start(start), library(&library)
217         {}
218         const Token * start;
219         const Library * library;
220     
221         struct Info {
222             const Token* tok;
223             ErrorPath errorPath;
224             bool known;
225         };
226     
227         void forward(const std::function<Progress(const Info&)>& f) const;
228         template<class F>
229         void forwardAll(F f) {
230             forward([&](const Info& info) {
231                 f(info);
232                 return Progress::Continue;
233             });
234         }
235         template<class Predicate>
236         Info forwardFind(Predicate pred) {
237             Info result{};
238             forward([&](const Info& info) {
239                 if (pred(info)) {
240                     result = info;
241                     return Progress::Break;
242                 }
243                 return Progress::Continue;
244             });
245             return result;
246         }
247     private:
248     
249         Progress forwardRecursive(const Token* tok, Info info, const std::function<PathAnalysis::Progress(const Info&)>& f) const;
250         Progress forwardRange(const Token* startToken, const Token* endToken, Info info, const std::function<Progress(const Info&)>& f) const;
251     
252         static const Scope* findOuterScope(const Scope * scope);
253     
254         static std::pair<bool, bool> checkCond(const Token * tok, bool& known);
255     };
