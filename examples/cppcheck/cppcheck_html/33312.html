23     ScratchPad::ScratchPad(MainWindow& mainWindow)
24         : QDialog(&mainWindow)
25         , mMainWindow(mainWindow)
26     {
27         mUI.setupUi(this);
28     
29         connect(mUI.mCheckButton, &QPushButton::clicked, this, &ScratchPad::checkButtonClicked);
30     }
31     
32     void ScratchPad::checkButtonClicked()
33     {
34         QString filename = mUI.lineEdit->text();
35         if (filename.isEmpty())
36             filename = "test.cpp";
37         mMainWindow.analyzeCode(mUI.plainTextEdit->toPlainText(), filename);
38     }
