62     void CheckThread::run()
63     {
64         mState = Running;
65     
66         if (!mFiles.isEmpty() || mAnalyseWholeProgram) {
67             mAnalyseWholeProgram = false;
68             qDebug() << "Whole program analysis";
69             const std::string &buildDir = mCppcheck.settings().buildDir;
70             if (!buildDir.empty()) {
71                 std::map<std::string,std::size_t> files2;
72                 for (QString file : mFiles)
73                     files2[file.toStdString()] = 0;
74                 mCppcheck.analyseWholeProgram(buildDir, files2);
75             }
76             mFiles.clear();
77             emit done();
78             return;
79         }
80     
81         QString file = mResult.getNextFile();
82         while (!file.isEmpty() && mState == Running) {
83             qDebug() << "Checking file" << file;
84             mCppcheck.check(file.toStdString());
85             runAddonsAndTools(nullptr, file);
86             emit fileChecked(file);
87     
88             if (mState == Running)
89                 file = mResult.getNextFile();
90         }
91     
92         ImportProject::FileSettings fileSettings = mResult.getNextFileSettings();
93         while (!fileSettings.filename.empty() && mState == Running) {
94             file = QString::fromStdString(fileSettings.filename);
95             qDebug() << "Checking file" << file;
96             mCppcheck.check(fileSettings);
97             runAddonsAndTools(&fileSettings, QString::fromStdString(fileSettings.filename));
98             emit fileChecked(file);
99     
100             if (mState == Running)
101                 fileSettings = mResult.getNextFileSettings();
102         }
103     
104         if (mState == Running)
105             mState = Ready;
106         else
107             mState = Stopped;
108     
109         emit done();
110     }

112     void CheckThread::runAddonsAndTools(const ImportProject::FileSettings *fileSettings, const QString &fileName)
113     {
114         QString dumpFile;
115     
116         foreach (const QString addon, mAddonsAndTools) {
117             if (addon == CLANG_ANALYZER || addon == CLANG_TIDY) {
118                 if (!fileSettings)
119                     continue;
120     
121                 if (!fileSettings->cfg.empty() && fileSettings->cfg.compare(0,5,"Debug") != 0)
122                     continue;
123     
124                 QStringList args;
125                 for (std::list<std::string>::const_iterator I = fileSettings->includePaths.begin(); I != fileSettings->includePaths.end(); ++I)
126                     args << ("-I" + QString::fromStdString(*I));
127                 for (std::list<std::string>::const_iterator i = fileSettings->systemIncludePaths.begin(); i != fileSettings->systemIncludePaths.end(); ++i)
128                     args << "-isystem" << QString::fromStdString(*i);
129                 foreach (QString D, QString::fromStdString(fileSettings->defines).split(";")) {
130                     args << ("-D" + D);
131                 }
132                 foreach (const std::string& U, fileSettings->undefs) {
133                     args << QString::fromStdString("-U" + U);
134                 }
135     
136                 const QString clangPath = CheckThread::clangTidyCmd();
137                 if (!clangPath.isEmpty()) {
138                     QDir dir(clangPath + "/../lib/clang");
139                     foreach (QString ver, dir.entryList()) {
140                         QString includePath = dir.absolutePath() + '/' + ver + "/include";
141                         if (ver[0] != '.' && QDir(includePath).exists()) {
142                             args << "-isystem" << includePath;
143                             break;
144                         }
145                     }
146                 }
147     
148     #ifdef Q_OS_WIN
149                 // To create compile_commands.json in windows see:
150                 // https://bitsmaker.gitlab.io/post/clang-tidy-from-vs2015/
151     
152                 foreach (QString includePath, mClangIncludePaths) {
153                     if (!includePath.isEmpty()) {
154                         includePath.replace("\\", "/");
155                         args << "-isystem" << includePath.trimmed();
156                     }
157                 }
158     
159                 args << "-U__STDC__" << "-fno-ms-compatibility";
160     #endif
161     
162                 if (!fileSettings->standard.empty())
163                     args << ("-std=" + QString::fromStdString(fileSettings->standard));
164                 else {
165                     switch (mCppcheck.settings().standards.cpp) {
166                     case Standards::CPP03:
167                         args << "-std=c++03";
168                         break;
169                     case Standards::CPP11:
170                         args << "-std=c++11";
171                         break;
172                     case Standards::CPP14:
173                         args << "-std=c++14";
174                         break;
175                     case Standards::CPP17:
176                         args << "-std=c++17";
177                         break;
178                     case Standards::CPP20:
179                         args << "-std=c++20";
180                         break;
181                     };
182                 }
183     
184                 QString analyzerInfoFile;
185     
186                 const std::string &buildDir = mCppcheck.settings().buildDir;
187                 if (!buildDir.empty()) {
188                     analyzerInfoFile = QString::fromStdString(AnalyzerInformation::getAnalyzerInfoFile(buildDir, fileSettings->filename, fileSettings->cfg));
189     
190                     QStringList args2(args);
191                     args2.insert(0,"-E");
192                     args2 << fileName;
193                     QProcess process;
194                     process.start(clangCmd(),args2);
195                     process.waitForFinished();
196                     const QByteArray &ba = process.readAllStandardOutput();
197                     const quint16 chksum = qChecksum(ba.data(), ba.length());
198     
199                     QFile f1(analyzerInfoFile + '.' + addon + "-E");
200                     if (f1.open(QIODevice::ReadOnly | QIODevice::Text)) {
201                         QTextStream in1(&f1);
202                         const quint16 oldchksum = in1.readAll().toInt();
203                         if (oldchksum == chksum) {
204                             QFile f2(analyzerInfoFile + '.' + addon + "-results");
205                             if (f2.open(QIODevice::ReadOnly | QIODevice::Text)) {
206                                 QTextStream in2(&f2);
207                                 parseClangErrors(addon, fileName, in2.readAll());
208                                 continue;
209                             }
210                         }
211                         f1.close();
212                     }
213                     f1.open(QIODevice::WriteOnly | QIODevice::Text);
214                     QTextStream out1(&f1);
215                     out1 << chksum;
216     
217                     QFile::remove(analyzerInfoFile + '.' + addon + "-results");
218                 }
219     
220                 if (addon == CLANG_ANALYZER) {
221                     /*
222                     // Using clang
223                     args.insert(0,"--analyze");
224                     args.insert(1, "-Xanalyzer");
225                     args.insert(2, "-analyzer-output=text");
226                     args << fileName;
227                     */
228                     // Using clang-tidy
229                     args.insert(0,"-checks=-*,clang-analyzer-*");
230                     args.insert(1, fileName);
231                     args.insert(2, "--");
232                 } else {
233                     args.insert(0,"-checks=*,-clang-analyzer-*,-llvm*");
234                     args.insert(1, fileName);
235                     args.insert(2, "--");
236                 }
237     
238                 {
239                     const QString cmd(clangTidyCmd());
240                     QString debug(cmd.contains(" ") ? ('\"' + cmd + '\"') : cmd);
241                     foreach (QString arg, args) {
242                         if (arg.contains(" "))
243                             debug += " \"" + arg + '\"';
244                         else
245                             debug += ' ' + arg;
246                     }
247                     qDebug() << debug;
248     
249                     if (!analyzerInfoFile.isEmpty()) {
250                         QFile f(analyzerInfoFile + '.' + addon + "-cmd");
251                         if (f.open(QIODevice::WriteOnly | QIODevice::Text)) {
252                             QTextStream out(&f);
253                             out << debug;
254                         }
255                     }
256                 }
257     
258                 QProcess process;
259                 process.start(clangTidyCmd(), args);
260                 process.waitForFinished(600*1000);
261                 const QString errout(process.readAllStandardOutput() + "\n\n\n" + process.readAllStandardError());
262                 if (!analyzerInfoFile.isEmpty()) {
263                     QFile f(analyzerInfoFile + '.' + addon + "-results");
264                     if (f.open(QIODevice::WriteOnly | QIODevice::Text)) {
265                         QTextStream out(&f);
266                         out << errout;
267                     }
268                 }
269     
270                 parseClangErrors(addon, fileName, errout);
271             } else {
272                 const QString python = CheckThread::pythonCmd();
273                 if (python.isEmpty())
274                     continue;
275     
276                 const QString addonFilePath = CheckThread::getAddonFilePath(mDataDir, addon + ".py");
277                 if (addonFilePath.isEmpty())
278                     continue;
279     
280                 if (dumpFile.isEmpty()) {
281                     const std::string buildDir = mCppcheck.settings().buildDir;
282                     mCppcheck.settings().buildDir.clear();
283                     mCppcheck.settings().dump = true;
284                     if (!buildDir.empty()) {
285                         mCppcheck.settings().dumpFile = AnalyzerInformation::getAnalyzerInfoFile(buildDir, fileName.toStdString(), fileSettings ? fileSettings->cfg : std::string()) + ".dump";
286                         dumpFile = QString::fromStdString(mCppcheck.settings().dumpFile);
287                     } else {
288                         dumpFile = fileName + ".dump";
289                     }
290                     if (fileSettings)
291                         mCppcheck.check(*fileSettings);
292                     else
293                         mCppcheck.check(fileName.toStdString());
294                     mCppcheck.settings().dump = false;
295                     mCppcheck.settings().dumpFile.clear();
296                     mCppcheck.settings().buildDir = buildDir;
297                 }
298     
299                 QStringList args;
300                 args << addonFilePath << "--cli" << dumpFile;
301                 if (addon == "misra" && !mMisraFile.isEmpty() && QFileInfo(mMisraFile).exists()) {
302                     if (mMisraFile.endsWith(".pdf", Qt::CaseInsensitive))
303                         args << "--misra-pdf=" + mMisraFile;
304                     else
305                         args << "--rule-texts=" + mMisraFile;
306                 }
307                 qDebug() << python << args;
308     
309                 QProcess process;
310                 QProcessEnvironment env = process.processEnvironment();
311                 if (!env.contains("PYTHONHOME") && !python.startsWith("python")) {
312                     env.insert("PYTHONHOME", QFileInfo(python).canonicalPath());
313                     process.setProcessEnvironment(env);
314                 }
315                 process.start(python, args);
316                 process.waitForFinished();
317                 const QString output(process.readAllStandardOutput());
318                 QFile f(dumpFile + '-' + addon + "-results");
319                 if (f.open(QIODevice::WriteOnly | QIODevice::Text)) {
320                     QTextStream out(&f);
321                     out << output;
322                     f.close();
323                 }
324                 parseAddonErrors(output, addon);
325             }
326         }
327     }

335     void CheckThread::parseAddonErrors(QString err, const QString &tool)
336     {
337         Q_UNUSED(tool);
338         QTextStream in(&err, QIODevice::ReadOnly);
339         while (!in.atEnd()) {
340             QString line = in.readLine();
341             if (!line.startsWith("{"))
342                 continue;
343     
344             const QJsonDocument doc = QJsonDocument::fromJson(line.toLocal8Bit());
345             const QJsonObject obj = doc.object();
346     
347             /*
348             msg = { 'file': location.file,
349                     'linenr': location.linenr,
350                     'column': location.column,
351                     'severity': severity,
352                     'message': message,
353                     'addon': addon,
354                     'errorId': errorId,
355                     'extra': extra}
356             */
357     
358             const std::string &filename = obj["file"].toString().toStdString();
359             const int lineNumber = obj["linenr"].toInt();
360             const int column = obj["column"].toInt();
361             const std::string severity = obj["severity"].toString().toStdString();
362             const std::string message = obj["message"].toString().toStdString();
363             const std::string id = (obj["addon"].toString() + "-" + obj["errorId"].toString()).toStdString();
364     
365             std::list<ErrorLogger::ErrorMessage::FileLocation> callstack;
366             callstack.push_back(ErrorLogger::ErrorMessage::FileLocation(filename, lineNumber, column));
367             ErrorLogger::ErrorMessage errmsg(callstack, filename, Severity::fromString(severity), message, id, false);
368     
369             if (isSuppressed(errmsg.toSuppressionsErrorMessage()))
370                 continue;
371     
372             mResult.reportErr(errmsg);
373         }
374     }

376     void CheckThread::parseClangErrors(const QString &tool, const QString &file0, QString err)
377     {
378         QList<ErrorItem> errorItems;
379         ErrorItem errorItem;
380         QRegExp r1("(.+):([0-9]+):([0-9]+): (note|warning|error|fatal error): (.*)");
381         QRegExp r2("(.*)\\[([a-zA-Z0-9\\-_\\.]+)\\]");
382         QTextStream in(&err, QIODevice::ReadOnly);
383         while (!in.atEnd()) {
384             QString line = in.readLine();
385     
386             if (line.startsWith("Assertion failed:")) {
387                 ErrorItem e;
388                 e.errorPath.append(QErrorPathItem());
389                 e.errorPath.last().file   = file0;
390                 e.errorPath.last().line   = 1;
391                 e.errorPath.last().column = 1;
392                 e.errorId = tool + "-internal-error";
393                 e.file0 = file0;
394                 e.message = line;
395                 e.severity = Severity::information;
396                 errorItems.append(e);
397                 continue;
398             }
399     
400             if (!r1.exactMatch(line))
401                 continue;
402             if (r1.cap(4) != "note") {
403                 errorItems.append(errorItem);
404                 errorItem = ErrorItem();
405                 errorItem.file0 = r1.cap(1);
406             }
407     
408             errorItem.errorPath.append(QErrorPathItem());
409             errorItem.errorPath.last().file = r1.cap(1);
410             errorItem.errorPath.last().line = r1.cap(2).toInt();
411             errorItem.errorPath.last().column = r1.cap(3).toInt();
412             if (r1.cap(4) == "warning")
413                 errorItem.severity = Severity::SeverityType::warning;
414             else if (r1.cap(4) == "error" || r1.cap(4) == "fatal error")
415                 errorItem.severity = Severity::SeverityType::error;
416     
417             QString message,id;
418             if (r2.exactMatch(r1.cap(5))) {
419                 message = r2.cap(1);
420                 const QString id1(r2.cap(2));
421                 if (id1.startsWith("clang"))
422                     id = id1;
423                 else
424                     id = tool + '-' + r2.cap(2);
425                 if (tool == CLANG_TIDY) {
426                     if (id1.startsWith("performance"))
427                         errorItem.severity = Severity::SeverityType::performance;
428                     else if (id1.startsWith("portability"))
429                         errorItem.severity = Severity::SeverityType::portability;
430                     else if (id1.startsWith("cert") || (id1.startsWith("misc") && !id1.contains("unused")))
431                         errorItem.severity = Severity::SeverityType::warning;
432                     else
433                         errorItem.severity = Severity::SeverityType::style;
434                 }
435             } else {
436                 message = r1.cap(5);
437                 id = CLANG_ANALYZER;
438             }
439     
440             if (errorItem.errorPath.size() == 1) {
441                 errorItem.message = message;
442                 errorItem.errorId = id;
443             }
444     
445             errorItem.errorPath.last().info = message;
446         }
447         errorItems.append(errorItem);
448     
449         foreach (const ErrorItem &e, errorItems) {
450             if (e.errorPath.isEmpty())
451                 continue;
452             Suppressions::ErrorMessage errorMessage;
453             errorMessage.setFileName(e.errorPath.back().file.toStdString());
454             errorMessage.lineNumber = e.errorPath.back().line;
455             errorMessage.errorId = e.errorId.toStdString();
456             errorMessage.symbolNames = e.symbolNames.toStdString();
457     
458             if (isSuppressed(errorMessage))
459                 continue;
460     
461             std::list<ErrorLogger::ErrorMessage::FileLocation> callstack;
462             foreach (const QErrorPathItem &path, e.errorPath) {
463                 callstack.push_back(ErrorLogger::ErrorMessage::FileLocation(path.file.toStdString(), path.info.toStdString(), path.line, path.column));
464             }
465             const std::string f0 = file0.toStdString();
466             const std::string msg = e.message.toStdString();
467             const std::string id = e.errorId.toStdString();
468             ErrorLogger::ErrorMessage errmsg(callstack, f0, e.severity, msg, id, false);
469             mResult.reportErr(errmsg);
470         }
471     }

473     bool CheckThread::isSuppressed(const Suppressions::ErrorMessage &errorMessage) const
474     {
475         foreach (const Suppressions::Suppression &suppression, mSuppressions) {
476             if (suppression.isSuppressed(errorMessage))
477                 return true;
478         }
479         return false;
480     }

482     QString CheckThread::clangCmd()
483     {
484         QString path = QSettings().value(SETTINGS_CLANG_PATH,QString()).toString();
485         if (!path.isEmpty())
486             path += '/';
487         path += "clang";
488     #ifdef Q_OS_WIN
489         path += ".exe";
490     #endif
491     
492         QProcess process;
493         process.start(path, QStringList() << "--version");
494         process.waitForFinished();
495         if (process.exitCode() == 0)
496             return path;
497     
498     #ifdef Q_OS_WIN
499         // Try to autodetect clang
500         if (QFileInfo("C:/Program Files/LLVM/bin/clang.exe").exists())
501             return "C:/Program Files/LLVM/bin/clang.exe";
502     #endif
503     
504         return QString();
505     }

552     QString CheckThread::getAddonFilePath(const QString &dataDir, const QString &addonFile)
553     {
554         const QStringList paths = QStringList() << "/" << "/addons/" << "/../addons/";
555     
556         if (!dataDir.isEmpty()) {
557             foreach (const QString p, paths) {
558                 const QString filePath(dataDir + p + addonFile);
559                 if (QFileInfo(filePath).exists())
560                     return filePath;
561             }
562         }
563     
564         const QString appPath = QApplication::applicationDirPath();
565         foreach (const QString p, paths) {
566             const QString filePath(appPath + p + addonFile);
567             if (QFileInfo(filePath).exists())
568                 return filePath;
569         }
570     
571         return QString();
572     }
